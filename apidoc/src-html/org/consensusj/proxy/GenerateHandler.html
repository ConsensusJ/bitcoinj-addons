001package org.consensusj.proxy;
002
003import org.consensusj.jsonrpc.JsonRpcRequest;
004import org.consensusj.jsonrpc.ratpack.JsonRpcClient;
005import ratpack.handling.Context;
006import ratpack.handling.Handler;
007
008import javax.inject.Inject;
009import javax.inject.Singleton;
010import java.util.Arrays;
011
012import static ratpack.jackson.Jackson.json;
013
014/**
015 * Get requests will generate a block (regtest only)
016 */
017@Singleton
018public class GenerateHandler implements Handler {
019    private final JsonRpcClient rpcClient;
020
021    @Inject
022    public GenerateHandler(JsonRpcClient jsonRpcClient) {
023        rpcClient = jsonRpcClient;
024    }
025
026    @Override
027    public void handle(Context ctx) {
028        JsonRpcRequest rpcReq = new JsonRpcRequest("setgenerate", Arrays.asList(true, 1));
029        rpcClient.call(rpcReq).then(rpcResponse -> ctx.render(json(rpcResponse)));
030    }
031}
