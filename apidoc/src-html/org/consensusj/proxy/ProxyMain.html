001package org.consensusj.proxy;
002
003import com.msgilligan.bitcoinj.rpc.RpcConfig;
004import org.consensusj.jsonrpc.ratpack.RpcProxyHandler;
005import ratpack.guice.Guice;
006import ratpack.server.BaseDir;
007import ratpack.server.RatpackServer;
008import ratpack.server.ServerConfig;
009
010/**
011 * Main class for a Bitcoin RPC reverse proxy server
012 */
013public class ProxyMain {
014    public static void main(String... args) throws Exception {
015        ServerConfig serverConfig = ServerConfig.of(config -> config
016                .port(5050)
017                .baseDir(BaseDir.find())
018                .json("proxy-config.json")
019                .require("/rpcclient", RpcConfig.class)
020        );
021        RatpackServer.start (server -> server
022            .serverConfig(serverConfig)
023            .registry(Guice.registry(b -> b.
024                    moduleConfig(BitcoinRpcProxyModule.class,
025                            serverConfig.get("/rpcclient", RpcConfig.class))))
026            .handlers(chain -> chain
027                    .post("rpc", RpcProxyHandler.class)
028                    .get("status", ChainStatusHandler.class)
029                    .get("gen", GenerateHandler.class)
030                    .get(ctx -> ctx.getResponse().send("Hello world! (Not RPC)"))
031                )
032        );
033    }
034}
