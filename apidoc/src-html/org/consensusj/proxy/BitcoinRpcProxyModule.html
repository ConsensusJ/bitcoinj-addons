001package org.consensusj.proxy;
002
003import org.consensusj.jsonrpc.ratpack.JsonRpcClientModule;
004import org.consensusj.jsonrpc.ratpack.RpcProxyHandler;
005
006/**
007 * Guice Module to create handlers and client for JSON-RPC Proxy
008 * Adds a few handlers to the JsonRpcClientModule
009 */
010public class BitcoinRpcProxyModule extends JsonRpcClientModule {
011
012    @Override
013    protected void configure() {
014        super.configure();
015        bind(RpcProxyHandler.class);
016        bind(ChainStatusHandler.class);
017        bind(GenerateHandler.class);
018    }
019}
