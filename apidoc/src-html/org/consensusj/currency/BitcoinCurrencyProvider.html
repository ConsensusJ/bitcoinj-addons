001package org.consensusj.currency;
002
003import org.javamoney.moneta.CurrencyUnitBuilder;
004import org.javamoney.moneta.spi.base.BaseCurrencyProviderSpi;
005
006import javax.money.CurrencyContext;
007import javax.money.CurrencyContextBuilder;
008import javax.money.CurrencyQuery;
009import javax.money.CurrencyUnit;
010import javax.money.spi.CurrencyProviderSpi;
011import java.util.Collections;
012import java.util.Set;
013
014/**
015 *  A BitcoinCurrencyProvider based on work in the javamoney-shelter.
016 *
017 * @author Sean Gilligan
018 * @author Werner Keil
019 */
020public class BitcoinCurrencyProvider extends BaseCurrencyProviderSpi implements CurrencyProviderSpi {
021    private final static int bitcoinFractionDigits = 8;
022
023    private static final CurrencyContext CONTEXT = CurrencyContextBuilder.of("BitcoinCurrencyContextProvider")
024            .build();
025
026    private final Set<CurrencyUnit> bitcoinSet;
027
028    public BitcoinCurrencyProvider() {
029        CurrencyUnit btcUnit = CurrencyUnitBuilder.of("BTC", CONTEXT)
030                .setDefaultFractionDigits(bitcoinFractionDigits)
031                .build();
032        bitcoinSet = Collections.singleton(btcUnit);
033    }
034    
035    /**
036     * Return a {@link CurrencyUnit} instances matching the given
037     * {@link javax.money.CurrencyQuery}.
038     *
039     * @param query the {@link javax.money.CurrencyQuery} containing the parameters determining the query. not null.
040     * @return the corresponding {@link CurrencyUnit}s matching, never null.
041     */
042    @Override
043    public Set<CurrencyUnit> getCurrencies(CurrencyQuery query){
044        // Query for currencyCode BTC or default query returns bitcoinSet else emptySet.
045        return (query.getCurrencyCodes().contains("BTC") ||
046                query.getCurrencyCodes().isEmpty()) ? bitcoinSet : Collections.emptySet();
047    }
048}
