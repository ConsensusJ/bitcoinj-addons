003import com.google.common.util.concurrent.ListenableFuture;
004import org.bitcoinj.core.Block;
005import org.bitcoinj.core.PeerGroup;
006import org.bitcoinj.core.Sha256Hash;
007import org.bitcoinj.core.Transaction;
008
009import java.util.List;
010import java.util.concurrent.CompletableFuture;
011import java.util.concurrent.CompletionException;
012
013//import static net.javacrumbs.futureconverter.java8guava.FutureConverter.toCompletableFuture;
014
015/**
016 *
017 */
018public class BtcrResolver {
019    private final PeerGroup peerGroup;
020
021    public BtcrResolver(PeerGroup peerGroup) {
022        this.peerGroup = peerGroup;
023    }
024
025    public CompletableFuture<Block> getBlock(Sha256Hash blockHash) {
026        ListenableFuture<Block> listenableFuture = peerGroup.getDownloadPeer().getBlock(blockHash);
027//        return toCompletableFuture(listenableFuture);
028        return null;
029    }
030    
031    public CompletableFuture<Transaction> resolve(Sha256Hash blockHash, int txIndex) {
032        return getBlock(blockHash)
033                .thenApply(block -> {
034                    List<Transaction> txList = block.getTransactions();
035                    if (txList == null) {
036                        throw new CompletionException(new Exception("Transaction not found: Empty block"));
037                    }
038                    Transaction tx = block.getTransactions().get(txIndex);
039                    if (tx == null) {
040                        throw new CompletionException(new Exception("Transaction not found (by index)"));
041                    }
042                    return tx;
043                });
044    }
045}
