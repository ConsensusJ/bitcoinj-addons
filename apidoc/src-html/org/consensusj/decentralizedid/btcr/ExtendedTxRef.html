001package org.consensusj.decentralizedid.btcr;
002
003import org.bitcoinj.core.Bech32;
004
005/**
006 * A Bech32 representation of a Bitcoin transaction output
007 */
008public class ExtendedTxRef {
009    final Bech32.Bech32Data bech32;
010
011    public ExtendedTxRef(String txRef) {
012        bech32 = Bech32.decode(standardize(txRef));
013    }
014
015    static ExtendedTxRef of(String extTxRefString) {
016        return new ExtendedTxRef(extTxRefString);
017    }
018
019    public String toString() {
020        return Bech32.encode(bech32);
021    }
022
023    /**
024     * Add the human readable part and remove the dashes to create a standard
025     * Bech32 encoding.
026     *
027     * @param txRef An extended TxRef (Bech32 with no human
028     * @return A standard Bech32 encoding that can be parsed by bitcoinj
029     */
030    static private String standardize(String txRef) {
031        return "txtest1" + txRef.replace("-", "");
032    }
033}
