001package org.consensusj.daemon.micronaut;
002
003import com.msgilligan.bitcoinj.json.pojo.BlockChainInfo;
004import com.msgilligan.bitcoinj.json.pojo.BlockInfo;
005import io.micronaut.context.annotation.Context;
006import io.micronaut.http.MediaType;
007import io.micronaut.http.annotation.Controller;
008import io.micronaut.http.annotation.Get;
009import io.micronaut.http.annotation.PathVariable;
010import org.bitcoinj.core.Sha256Hash;
011import org.consensusj.bitcoin.services.WalletAppKitService;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015
016/**
017 * Embryonic implementation of Bitcoin Core REST API. Incomplete and incorrect at ths point.
018 */
019@Controller("/rest")
020@Context
021public class BitcoinRestController {
022    private static final Logger log = LoggerFactory.getLogger(BitcoinRestController.class);
023    private final WalletAppKitService walletAppKitService;
024
025    public BitcoinRestController(WalletAppKitService walletAppKitService) {
026        log.info("Constructing JsonRpcController");
027        this.walletAppKitService = walletAppKitService;
028    }
029
030    @Get(uri="/block/{hash}.json", produces = MediaType.APPLICATION_JSON)
031    public BlockInfo block(@PathVariable String hash) {
032        log.info("chaininfo REST GET call");
033        return walletAppKitService.getBlockInfo(Sha256Hash.wrap(hash), BlockInfo.IncludeTxFlag.YES);
034    }
035
036    @Get(uri="/block/notxdetails/{hash}.json", produces = MediaType.APPLICATION_JSON)
037    public BlockInfo blockNoTxDetails(@PathVariable String hash) {
038        log.info("chaininfo REST GET call");
039        return walletAppKitService.getBlockInfo(Sha256Hash.wrap(hash),  BlockInfo.IncludeTxFlag.IDONLY);
040    }
041
042    @Get("/chaininfo.json")
043    public BlockChainInfo chaininfo() {
044        log.info("chaininfo REST GET call");
045        return walletAppKitService.getblockchaininfo();
046    }
047
048}
