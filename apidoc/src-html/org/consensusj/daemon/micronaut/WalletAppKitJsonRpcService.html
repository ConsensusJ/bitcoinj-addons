001package org.consensusj.daemon.micronaut;
002
003import org.consensusj.bitcoin.services.WalletAppKitService;
004import org.consensusj.jsonrpc.introspection.DelegatingJsonRpcService;
005import org.consensusj.jsonrpc.introspection.JsonRpcServiceWrapper;
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009import javax.inject.Singleton;
010import java.lang.reflect.Method;
011import java.util.Map;
012
013/**
014 * JSON-RPC wrapper for WalletAppKitService
015 */
016@Singleton
017public class WalletAppKitJsonRpcService extends DelegatingJsonRpcService {
018    private static Logger log = LoggerFactory.getLogger(WalletAppKitJsonRpcService.class);
019    private static final Map<String, Method> methods = JsonRpcServiceWrapper.reflect(WalletAppKitService.class);
020
021    public WalletAppKitJsonRpcService(WalletAppKitService service) {
022        super(methods, service);
023        log.info("Constructing WalletAppKitJsonRpcService");
024    }
025}
