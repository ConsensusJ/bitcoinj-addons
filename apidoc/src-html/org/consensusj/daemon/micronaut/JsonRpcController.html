001package org.consensusj.daemon.micronaut;
002
003import com.fasterxml.jackson.databind.PropertyNamingStrategy;
004import com.msgilligan.bitcoinj.json.pojo.BlockInfo;
005import com.msgilligan.bitcoinj.json.pojo.ServerInfo;
006import io.micronaut.context.annotation.Context;
007import io.micronaut.core.annotation.TypeHint;
008import io.micronaut.http.MediaType;
009import io.micronaut.http.annotation.Body;
010import io.micronaut.http.annotation.Controller;
011import io.micronaut.http.annotation.Post;
012import org.bitcoinj.core.Sha256Hash;
013import org.consensusj.bitcoin.services.WalletAppKitService;
014import org.consensusj.jsonrpc.JsonRpcRequest;
015import org.consensusj.jsonrpc.JsonRpcResponse;
016import org.consensusj.jsonrpc.JsonRpcService;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020import java.util.ArrayList;
021import java.util.HashSet;
022import java.util.LinkedHashMap;
023import java.util.concurrent.CompletableFuture;
024
025/**
026 * Create a REST endpoint from a JsonRpcService
027 * Here we're using Micronaut annotations to wrap the JsonRpcService
028 * in an HTTP environment and to serialize JSON to and from Java POJOs.
029 */
030@TypeHint(
031        value = {
032                PropertyNamingStrategy.UpperCamelCaseStrategy.class,
033                ArrayList.class,
034                LinkedHashMap.class,
035                HashSet.class,
036                JsonRpcRequest.class,
037                JsonRpcResponse.class,
038                ServerInfo.class,
039                BlockInfo.class,
040                Sha256Hash.class,
041                WalletAppKitService.class
042        },
043        accessType = {TypeHint.AccessType.ALL_DECLARED_CONSTRUCTORS, TypeHint.AccessType.ALL_DECLARED_METHODS}
044)
045@Controller("/rpc")
046@Context
047public class JsonRpcController {
048    private static Logger log = LoggerFactory.getLogger(JsonRpcController.class);
049    private final JsonRpcService jsonRpcService;
050
051    public JsonRpcController(WalletAppKitJsonRpcService walletAppKitService) {
052        log.info("Constructing JsonRpcController");
053        jsonRpcService = walletAppKitService;
054    }
055
056    @Post(produces = MediaType.APPLICATION_JSON)
057    public CompletableFuture<JsonRpcResponse<Object>> index(@Body JsonRpcRequest req) {
058        log.info("JSON-RPC call: {}", req.getMethod());
059        return jsonRpcService.call(req);
060    }
061
062}
