001package org.consensusj.daemon.micronaut;
002
003import com.fasterxml.jackson.databind.Module;
004import com.msgilligan.bitcoinj.json.conversion.RpcServerModule;
005import io.micronaut.context.annotation.Factory;
006import org.bitcoinj.core.Context;
007import org.bitcoinj.core.NetworkParameters;
008import org.bitcoinj.kits.WalletAppKit;
009import org.bitcoinj.params.MainNetParams;
010import org.consensusj.bitcoin.services.WalletAppKitService;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014import javax.inject.Singleton;
015import java.io.File;
016
017/**
018 * Initialize bitcoinj components, etc.
019 */
020@Factory
021public class BitcoinFactory {
022    private static Logger log = LoggerFactory.getLogger(BitcoinFactory.class);
023
024    @Singleton
025    public NetworkParameters networkParameters() {
026        log.info("Returning NetworkParameters bean");
027        return MainNetParams.get();
028    }
029
030    @Singleton
031    public Context getContext(NetworkParameters params) {
032        log.info("Returning bitcoinj Context bean");
033        return new Context(params);
034    }
035
036    @Singleton
037    public WalletAppKit getKit(NetworkParameters params) throws Exception {
038        log.info("Returning WalletAppKit bean");
039        // TODO: make File(".") and filePrefix configurable
040        File directory = new File(".");
041        String filePrefix = "BitcoinJDaemon";
042
043        return new WalletAppKit(params, directory, filePrefix);
044    }
045
046    @Singleton
047    public WalletAppKitService walletAppKitService(NetworkParameters params, Context context, WalletAppKit kit) {
048        log.info("Returning WalletAppKitService bean");
049        WalletAppKitService service = new WalletAppKitService(params, context, kit);
050        service.start();
051        return service;
052    }
053
054    @Singleton
055    public Module jacksonModule() {
056        return new RpcServerModule(null);
057    }
058}
