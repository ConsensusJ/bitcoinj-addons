001package org.consensusj.daemon.micronaut;
002
003import io.micronaut.context.annotation.Bean;
004import io.micronaut.context.annotation.Factory;
005import org.bitcoinj.core.Context;
006import org.bitcoinj.core.NetworkParameters;
007import org.bitcoinj.kits.WalletAppKit;
008import org.bitcoinj.params.MainNetParams;
009import org.consensusj.bitcoin.services.WalletAppKitService;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013import javax.inject.Singleton;
014import java.io.File;
015
016/**
017 * Initialize bitcoinj components, etc.
018 */
019@Factory
020public class BitcoinFactory {
021    private static Logger log = LoggerFactory.getLogger(BitcoinFactory.class);
022
023    @Bean
024    @Singleton
025    public NetworkParameters networkParameters() {
026        log.info("Returning NetworkParameters bean");
027        return MainNetParams.get();
028    }
029
030    @Bean
031    @Singleton
032    public Context getContext(NetworkParameters params) {
033        log.info("Returning bitcoinj Context bean");
034        return new Context(params);
035    }
036
037    @Bean
038    @Singleton
039    public WalletAppKit getKit(NetworkParameters params) throws Exception {
040        log.info("Returning WalletAppKit bean");
041        // TODO: make File(".") and filePrefix configurable
042        File directory = new File(".");
043        String filePrefix = "BitcoinJDaemon";
044
045        return new WalletAppKit(params, directory, filePrefix);
046    }
047
048    @Bean
049    @io.micronaut.context.annotation.Context
050    public WalletAppKitService walletAppKitService(NetworkParameters params, Context context, WalletAppKit kit) {
051        log.info("Returning WalletAppKitService bean");
052        WalletAppKitService service = new WalletAppKitService(params, context, kit);
053        service.start();
054        return service;
055    }
056}
