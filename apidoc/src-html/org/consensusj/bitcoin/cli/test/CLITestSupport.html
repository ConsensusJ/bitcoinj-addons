001package org.consensusj.bitcoin.cli.test;
002
003import com.msgilligan.bitcoinj.rpc.test.TestServers;
004import org.consensusj.bitcoin.cli.CliCommand;
005
006import java.io.ByteArrayInputStream;
007import java.io.ByteArrayOutputStream;
008import java.io.InputStream;
009import java.io.PrintStream;
010import java.io.UnsupportedEncodingException;
011import java.nio.charset.StandardCharsets;
012
013/**
014 *  Support functions for testing command-line tools
015 */
016public interface CLITestSupport {
017    String rpcUser = TestServers.getInstance().getRpcTestUser();
018    String rpcPassword = TestServers.getInstance().getRpcTestPassword();
019
020    default String[] parseCommandLine(String line) {
021        String[] args = line.split(" ");     // (Overly?) simple parsing of string into args[]
022        return args;
023    }
024
025    /**
026     * Run a command and capture status and output
027     *
028     * @param command Command object instance to run
029     * @return Object containing status, stdout, stderr
030     */
031    default CLICommandResult runCommand(CliCommand command) throws UnsupportedEncodingException {
032        // Setup CommandResult to capture status and streams
033        CLICommandResult result = new CLICommandResult();
034        InputStream is = new ByteArrayInputStream(new byte[0]);
035        ByteArrayOutputStream bos = new ByteArrayOutputStream();
036        PrintStream cos = new PrintStream(bos);
037        ByteArrayOutputStream bes = new ByteArrayOutputStream();
038        PrintStream ces = new PrintStream(bes);
039
040        // Run the command
041        result.status = command.run(is, cos, ces);
042
043        // Put output and error streams in strings
044        String charset = StandardCharsets.UTF_8.toString();
045        result.output = bos.toString(charset);
046        result.error = bes.toString(charset);
047
048        return result;
049    }
050
051}
