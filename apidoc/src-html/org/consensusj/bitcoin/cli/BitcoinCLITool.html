001package org.consensusj.bitcoin.cli;
002
003import org.bitcoinj.core.Sha256Hash;
004import org.consensusj.jsonrpc.JsonRpcException;
005
006import java.io.IOException;
007import java.util.ArrayList;
008import java.util.List;
009
010/**
011 * An attempt at cloning the bitcoin-cli tool, but using Java and bitcoinj
012 *
013 */
014public class BitcoinCLITool extends BitcoinCliCommand {
015    public final static String commandName = "cj-bitcoin-cli";
016
017    public BitcoinCLITool(String[] args) {
018        super(commandName, new CliOptions(), args);
019    }
020
021    /**
022     * main method for bitcoinj-cli tool.
023     *
024     * See {@link CliOptions} for options and https://bitcoin.org/en/developer-reference#bitcoin-core-apis[Bitcoin Core JSON-RPC API]
025     * for the methods and parameters. Users can use `-?` to get general help or `help <command>` to get help
026     * on a specific command.
027     *
028     * @param args options, JSON-RPC method, JSON-RPC parameters
029     */
030    public static void main(String[] args) {
031        BitcoinCLITool command = new BitcoinCLITool(args);
032        Integer status = command.run();
033        System.exit(status);
034    }
035
036    public Integer runImpl() throws IOException {
037        List<String> args = line.getArgList();
038        if (args.size() == 0) {
039            printError("jsonrpc method required");
040            printHelp();
041            return(1);
042        }
043        String method = args.get(0);
044        args.remove(0); // remove method from list
045        List<Object> typedArgs = convertParameters(method, args);
046        Object result;
047        try {
048            result = client.send(method, typedArgs);
049        } catch (JsonRpcException e) {
050            e.printStackTrace();
051            return 1;
052        }
053        if (result != null) {
054            pwout.println(result.toString());
055        }
056        return 0;
057    }
058
059    /**
060     * Convert params from strings to Java types that will map to correct JSON types
061     *
062     * TODO: Make this better and complete
063     *
064     * @param method the JSON-RPC method
065     * @param params Params with String type
066     * @return Params with correct Java types for JSON
067     */
068    protected List<Object> convertParameters(String method, List<String> params) {
069        List<Object> typedParams = new ArrayList<>();
070        switch (method) {
071            case "generate":
072            case "setgenerate":
073                typedParams.add(Integer.valueOf(params.get(0)));
074                break;
075
076            case "getblockhash":
077                typedParams.add(Integer.valueOf(params.get(0)));
078                break;
079                
080            default:
081                // Default (for now) is to leave them all as strings
082                for (String string : params) {
083                    typedParams.add(string);
084                }
085
086        }
087        return typedParams;
088    }
089}
