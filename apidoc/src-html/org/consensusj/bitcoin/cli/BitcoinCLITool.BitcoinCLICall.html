027public class BitcoinCLITool extends BaseJsonRpcTool {
028    public final static String commandName = "cj-bitcoin-cli";
029    private final Options options;

031    public BitcoinCLITool() {
032        this.options = new BitcoinRpcCliOptions();
033    }
034

036    public String name() {
037        return commandName;
038    }
039
040    @Override
041    public Options options() {
042        return options;
043    }

045    /**
046     * main method for bitcoinj-cli tool.
047     *
048     * See {@link BitcoinRpcCliOptions} for options and https://bitcoin.org/en/developer-reference#bitcoin-core-apis[Bitcoin Core JSON-RPC API]
049     * for the methods and parameters. Users can use `-?` to get general help or `help <command>` to get help
050     * on a specific command.
051     *
052     * @param args options, JSON-RPC method, JSON-RPC parameters
053     */
054    public static void main(String[] args) {
055        BitcoinCLITool command = new BitcoinCLITool();
056        int status = command.run(System.out, System.err, args);
057        System.exit(status);
058    }
059
060    @Override
061    public BitcoinCLICall createCall(PrintWriter out, PrintWriter err, String... args) {
062        return new BitcoinCLICall(this, out, err, args);
063    }
064
065    /**
066     * Convert params from strings to Java types that will map to correct JSON types
067     *
068     * TODO: Make this better and complete
069     *
070     * @param method the JSON-RPC method
071     * @param params Params with String type
072     * @return Params with correct Java types for JSON
073     */
074    protected List<Object> convertParameters(String method, List<String> params) {
075        List<Object> typedParams = new ArrayList<>();
076        switch (method) {
077            case "generate":
078            case "setgenerate":
079                typedParams.add(Integer.valueOf(params.get(0)));
080                break;
081
082            case "generatetoaddress":
083                typedParams.add(Integer.valueOf(params.get(0)));
084                typedParams.add(params.get(1));
085                if (params.size() >= 3) {
086                    typedParams.add(Integer.valueOf(params.get(2)));
087                }
088                break;
089
090            case "getblockhash":
091                typedParams.add(Integer.valueOf(params.get(0)));
092                break;
093                
094            default:
095                // Default parameter conversion
096                for (String string : params) {
097                    typedParams.add(convertParam(string));
098                }
099
100        }
101        return typedParams;
102    }
103
104    public static class BitcoinCLICall extends BaseJsonRpcTool.CommonsCLICall {
105        private BitcoinClient client;
106        private final boolean rpcWait;
107
108        public BitcoinCLICall(BitcoinCLITool tool, PrintWriter out, PrintWriter err, String[] args) {
109            super(tool, out, err, args);
110            rpcWait = line.hasOption("rpcwait");
111        }
112
113        @Override
114        public BitcoinClient rpcClient() {
115            // Not threadsafe
116            // This needs work if there are ever going to be multiple clients calling this method
117            if (client == null) {
118                System.out.println("Connecting to: " + getRPCConfig().getURI());
119                RpcConfig config = getRPCConfig();
120                client = createClient(config);
121                if (rpcWait) {
122                    boolean available = false;   // Wait up to 1 hour
123                    try {
124                        available = client.waitForServer(60*60);
125                    } catch (JsonRpcException e) {
126                        rpcTool.printError(this, "JSON-RPC Exception: " + e.getMessage());
127                        throw new ToolException(1, e.getMessage());
128                    }
129                    if (!available) {
130                        rpcTool.printError(this,"Timeout error.");
131                        throw new ToolException(1, "Timeout error.");
132                    }
133                }
134            }
135            return client;
136        }
137
138        /**
139         * Override this method to customize the client implementation subclass
140         * @param config
141         * @return
142         */
143        protected BitcoinClient createClient(RpcConfig config) {
144            return new BitcoinClient( config.getNetParams(),
145                    config.getURI(),
146                    config.getUsername(),
147                    config.getPassword());
148        }
149
150        public RpcConfig getRPCConfig() {
151            RpcConfig confFileConfig = BitcoinConfFile.readDefaultConfig().getRPCConfig();
152            URI uri = getServerURI(confFileConfig.getURI());
153            String user = line.getOptionValue("rpcuser", confFileConfig.getUsername());
154            String pass = line.getOptionValue("rpcpassword", confFileConfig.getPassword());
155            NetworkParameters netParams;
156            if (line.hasOption("regtest")) {
157                netParams = RegTestParams.get();
158            } else if (line.hasOption(("testnet"))) {
159                netParams = TestNet3Params.get();
160            } else {
161                netParams = MainNetParams.get();
162            }
163            RpcConfig cfg = new RpcConfig(netParams, uri, user, pass);
164            return cfg;
165        }
166
167        private URI getServerURI(URI confFileURI) {
168            String scheme = confFileURI.getScheme();
169            String host = confFileURI.getHost();
170            int port = confFileURI.getPort();
171            String file = confFileURI.getPath();
172
173            if (line.hasOption("rpcssl")) {
174                scheme = "https";
175            }
176            if (line.hasOption("rpcconnect")) {
177                host = line.getOptionValue("rpcconnect", host);
178            }
179            if (line.hasOption("regtest"))  {
180                port = RPCPORT_REGTEST;
181            } else if (line.hasOption("testnet")) {
182                port = RPCPORT_TESTNET;
183            }
184            if (line.hasOption("rpcport")) {
185                String portString = line.getOptionValue("rpcport");
186                port = Integer.parseInt(portString);
187            }
188            URI rpcServerURI = null;
189            try {
190                rpcServerURI = new URI(scheme, null, host, port, file, null, null);
191            } catch (URISyntaxException e) {
192                // We should be careful that this never happens
193                e.printStackTrace();
194                // But if it does, throw an unchecked exception
195                throw new RuntimeException(e);
196            }
197            return rpcServerURI;
198        }
