113    /**
114     * Partial implementation of `getblock`
115     *
116     * In the case where verbosity = 0, a a block is returned without transactions (which is incorrect)
117     * For verbosity = 1, we return a BlockInfo, but some fields may be missing or incorrect, including a list of TXIDs
118     * For verbosity = 2, we are currently throwing an exception.
119     * 
120     * @param blockHashString The hash of the block we are to return (as hex)
121     * @param verbosity Specifies the format to return (currently only `1` halfway works)
122     * @return Either a `Block` or `BlockInfo` depending upon verbosity.
123     */
124    @Override
125    public Object getblock(String blockHashString, Integer verbosity) {
126        return getblock2(Sha256Hash.wrap(blockHashString), verbosity);
127    }

129    public Object getblock2(Sha256Hash blockHash, Integer verbosity) {
130        int verbosityInt = verbosity != null ? verbosity : 1;
131        switch(verbosityInt) {
132            case 0: return getBlock(blockHash);
133            case 1: return getBlockInfo(blockHash, BlockInfo.IncludeTxFlag.IDONLY);
134            case 2: return getBlockInfo(blockHash, BlockInfo.IncludeTxFlag.YES);
135            default: throw new IllegalArgumentException("Unknown verbosity parameter");
136        }
137    }

219    private Object getBlock(Sha256Hash blockHash) {
220        // TODO: This block should have transactions
221        StoredBlock storedBlock;
222        try {
223            storedBlock = getStoredBlockByHash(kit.chain(), blockHash);
224        } catch (BlockStoreException e) {
225            throw new RuntimeException(e);
226        }
227        return blockToHex(storedBlock.getHeader());
228    }
229
230    public BlockInfo getBlockInfo(Sha256Hash blockHash, BlockInfo.IncludeTxFlag includeTx) {
231        BlockInfo blockInfo;
232        try {
233            blockInfo = getBlockInfoByHash(kit.chain(), blockHash, includeTx);
234            log.info("blockinfo: {}, {}", blockInfo.hash, blockInfo.height);
235        } catch (BlockStoreException e) {
236            throw new RuntimeException(e);
237        }
238        return blockInfo;
239    }

241    private static StoredBlock getStoredBlockByHash(AbstractBlockChain blockChain, Sha256Hash blockHash) throws BlockStoreException {
242        return blockChain.getBlockStore().get(blockHash);
243    }
244
245    /**
246     * Get a BlockInfo for the specified hash
247     *
248     * @param blockChain The blockchain object to pull the data from
249     * @param blockHash The hash of the desired block
250     * @param includeTx whether to include transactions (currently must be false)
251     * @return block information (currently incomplete and untested)
252     * @throws BlockStoreException Something went wrong
253     */
254    private static BlockInfo getBlockInfoByHash(AbstractBlockChain blockChain, Sha256Hash blockHash, BlockInfo.IncludeTxFlag includeTx) throws BlockStoreException {
255        if (includeTx == BlockInfo.IncludeTxFlag.YES) {
256            throw new IllegalArgumentException("Including transactions not supported yet");
257        }
258        StoredBlock block = getStoredBlockByHash(blockChain, blockHash);
259        Block header = block.getHeader();
260        int blockHeight = block.getHeight();
261        int confirmations = blockChain.getBestChainHeight() - blockHeight;
262        log.trace("building BlockInfo for hash: {} height: {}", blockHash, blockHeight);
263        return new BlockInfo(header.getHash(),
264                confirmations,
265                header.getMessageSize(),
266                blockHeight,
267                (int) header.getVersion(),
268                header.getMerkleRoot(),
269                -1,     // Unknown number of Transactions
270                (includeTx == BlockInfo.IncludeTxFlag.IDONLY) ? hashListFromTxList(header.getTransactions()) : null,
271                (int) header.getTimeSeconds(),
272                header.getNonce(),
273                null, // TODO: Return "bits" here
274                new BigDecimal(header.getDifficultyTargetAsInteger()),  // TODO: Verify this is correct
275                block.getChainWork().toString(),
276                block.getPrev(blockChain.getBlockStore()).getHeader().getHash(),
277                null);  // TODO: Extend BlockStore to make this information retrievable
278    }
279
280    private static BlockInfo.Sha256HashList hashListFromTxList(List<Transaction> txList) {
281        if (txList == null) {
282            return null;
283        } else {
284            List<Sha256Hash> list = txList.stream()
285                    .map(Transaction::getTxId)
286                    .collect(Collectors.toList());
287            return new BlockInfo.Sha256HashList(list);
288        }
289    }

298        return HexUtil.bytesToHexString(block.bitcoinSerialize());
