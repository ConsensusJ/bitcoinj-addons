001package org.consensusj.jsonrpc;
002
003import com.fasterxml.jackson.databind.JavaType;
004import com.fasterxml.jackson.databind.ObjectMapper;
005
006import java.io.IOException;
007import java.net.URI;
008import java.util.Arrays;
009import java.util.List;
010
011/**
012 * Abstract Base class for a strongly-typed JSON-RPC client. This abstract class handles
013 * the use of Jackson to map from JSON to Java, but leaves the core `send` method as
014 * `abstract` to be implemented by subclasses allowing implementation with alternative
015 * HTTP client libraries.
016 */
017public abstract class AbstractRpcClient implements DynamicRpcMethodSupport {
018    protected final ObjectMapper mapper;
019    private final JavaType defaultType;
020
021    public AbstractRpcClient() {
022        mapper = new ObjectMapper();
023        defaultType = mapper.getTypeFactory().constructType(Object.class);
024    }
025
026    public abstract URI getServerURI();
027
028    /**
029     * Subclasses must implement this method to actually send the request
030     * @param pass:[<R>] Type of result object
031     * @param request The request to send
032     * @param responseType The response to expected (used by Jackson for conversion)
033     * @return A JSON RPC Response with `result` of type `R`
034     * @throws IOException network error
035     * @throws JsonRpcStatusException JSON RPC status error
036     */
037    protected abstract <R> JsonRpcResponse<R> sendRequestForResponse(JsonRpcRequest request, JavaType responseType) throws IOException, JsonRpcStatusException;
038
039    /**
040     * Create a JsonRpcRequest from method and parameters
041     *
042     * Currently builds JSON-RPC 1.0 request, this method can be overridden for clients
043     * that need JSON-RPC 2.0 (e.g. Ethereum)
044     *
045     * @param method name of method to call
046     * @param params parameter Java objects
047     * @return A ready-to-send JsonRpcRequest
048     */
049    protected JsonRpcRequest buildJsonRequest(String method, List<Object> params) {
050        return new JsonRpcRequest(method, params);
051    }
052
053    private <R> R sendForResult(String method, JavaType responseType, List<Object> params) throws IOException, JsonRpcStatusException {
054        JsonRpcRequest request = buildJsonRequest(method, params);
055        JsonRpcResponse<R> response = sendRequestForResponse(request, responseType);
056
057//        assert response != null;
058//        assert response.getJsonrpc() != null;
059//        assert response.getJsonrpc().equals("2.0");
060//        assert response.getId() != null;
061//        assert response.getId().equals(request.getId());
062
063        if (response.getError() != null && response.getError().getCode() != 0) {
064            throw new JsonRpcStatusException(
065                    response.getError().getMessage(),
066                    200,    // If response code wasn't 200 we couldn't be here
067                    null,
068                    response.getError().getCode(),
069                    null,
070                    response);
071        }
072        return response.getResult();
073    }
074
075    /**
076     * JSON-RPC remote method call that returns 'response.result`
077     *
078     * @param pass:[<R>] Type of result object
079     * @param method JSON RPC method call to send
080     * @param resultType desired result type as a Java class object
081     * @param params JSON RPC params
082     * @return the 'response.result' field of the JSON RPC response converted to type R
083     */
084    protected <R> R send(String method, Class<R> resultType, List<Object> params) throws IOException, JsonRpcStatusException {
085        // Construct a JavaType object so we can tell Jackson what type of result we are expecting.
086        // (We can't use R because of type erasure)
087        JavaType responseType = mapper.getTypeFactory().
088                constructParametricType(JsonRpcResponse.class, resultType);
089        return sendForResult(method, responseType, params);
090    }
091
092    /**
093     * Varargs version
094     */
095    protected <R> R send(String method, Class<R> resultType, Object... params) throws IOException, JsonRpcStatusException {
096        return send(method, resultType, Arrays.asList(params));
097    }
098
099
100    /**
101     * JSON-RPC remote method call that returns 'response.result`
102     *
103     * @param pass:[<R>] Type of result object
104     * @param method JSON RPC method call to send
105     * @param resultType desired result type as a Jackson JavaType object
106     * @param params JSON RPC params
107     * @return the 'response.result' field of the JSON RPC response converted to type R
108     */
109    protected <R> R send(String method, JavaType resultType, List<Object> params) throws IOException, JsonRpcStatusException {
110        // Construct a JavaType object so we can tell Jackson what type of result we are expecting.
111        // (We can't use R because of type erasure)
112        JavaType responseType = mapper.getTypeFactory().
113                constructParametricType(JsonRpcResponse.class, resultType);
114        return sendForResult(method, responseType, params);
115    }
116
117    /**
118     * Varargs version
119     */
120    protected <R> R send(String method, JavaType resultType, Object... params) throws IOException, JsonRpcStatusException {
121        return send(method, resultType, Arrays.asList(params));
122    }
123
124    /**
125     * Call an RPC method and return default object type.
126     *
127     * Caller should cast returned object to the correct type.
128     *
129     * Useful for:
130     * * Dynamically-dispatched JSON-RPC methods calls via Groovy subclasses
131     * * Simple (not client-side validated) command line utilities
132     * * Functional tests that need to send incorrect types to the server to test error handling
133     *
134     * @param pass:[<R>] Type of result object
135     * @param method JSON RPC method call to send
136     * @param params JSON RPC parameters as a `List`
137     * @return the 'response.result' field of the JSON RPC response cast to type R
138     * @throws IOException network error
139     * @throws JsonRpcStatusException JSON RPC status error
140     */
141    @Override
142    public <R> R send(String method, List<Object> params) throws IOException, JsonRpcStatusException {
143        return send(method, defaultType, params);
144    }
145
146    /**
147     * Call an RPC method and return default object type.
148     *
149     * Convenience version that takes `params` as array/varargs.
150     *
151     * @param pass:[<R>] Type of result object
152     * @param method JSON RPC method call to send
153     * @param params JSON RPC parameters as array or varargs
154     * @return the 'response.result' field of the JSON RPC response cast to type R
155     * @throws IOException network error
156     * @throws JsonRpcStatusException JSON RPC status error
157     */
158    public <R> R send(String method, Object... params) throws IOException, JsonRpcStatusException {
159        return send(method, Arrays.asList(params));
160    }
161
162}
