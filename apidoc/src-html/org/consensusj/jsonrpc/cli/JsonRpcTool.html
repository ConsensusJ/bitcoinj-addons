001package org.consensusj.jsonrpc.cli;
002
003import org.consensusj.jsonrpc.JsonRpcException;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import java.io.IOException;
008import java.util.ArrayList;
009import java.util.List;
010
011/**
012 * JSON-RPC Command-line tool with logging support.
013 */
014public class JsonRpcTool extends CliCommand {
015    public final static String commandName = "jsonrpc";
016    // For a GraalVM command-line tool we muse configure Java Logging in main
017    // before initializing this Logger object
018    private static Logger log;
019
020
021    public JsonRpcTool(String[] args) {
022        super(commandName, new CliOptions(), args);
023    }
024
025    /**
026     * main method for bitcoinj-cli tool.
027     *
028     * See {@link CliOptions} for options and https://bitcoin.org/en/developer-reference#bitcoin-core-apis[Bitcoin Core JSON-RPC API]
029     * for the methods and parameters. Users can use `-?` to get general help or `help <command>` to get help
030     * on a specific command.
031     *
032     * @param args options, JSON-RPC method, JSON-RPC parameters
033     */
034    public static void main(String[] args) {
035        JavaLoggingSupport.configure("org.consensusj.jsonrpc");
036        log = LoggerFactory.getLogger(JsonRpcTool.class);
037        JsonRpcTool command = new JsonRpcTool(args);
038        log.trace("About to run command object");
039        int status = command.run();
040        log.trace("Command object completed with status: {}", status);
041        System.exit(status);
042    }
043
044    @Override
045    public Integer runImpl() throws IOException {
046        List<String> args = line.getArgList();
047        if (args.size() == 0) {
048            printError("jsonrpc method required");
049            printHelp();
050            return(1);
051        }
052        String method = args.get(0);
053        args.remove(0); // remove method from list
054        List<Object> typedArgs = convertParameters(method, args);
055        Object result;
056        try {
057            result = client.send(method, typedArgs);
058        } catch (JsonRpcException e) {
059            e.printStackTrace();
060            return 1;
061        }
062        if (result != null) {
063            pwout.println(result.toString());
064        }
065        return 0;
066    }
067
068    /**
069     * Convert params from strings to Java types that will map to correct JSON types
070     *
071     * TODO: Make this better and complete
072     *
073     * @param method the JSON-RPC method
074     * @param params Params with String type
075     * @return Params with correct Java types for JSON
076     */
077    protected List<Object> convertParameters(String method, List<String> params) {
078        // Default (for now) is to leave them all as strings
079        return new ArrayList<>(params);
080    }
081
082}
