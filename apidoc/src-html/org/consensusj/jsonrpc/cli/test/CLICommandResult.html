001package org.consensusj.jsonrpc.cli.test;
002
003import java.io.ByteArrayOutputStream;
004import java.io.UnsupportedEncodingException;
005import java.nio.charset.StandardCharsets;
006
007/**
008 * Holds all the results (status code, stdout, stderr) of a CLI command run by a test
009 */
010public class CLICommandResult {
011    static final String charset = StandardCharsets.UTF_8.toString();
012
013    public final int status;
014    public final String output;
015    public final String error;
016
017    public CLICommandResult(int status, String output, String error) {
018        this.status = status;
019        this.output = output;
020        this.error = error;
021    }
022
023    public CLICommandResult(int status, ByteArrayOutputStream os, ByteArrayOutputStream es) throws UnsupportedEncodingException {
024        this(status, os.toString(charset), es.toString(charset));
025    }
026
027}
