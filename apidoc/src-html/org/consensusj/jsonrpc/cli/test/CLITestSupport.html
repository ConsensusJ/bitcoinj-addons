001package org.consensusj.jsonrpc.cli.test;
002
003import org.consensusj.jsonrpc.cli.BaseJsonRpcTool;
004
005import java.io.ByteArrayOutputStream;
006import java.io.PrintStream;
007import java.io.UnsupportedEncodingException;
008
009/**
010 *  Support functions for testing command-line tools
011 */
012public interface CLITestSupport {
013
014    default String[] parseCommandLine(String line) {
015        String[] args = line.split(" ");     // (Overly?) simple parsing of string into args[]
016        return args;
017    }
018
019    /**
020     * Run a command and capture status and output
021     *
022     * @param tool Command object instance to run
023     * @return Object containing status, stdout, stderr
024     */
025    default CLICommandResult runTool(BaseJsonRpcTool tool, String... args) throws UnsupportedEncodingException {
026        // Setup to capture output streams
027        ByteArrayOutputStream bos = new ByteArrayOutputStream();
028        ByteArrayOutputStream bes = new ByteArrayOutputStream();
029        PrintStream cos = new PrintStream(bos);
030        PrintStream ces = new PrintStream(bes);
031
032        // Run the command
033        int status = tool.run(cos, ces, args);
034
035        return new CLICommandResult(status, bos, bes);
036    }
037}
