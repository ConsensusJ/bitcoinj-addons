001package org.consensusj.jsonrpc.cli;
002
003
004import org.apache.commons.cli.Option;
005import org.apache.commons.cli.OptionGroup;
006import org.apache.commons.cli.Options;
007
008/**
009 *
010 */
011public class CliOptions extends Options {
012    public CliOptions() {
013        super();
014        this.addOption("?", null, false, "This help message")
015            .addOption(Option.builder()
016                    .longOpt("url")
017                    .desc("URL for the JSON-RPC endpoint")
018                    .hasArg()
019                    .argName("url")
020                    .build())
021            .addOption((Option.builder("v")
022                    .longOpt("verbose")
023                    .desc("Print verbose output")
024                    .build()))
025            // Bitcoin-CLI style options
026            .addOption(Option.builder()
027                    .longOpt("rpcconnect")
028                    .desc("Send commands to node running on <ip> (default: 127.0.0.1)")
029                    .hasArg()
030                    .argName("ip")
031                    .build())
032            .addOption(Option.builder()
033                    .longOpt("rpcport")
034                    .desc("Connect to JSON-RPC on <port> (default: 8080)")
035                    .hasArg()
036                    .argName("port")
037                    .build())
038            .addOption(Option.builder()
039                    .longOpt("rpcuser")
040                    .desc("Username for JSON-RPC connections")
041                    .hasArg()
042                    .argName("user")
043                    .build())
044            .addOption(Option.builder().
045                    longOpt("rpcpassword")
046                    .desc("Password for JSON-RPC connections")
047                    .hasArg()
048                    .argName("pw")
049                    .build())
050            .addOption(null, "rpcssl", false, "Use https for JSON-RPC connections");
051    }
052
053}
