001package org.consensusj.jsonrpc.cli;
002
003import org.apache.commons.cli.Options;
004import org.slf4j.Logger;
005import org.slf4j.LoggerFactory;
006
007import java.util.ArrayList;
008import java.util.List;
009
010/**
011 * JSON-RPC Command-line tool with logging support.
012 */
013public class GenericJsonRpcTool extends BaseJsonRpcTool {
014    public final static String commandName = "jsonrpc";
015
016    // For a GraalVM command-line tool we muse configure Java Logging in main
017    // before initializing this Logger object
018    private static Logger log;
019
020    /**
021     * main method for jsonrpc tool.
022     *
023     * See {@link JsonRpcToolOptions} for options and https://bitcoin.org/en/developer-reference#bitcoin-core-apis[Bitcoin Core JSON-RPC API]
024     * for the methods and parameters. Users can use `-?` to get general help or `help <command>` to get help
025     * on a specific command.
026     *
027     * @param args options, JSON-RPC method, JSON-RPC parameters
028     */
029    public static void main(String[] args) {
030        JavaLoggingSupport.configure("org.consensusj.jsonrpc");
031        log = LoggerFactory.getLogger(GenericJsonRpcTool.class);
032        GenericJsonRpcTool tool = new GenericJsonRpcTool();
033        log.trace("About to run command object");
034        int status = tool.run(System.out, System.err, args);
035        log.trace("Command object completed with status: {}", status);
036        System.exit(status);
037    }
038
039    public GenericJsonRpcTool() {
040        super();
041    }
042
043    @Override public String name() {
044        return commandName;
045    }
046
047    @Override
048    public Options options() {
049        return new JsonRpcToolOptions();
050    }
051}
