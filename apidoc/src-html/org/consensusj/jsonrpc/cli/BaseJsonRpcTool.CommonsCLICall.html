001package org.consensusj.jsonrpc.cli;
002
003import org.apache.commons.cli.CommandLine;
004import org.apache.commons.cli.CommandLineParser;
005import org.apache.commons.cli.DefaultParser;
006import org.apache.commons.cli.HelpFormatter;
007import org.apache.commons.cli.Options;
008import org.apache.commons.cli.ParseException;
009import org.consensusj.jsonrpc.JsonRpcException;
010import org.consensusj.jsonrpc.RpcClient;
011
012import java.io.IOException;
013import java.io.PrintWriter;
014import java.net.URI;
015import java.net.URISyntaxException;
016import java.util.ArrayList;
017import java.util.List;
018
019/**
020 * An abstract base class for JsonRpcClientTool that uses Apache Commons CLI
021 */
022public abstract class BaseJsonRpcTool implements JsonRpcClientTool {
023    private final static String name = "jsonrpc";
024    protected final String usage ="usage string";
025    protected final HelpFormatter formatter = new HelpFormatter();
026
027
028    public BaseJsonRpcTool() {
029        formatter.setLongOptPrefix("-");
030    }
031
032    @Override
033    public String name() {
034        return name;
035    }
036
037    public String usage() {
038        return usage;
039    }
040
041    abstract public Options options();
042
043    @Override
044    public CommonsCLICall createCall(PrintWriter out, PrintWriter err, String... args) {
045        return new CommonsCLICall(this, out, err, args);
046    }
047
048    @Override
049    public void run(Call call) {
050        run((CommonsCLICall) call);
051    }
052    
053    public void run(CommonsCLICall call) {
054        List<String> args = call.line.getArgList();
055        if (args.size() == 0) {
056            printError(call,"jsonrpc method required");
057            printHelp(call, usage);
058            throw new ToolException(1, "jsonrpc method required");
059        }
060        String method = args.get(0);
061        args.remove(0); // remove method from list
062        List<Object> typedArgs = convertParameters(method, args);
063        Object result;
064        try {
065            result = call.rpcClient().send(method, typedArgs);
066        } catch (JsonRpcException e) {
067            e.printStackTrace();
068            throw new ToolException(1, e.getMessage());
069        } catch (IOException e) {
070            e.printStackTrace();
071            throw new ToolException(1, e.getMessage());
072        }
073        if (result != null) {
074            call.out.println(result.toString());
075        }
076    }
077
078    /**
079     * Convert params from strings to Java types that will map to correct JSON types
080     *
081     * TODO: Make this better and complete
082     *
083     * @param method the JSON-RPC method
084     * @param params Params with String type
085     * @return Params with correct Java types for JSON
086     */
087    protected List<Object> convertParameters(String method, List<String> params) {
088        List<Object> converted = new ArrayList<>();
089        for (String param : params) {
090            Object convertedParam = convertParam(param);
091            converted.add(convertedParam);
092        }
093        return converted;
094    }
095
096    /**
097     * Convert a single param from a command-line option {@code String} to a type more appropriate
098     * for Jackson/JSON-RPC.
099     *
100     * @param param A string parameter to convert
101     * @return The input parameter, possibly converted to a different type
102     */
103    protected Object convertParam(String param) {
104        Object result;
105        Long l = toLong(param);
106        if (l != null) {
107            // If the param is a valid Long, return a Long
108            result = l;
109        } else {
110            // Else, return a Boolean or String
111            switch (param) {
112                case "false":
113                    result = Boolean.FALSE;
114                    break;
115                case "true":
116                    result = Boolean.TRUE;
117                    break;
118                default:
119                    result = param;
120            }
121        }
122        return result;
123    }
124
125    // Convert to Long (if possible), else return null
126    protected static Long toLong(String strNum) {
127        try {
128            Long l = Long.parseLong(strNum);
129            return l;
130        } catch (NumberFormatException nfe) {
131            return null;
132        }
133    }
134
135    public void printHelp(Call call, String usage) {
136        int leftPad = 4;
137        int descPad = 2;
138        int helpWidth = 120;
139        String header = "";
140        String footer = "";
141        formatter.printHelp(call.err, helpWidth, usage, header, options(), leftPad, descPad, footer, false);
142    }
143
144    public void printError(Call call, String str) {
145        call.err.println(str);
146    }
147
148
149    public static class CommonsCLICall extends JsonRpcClientTool.Call {
150        protected final BaseJsonRpcTool rpcTool;
151        public final CommandLine line;
152        public final boolean verbose;
153        private RpcClient client;
154
155        public CommonsCLICall(BaseJsonRpcTool parentTool, PrintWriter out, PrintWriter err, String[] args) {
156            super(out, err, args);
157            this.rpcTool = parentTool;
158            CommandLineParser parser = new DefaultParser();
159            try {
160                this.line = parser.parse(rpcTool.options(), args);
161            } catch (ParseException e) {
162                throw new JsonRpcClientTool.ToolException(1, "Parser error");
163            }
164            if (line.hasOption("?")) {
165                rpcTool.printHelp(this, rpcTool.usage());
166                throw new JsonRpcClientTool.ToolException(1, "Help Option was chosen");
167            }
168            verbose = line.hasOption("v");
169            if (verbose) {
170                JavaLoggingSupport.setVerbose();
171            }
172            // TODO: Add rpcwait option for non-Bitcoin JsonRPC???
173        }
174
175        @Override
176        public RpcClient rpcClient() {
177            if (client == null) {
178                URI uri;
179                String urlString;
180                if ((urlString = line.getOptionValue("url")) != null ) {
181                    try {
182                        uri = new URI(urlString);
183                    } catch (URISyntaxException e) {
184                        throw new RuntimeException(e);
185                    }
186                } else {
187                    uri = URI.create("http://localhost:8080/");  // Hardcoded for now
188                }
189                String rpcUser = null;
190                String rpcPassword = null;
191                String rawUserInfo = uri.getRawUserInfo();
192                if (rawUserInfo != null) {
193                    String[] split = rawUserInfo.split(":");
194                    rpcUser = split[0];
195                    rpcPassword = split[1];
196                }
197                client = new RpcClient(uri, rpcUser, rpcPassword);
198            }
199            return client;
200        }
201    }
202}
