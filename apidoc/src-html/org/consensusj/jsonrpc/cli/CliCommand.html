001package org.consensusj.jsonrpc.cli;
002
003import org.apache.commons.cli.Options;
004import org.consensusj.jsonrpc.JsonRpcException;
005import org.apache.commons.cli.CommandLine;
006import org.apache.commons.cli.CommandLineParser;
007import org.apache.commons.cli.DefaultParser;
008import org.apache.commons.cli.HelpFormatter;
009import org.apache.commons.cli.ParseException;
010
011import java.io.IOException;
012import java.io.InputStream;
013import java.io.OutputStreamWriter;
014import java.io.PrintStream;
015import java.io.PrintWriter;
016import java.net.URI;
017import java.net.URISyntaxException;
018import java.nio.charset.StandardCharsets;
019
020import org.consensusj.jsonrpc.RpcClient;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023
024/**
025 * Base class for CLI commands that use JSON-RPC
026 */
027public abstract class CliCommand {
028    private static Logger log = LoggerFactory.getLogger(CliCommand.class);
029    static final String defaultproto = "http";
030    static final String defaulthost = "127.0.0.1";
031    static final int defaultport = 8332;
032    static final String defaultfile = "/";
033
034    protected CommandLine line;
035    protected CommandLineParser parser;
036    protected Options options;
037    protected String name;
038    protected String usage;
039    protected HelpFormatter formatter = null;
040
041    protected InputStream in;
042    protected PrintWriter pwout;
043    protected PrintWriter pwerr;
044
045    protected boolean verbose = false;
046
047    protected RpcClient client = null;
048
049    protected CliCommand(String name, Options options, String[] args) {
050        this(name, name, options, args);
051    }
052
053    protected CliCommand(String name, String usage, Options options, String[] args) {
054        this.name = name;
055        this.usage = usage;
056        this.options = options;
057        parser = new DefaultParser();
058        try {
059            this.line = this.parser.parse(options, args);
060        } catch (ParseException e) {
061            e.printStackTrace();
062            throw new RuntimeException(e);
063        }
064    }
065    
066    /**
067     * Can be used by subclasses to set a narrower type
068     * @param client An instance of BitcoinClient or subclass
069     */
070    protected void setClient(RpcClient client) {
071        this.client = client;
072    }
073
074    public void printHelp() {
075        if (formatter == null) {
076            formatter = new HelpFormatter();
077            formatter.setLongOptPrefix("-");
078        }
079        int leftPad = 4;
080        int descPad = 2;
081        int helpWidth = 120;
082        String header = "";
083        String footer = "";
084        formatter.printHelp(pwout, helpWidth, usage, header, options, leftPad, descPad, footer, false);
085    }
086
087    public void printError(String str) {
088        pwout.println(str);
089    }
090
091    /**
092     * Check Options and Arguments
093     *
094     * Override to customize behavior.
095     *
096     * @return status code
097     */
098    public int checkArgs() {
099        if (line.hasOption("?")) {
100            printHelp();
101            // Return 1 so tool can exit
102            return 1;
103        }
104        verbose = line.hasOption("v");
105        if (verbose) {
106            JavaLoggingSupport.setVerbose();
107        }
108        return 0;
109    }
110
111    public RpcClient getClient() {
112        if (client == null) {
113            URI uri;
114            String urlString;
115            if ((urlString = line.getOptionValue("url")) != null ) {
116                try {
117                    uri = new URI(urlString);
118                } catch (URISyntaxException e) {
119                    throw new RuntimeException(e);
120                }
121            } else {
122                uri = URI.create("http://localhost:8080/");  // Hardcoded for now
123            }
124            String rpcUser = null;
125            String rpcPassword = null;
126            String rawUserInfo = uri.getRawUserInfo();
127            if (rawUserInfo != null) {
128                String[] split = rawUserInfo.split(":");
129                rpcUser = split[0];
130                rpcPassword = split[1];
131            }
132            client = new RpcClient(uri, rpcUser, rpcPassword);
133        }
134        return client;
135    }
136
137    /**
138     * Initial a client
139     * TODO: Add rpcwait option for non-Bitcoin JsonRPC???
140     * @return status code
141     */
142    public int preflight() {
143        getClient();
144        return 0;
145    }
146
147    public int run() {
148        return run(System.in, System.out, System.err);
149    }
150
151    public Integer run(InputStream in, PrintStream out, PrintStream err) {
152        this.in = in;
153        this.pwout = new PrintWriter(new OutputStreamWriter(out, StandardCharsets.UTF_8), true);
154        this.pwerr = new PrintWriter(new OutputStreamWriter(err, StandardCharsets.UTF_8), true);
155
156        Integer status = checkArgs();
157        if (status != 0) {
158            return status;
159        }
160
161        status = preflight();
162        if (status != 0) {
163            return status;
164        }
165
166        try {
167            status = runImpl();
168        } catch (JsonRpcException e) {
169            e.printStackTrace();
170            status = 1;
171        } catch (IOException e) {
172            e.printStackTrace();
173            status = 1;
174        }
175        return status;
176    }
177    
178    /**
179     * Implement in subclasses
180     * @return status code
181     */
182    abstract protected Integer runImpl() throws IOException, JsonRpcException;
183}
