001package org.consensusj.jsonrpc;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonProperty;
005import com.fasterxml.jackson.databind.annotation.JsonSerialize;
006import com.msgilligan.bitcoinj.json.conversion.NumberStringSerializer;
007
008import java.util.Collections;
009import java.util.LinkedList;
010import java.util.List;
011import java.util.concurrent.atomic.AtomicLong;
012
013/**
014 * JSON-RPC Request POJO
015 */
016public class JsonRpcRequest {
017    public static final String JSON_RPC_VERSION_1 = "1.0";
018    public static final String JSON_RPC_VERSION_2 = "2.0";
019    private static final String DEFAULT_JSON_RPC_VERSION = JSON_RPC_VERSION_1;
020    private static AtomicLong nextRequestId =  new AtomicLong(0);
021
022    private final String  jsonrpc;   // version
023    private final String  method;
024    private final String  id;
025    private final List<Object> params;
026
027    /**
028     * For use by Jackson deserialization
029     * @param jsonrpc filled from the JSON object
030     * @param method filled from the JSON object
031     * @param id filled from the JSON object
032     * @param params filled from the JSON object
033     */
034    @JsonCreator
035    public JsonRpcRequest(@JsonProperty("jsonrpc")  String jsonrpc,
036                          @JsonProperty("method")   String method,
037                          @JsonProperty("id")       String id,
038                          @JsonProperty("params")   List<Object> params) {
039        this.jsonrpc = jsonrpc;
040        this.method = method;
041        this.id = id;
042        this.params = params;
043    }
044
045    /**
046     * For creating a JSON RPC request for serialization
047     * @param method Method of remote procedure to call
048     * @param params Parameters to serialize
049     */
050    public JsonRpcRequest(String method, List<Object> params) {
051        this(method, params, DEFAULT_JSON_RPC_VERSION);
052    }
053
054    public JsonRpcRequest(String method, List<Object> params, String jsonRpcVersion) {
055        this.jsonrpc = jsonRpcVersion;
056        this.id =  Long.toString(JsonRpcRequest.nextRequestId.incrementAndGet());
057        this.method = method;
058        this.params = removeTrailingNulls(params);
059    }
060
061    /**
062     * Convenience constructor for requests with empty parameter list
063     * @param method method name string
064     */
065    public JsonRpcRequest(String method) {
066        this(method, Collections.emptyList());
067    }
068
069    public String getJsonrpc() {
070        return jsonrpc;
071    }
072
073    public String getMethod() {
074        return method;
075    }
076
077    @JsonSerialize(using=NumberStringSerializer.class)
078    public String getId() {
079        return id;
080    }
081
082    public List<Object> getParams() {
083        return params;
084    }
085
086    /**
087     * Remove trailing nulls (all nulls *following* the last non-null object)
088     *
089     * This allows convenience methods to use `null` parameters to indicate the
090     * server-determined default should be used. If `null` were actually passed as
091     * JSON, then the server default would be overridden.  `null` can be used before
092     * the last non-null element, but those `null`s will be sent to the server.
093     */
094    private static List<Object> removeTrailingNulls(List<Object> params) {
095        LinkedList<Object> cleaned = new LinkedList<>(params);
096        while ((cleaned.size() > 0) && (cleaned.getLast() == null)) {
097            cleaned.removeLast();
098        }
099        return cleaned;
100    }
101}
