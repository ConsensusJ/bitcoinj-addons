025/**
026 * = JSON-RPC Client
027 *
028 * This is a concrete class with basic JSON-RPC functionality. Abstract `send` method is implemented
029 * using `HttpURLConnection`.
030 *
031 * Uses strongly-typed POJOs representing {@link JsonRpcRequest} and {@link JsonRpcResponse}. The
032 * response object uses a parameterized type for the object that is the actual JSON-RPC `result`.
033 * Using strong types and Jackson to serialize/deserialize to/from strongly-typed POJO's without
034 * using intermediate `Map` or `JsonNode` types.
035 *
036 */
037public class RPCClient extends AbstractRPCClient {
038    private static final Logger log = LoggerFactory.getLogger(RPCClient.class);
039    private URI serverURI;
040    private String username;
041    private String password;
042    private static final boolean disableSslVerification = false;
043    private static final String UTF8 = StandardCharsets.UTF_8.name();
044
045    static {
046        if (disableSslVerification) {
047            // Disable checks that prevent using a self-signed SSL certificate
048            // TODO: Should checks be enabled by default for security reasons?
049            disableSslVerification();
050        }
051    }
052
053    /**
054     * Construct a JSON-RPC client from URI, username, and password
055     *
056     * @param server server URI should not contain username/password
057     * @param rpcUser username for the RPC HTTP connection
058     * @param rpcPassword password for the RPC HTTP connection
059     */
060    public RPCClient(URI server, final String rpcUser, final String rpcPassword) {
061        super();
062        this.serverURI = server;
063        this.username = rpcUser;
064        this.password = rpcPassword;
065    }
066
067    /**
068     * Get the URI of the server this client connects to
069     * @return Server URI
070     */
071    @Override
072    public URI getServerURI() {
073        return serverURI;
074    }
075
076    /**
077     * Send a JSON-RPC request to the server and return a JSON-RPC response.
078     *
079     * @param request JSON-RPC request
080     * @param responseType Response type to deserialize to
081     * @return JSON-RPC response
082     * @throws IOException when thrown by the underlying HttpURLConnection
083     * @throws JsonRPCStatusException when the HTTP response code is other than 200
084     */
085    @Override
086    protected <R> JsonRpcResponse<R> send(JsonRpcRequest request, JavaType responseType) throws IOException, JsonRPCStatusException {
087        HttpURLConnection connection = openConnection();
088
089        // TODO: Make sure HTTP keep-alive will work
090        // See: http://docs.oracle.com/javase/7/docs/technotes/guides/net/http-keepalive.html
091        // http://developer.android.com/reference/java/net/HttpURLConnection.html
092        // http://android-developers.blogspot.com/2011/09/androids-http-clients.html
093
094        if (log.isDebugEnabled()) {
095            log.debug("Req json: {}", mapper.writeValueAsString(request));
096        }
097
098        OutputStream requestStream = connection.getOutputStream();
099        mapper.writeValue(requestStream, request);
100        requestStream.close();
101
102        int responseCode = connection.getResponseCode();
103        log.debug("Response code: {}", responseCode);
104
105        if (responseCode != 200) {
106            handleBadResponseCode(responseCode, connection);
107        }
108
109        JsonRpcResponse<R> responseJson;
110        try {
111            if (log.isDebugEnabled()) {
112                // If logging enabled, copy InputStream to string and log
113                String responseBody = convertStreamToString(connection.getInputStream());
114                log.debug("responseBody: {}", responseBody);
115                responseJson = mapper.readValue(responseBody, responseType);
116            } else {
117                // Otherwise convert directly to responseType
118                responseJson = mapper.readValue(connection.getInputStream(), responseType);
119            }
120        } catch (JsonProcessingException e) {
121            log.error("JsonProcessingException: {}", e);
122            // TODO: Map to some kind of JsonRPC exception similar to JsonRPCStatusException
123            throw e;
124        }
125        log.debug("Resp json: {}", responseJson);
126        connection.disconnect();
127        return responseJson;
128    }
129
130    /**
131     * Prepare and throw JsonRPCStatusException with all relevant info
132     * @param responseCode Non-success response code
133     * @param connection the current connection
134     * @throws IOException IO Error
135     * @throws JsonRPCStatusException An exception containing the HTTP status coe and a message
136     */
137    private void handleBadResponseCode(int responseCode, HttpURLConnection connection) throws IOException, JsonRPCStatusException
138    {
139        String responseMessage = connection.getResponseMessage();
140        String exceptionMessage = responseMessage;
141        int jsonRPCCode = 0;
142        JsonRpcResponse bodyJson = null;    // Body as JSON if available
143        String bodyString = null;           // Body as String if not JSON
144        InputStream errorStream = connection.getErrorStream();
145        if (errorStream != null) {
146            if (connection.getContentType().equals("application/json")) {
147                // We got a JSON error response, parse it
148                bodyJson = mapper.readValue(errorStream, JsonRpcResponse.class);
149                JsonRpcError error = bodyJson.getError();
150                if (error != null) {
151                    // If there's a more specific message in the JSON use it instead.
152                    exceptionMessage = error.getMessage();
153                    jsonRPCCode = error.getCode();
154                    log.error("json error code: {}, message: {}", jsonRPCCode, exceptionMessage);
155                }
156            } else {
157                // No JSON, read response body as string
158                bodyString = convertStreamToString(errorStream);
159                log.error("error string: {}", bodyString);
160                errorStream.close();
161            }
162        }
163        throw new JsonRPCStatusException(exceptionMessage, responseCode, responseMessage, jsonRPCCode, bodyString, bodyJson);
164    }
165
166    private static String convertStreamToString(java.io.InputStream is) {
167        java.util.Scanner s = new java.util.Scanner(is, UTF8).useDelimiter("\\A");
168        return s.hasNext() ? s.next() : "";
169    }
170
171    private HttpURLConnection openConnection() throws IOException {
172        HttpURLConnection connection =  (HttpURLConnection) serverURI.toURL().openConnection();
173        connection.setDoOutput(true); // For writes
174        connection.setRequestMethod("POST");
175        connection.setRequestProperty("Accept-Charset", UTF8);
176        connection.setRequestProperty("Content-Type", "application/json;charset=" +  UTF8);
177        connection.setRequestProperty("Connection", "close");   // Avoid EOFException: http://stackoverflow.com/questions/19641374/android-eofexception-when-using-httpurlconnection-headers
178
179        String auth = username + ":" + password;
180        String basicAuth = "Basic " + base64Encode(auth);
181        connection.setRequestProperty ("Authorization", basicAuth);
182
183        return connection;
184    }
185
186    /**
187     * Encode username password as Base64 for basic authentication
188     *
189     * We're using an internal `Base64` utility class here (copied from Android) in order
190     * to have working, consistent behavior on JavaSE and Android. Prior to Android 8.0,
191     * Android has it's own implementation that differs from the JavaSE version.
192     * 
193     * @param authString An authorization string of the form `username:password`
194     * @return A compliant Base64 encoding of `authString`
195     */
196    protected static String base64Encode(String authString) {
197        return Base64.encodeToString(authString.getBytes(),Base64.NO_WRAP).trim();
198    }

200    // TODO: Allow for self-signed certificates without disabling all verification
201    private static void disableSslVerification() {
202        try
203        {
204            // Create a trust manager that does not validate certificate chains
205            TrustManager[] trustAllCerts = new TrustManager[] {new X509TrustManager() {
206                public java.security.cert.X509Certificate[] getAcceptedIssuers() {
207                    return null;
208                }
209                public void checkClientTrusted(X509Certificate[] certs, String authType) {
210                }
211                public void checkServerTrusted(X509Certificate[] certs, String authType) {
212                }
213            }
214            };
215
216            // Install the all-trusting trust manager
217            SSLContext sc = SSLContext.getInstance("SSL");
218            sc.init(null, trustAllCerts, new java.security.SecureRandom());
219            HttpsURLConnection.setDefaultSSLSocketFactory(sc.getSocketFactory());
220
221            // Create all-trusting host name verifier
222            HostnameVerifier allHostsValid = new HostnameVerifier() {
223                public boolean verify(String hostname, SSLSession session) {
224                    return true;
225                }
226            };
227
228            // Install the all-trusting host verifier
229            HttpsURLConnection.setDefaultHostnameVerifier(allHostsValid);
230        } catch (NoSuchAlgorithmException | KeyManagementException e ) {
231            log.error("Exception in disableSslVerification{}", e);
232            e.printStackTrace();
233        }
234    }
