001package org.consensusj.jsonrpc.daemon;
002
003import org.consensusj.jsonrpc.introspection.AbstractJsonRpcService;
004import org.consensusj.jsonrpc.introspection.JsonRpcServiceWrapper;
005import org.slf4j.Logger;
006import org.slf4j.LoggerFactory;
007
008import javax.inject.Singleton;
009import java.lang.invoke.MethodHandles;
010import java.lang.reflect.Method;
011import java.util.Map;
012
013/**
014 * Simple Echo JSON-RPC Service
015 */
016@Singleton
017public class EchoJsonRpcService extends AbstractJsonRpcService {
018    private static Logger log = LoggerFactory.getLogger(EchoJsonRpcService.class);
019    private static final Map<String, Method> methods = JsonRpcServiceWrapper.reflect(MethodHandles.lookup().lookupClass());
020
021    public EchoJsonRpcService() {
022        super(methods);
023    }
024
025    public String echo(String message) {
026        log.debug("EchoJsonRpcService: echo {}",message);
027        return message;
028    }
029
030}
