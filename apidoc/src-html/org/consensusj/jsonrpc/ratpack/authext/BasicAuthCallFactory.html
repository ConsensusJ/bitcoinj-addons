001package org.consensusj.jsonrpc.ratpack.authext;
002
003import org.consensusj.jsonrpc.util.Base64;
004import okhttp3.Call;
005import okhttp3.Request;
006import ratpack.retrofit.internal.RatpackCallFactory;
007
008/**
009 * Workaround solution for adding Basic Auth Header
010 * Ratpack 1.5 will have a proper mechanism for doing this.
011 */
012public class BasicAuthCallFactory implements okhttp3.Call.Factory {
013    private final String authString;
014
015    public BasicAuthCallFactory(String user, String password) {
016        this.authString = authString(user, password);
017    }
018
019    @Override
020    public Call newCall(Request request) {
021        return RatpackCallFactory.INSTANCE.newCall(request
022                .newBuilder()
023                .header("Authorization", authString)
024                .build());
025    }
026
027    private static String authString(String username, String password) {
028        String auth = username + ":" + password;
029        return "Basic " + Base64.encodeToString(auth.getBytes(),Base64.NO_WRAP).trim();
030    }
031}
