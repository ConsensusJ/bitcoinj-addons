001package org.consensusj.jsonrpc.ratpack;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
005import com.google.inject.Provides;
006import com.msgilligan.bitcoinj.json.conversion.RpcClientModule;
007import com.msgilligan.bitcoinj.json.conversion.RpcServerModule;
008import com.msgilligan.bitcoinj.rpc.RpcConfig;
009import org.consensusj.jsonrpc.util.Base64;
010import ratpack.exec.Execution;
011import ratpack.func.Factory;
012import ratpack.guice.ConfigurableModule;
013import ratpack.handling.Context;
014import ratpack.http.client.HttpClient;
015import ratpack.retrofit.RatpackRetrofit;
016import retrofit2.converter.jackson.JacksonConverterFactory;
017
018/**
019 * Create an asynchronous JSON RPC Client using RatpackRetrofit
020 * TODO: Remove specific dependencies on Bitcoin
021 */
022public class JsonRpcClientModule extends ConfigurableModule<RpcConfig> {
023    @Override
024    protected void configure() {
025    }
026
027    @Provides
028    ObjectMapper provideObjectMapper(RpcConfig config) {
029        return new ObjectMapper()
030                .registerModule(new Jdk8Module())
031                .registerModule(new RpcClientModule(config.getNetParams()))
032                .registerModule(new RpcServerModule(config.getNetParams()));
033    }
034    
035    @Provides
036    JacksonConverterFactory provideJacksonConverterFactory(ObjectMapper mapper) {
037        return JacksonConverterFactory.create(mapper);
038    }
039
040    @Provides
041    JsonRpcClient provideJsonRpcClient(Factory<? extends HttpClient> httpClientFactory,
042                                       JacksonConverterFactory converterFactory,
043                                       RpcConfig rpcConfig) {
044        return RatpackRetrofit
045                .client(rpcConfig.getURI())
046                .httpClient(httpClientFactory)
047                .configure(b -> {
048                    b.addConverterFactory(converterFactory);
049                })
050                .build(JsonRpcClient.class);
051    }
052
053    /**
054     * Return an HttpClient Factory that will create a client that adds an Authorization header to HTTP requests
055     *
056     * @param rpcConfig JSON-RPC Configuration with username and password
057     * @return A factory that will produce an HttpClient tthat sets the "Authorization" header
058     */
059    @Provides
060    Factory<? extends HttpClient> authClientFactory(RpcConfig rpcConfig) {
061        final String auth = authString(rpcConfig.getUsername(), rpcConfig.getPassword());
062        return () -> clientFactory
063                .create()
064                .copyWith(
065                        spec -> spec.requestIntercept(
066                                requestSpec -> requestSpec.headers(
067                                        mutableHeaders ->
068                                                mutableHeaders.add("Authorization", auth)
069                                )
070                        )
071                );
072    }
073    
074    /**
075     * HttpClient Factory borrowed from the private lambda in RatpackRetrofit.java
076     */
077    private Factory<? extends HttpClient> clientFactory = () -> {
078        Execution exec = Execution.current();
079        return exec
080                .maybeGet(HttpClient.class)
081                .orElseGet(() ->
082                    exec.get(Context.class).get(HttpClient.class)
083                );
084    };
085
086    private static String authString(String username, String password) {
087        String auth = username + ":" + password;
088        return "Basic " + Base64.encodeToString(auth.getBytes(),Base64.NO_WRAP).trim();
089    }
090}
