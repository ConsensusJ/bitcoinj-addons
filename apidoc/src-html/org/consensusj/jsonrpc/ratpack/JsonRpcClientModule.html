001package org.consensusj.jsonrpc.ratpack;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import com.fasterxml.jackson.datatype.jdk8.Jdk8Module;
005import com.google.inject.Provides;
006import com.msgilligan.bitcoinj.json.conversion.RpcClientModule;
007import com.msgilligan.bitcoinj.json.conversion.RpcServerModule;
008import org.consensusj.jsonrpc.ratpack.authext.BasicAuthCallFactory;
009import com.msgilligan.bitcoinj.rpc.RPCConfig;
010import ratpack.guice.ConfigurableModule;
011import ratpack.retrofit.RatpackRetrofit;
012import retrofit2.converter.jackson.JacksonConverterFactory;
013
014/**
015 * Create an asynchronous JSON RPC Client using RatpackRetrofit
016 * TODO: Remove specific dependencies on Bitcoin
017 */
018public class JsonRpcClientModule extends ConfigurableModule<RPCConfig> {
019    @Override
020    protected void configure() {
021    }
022
023    @Provides
024    ObjectMapper provideObjectMapper(RPCConfig config) {
025        return new ObjectMapper()
026                .registerModule(new Jdk8Module())
027                .registerModule(new RpcClientModule(config.getNetParams()))
028                .registerModule(new RpcServerModule());
029    }
030    
031    @Provides
032    JacksonConverterFactory provideJacksonConverterFactory(ObjectMapper mapper) {
033        return JacksonConverterFactory.create(mapper);
034    }
035
036    @Provides
037    JsonRpcClient provideJsonRpcClient(JacksonConverterFactory converterFactory,
038                                       okhttp3.Call.Factory callFactory,
039                                       RPCConfig rpcConfig) {
040        return RatpackRetrofit
041                .client(rpcConfig.getURI())
042                .configure(b -> {
043                    b.addConverterFactory(converterFactory);
044                    b.callFactory(callFactory);
045                })
046                .build(JsonRpcClient.class);
047    }
048
049    @Provides
050    okhttp3.Call.Factory provideCallFactory(RPCConfig config) {
051        return new BasicAuthCallFactory(config.getUsername(), config.getPassword());
052    }
053
054}
