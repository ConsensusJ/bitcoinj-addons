001package org.consensusj.jsonrpc.ratpack;
002
003import org.consensusj.jsonrpc.JsonRpcRequest;
004import ratpack.handling.Context;
005import ratpack.handling.Handler;
006
007import static ratpack.jackson.Jackson.fromJson;
008import static ratpack.jackson.Jackson.json;
009
010import javax.inject.Inject;
011import javax.inject.Singleton;
012import java.util.Arrays;
013import java.util.List;
014
015/**
016 * JsonRPC proxy handler
017 * Relay allowed RPC methods to a URI
018 */
019@Singleton
020public class RpcProxyHandler implements Handler {
021    private final List<String> allowedMethods =  Arrays.asList("getblockcount", "setgenerate");
022    private final JsonRpcClient rpcClient;
023
024    @Inject
025    public RpcProxyHandler(JsonRpcClient jsonRpcClient) {
026        rpcClient = jsonRpcClient;
027    }
028
029    @Override
030    public void handle(Context ctx)  {
031        ctx.parse(fromJson(JsonRpcRequest.class)).then(rpcReq -> {
032            if (allowedMethods.contains(rpcReq.getMethod())) {
033                rpcClient.call(rpcReq).then(rpcResponse -> ctx.render(json(rpcResponse)));
034            } else {
035                // Should we send a JsonRpcResponse here?
036                ctx.getResponse().status(403).send("JSON-RPC method not allowed by proxy");
037            }
038        });
039    }
040}
