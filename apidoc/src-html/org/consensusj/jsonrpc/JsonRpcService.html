001package org.consensusj.jsonrpc;
002
003import java.util.concurrent.CompletableFuture;
004
005/**
006 * Asynchronous JSON-RPC Server/Service
007 *
008 * A service-layer abstraction of JSON-RPC using plain old Java objects (POJOs)
009 * Can easily be used in controllers from Java Web frameworks.
010 *
011 */
012public interface JsonRpcService {
013    /**
014     * Handle a JSON-RPC Request POJO and return a Response POJO
015     * @param req The Request POJO
016     * @return the Response POJO
017     */
018    <T> CompletableFuture<JsonRpcResponse<T>> call(JsonRpcRequest req);
019}
