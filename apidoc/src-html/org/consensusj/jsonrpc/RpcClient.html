026/**
027 * = JSON-RPC Client
028 *
029 * This is a concrete class with basic JSON-RPC functionality. Abstract `sendRequestForResponse` method is implemented
030 * using `HttpURLConnection`.
031 *
032 * Uses strongly-typed POJOs representing {@link JsonRpcRequest} and {@link JsonRpcResponse}. The
033 * response object uses a parameterized type for the object that is the actual JSON-RPC `result`.
034 * Using strong types and Jackson to serialize/deserialize to/from strongly-typed POJO's without
035 * using intermediate `Map` or `JsonNode` types.
036 *
037 */
038public class RpcClient extends AbstractRpcClient {
039    private static final Logger log = LoggerFactory.getLogger(RpcClient.class);
040    private URI serverURI;
041    private String username;
042    private String password;
043    private static final boolean disableSslVerification = false;
044    private static final String UTF8 = StandardCharsets.UTF_8.name();
045
046    static {
047        if (disableSslVerification) {
048            // Disable checks that prevent using a self-signed SSL certificate
049            // TODO: Should checks be enabled by default for security reasons?
050            disableSslVerification();
051        }
052    }
053
054    /**
055     * Construct a JSON-RPC client from URI, username, and password
056     *
057     * @param server server URI should not contain username/password
058     * @param rpcUser username for the RPC HTTP connection
059     * @param rpcPassword password for the RPC HTTP connection
060     */
061    public RpcClient(URI server, final String rpcUser, final String rpcPassword) {
062        super();
063        log.debug("Constructing JSON-RPC client for: {}", server);
064        this.serverURI = server;
065        this.username = rpcUser;
066        this.password = rpcPassword;
067    }
068
069    /**
070     * Get the URI of the server this client connects to
071     * @return Server URI
072     */
073    @Override
074    public URI getServerURI() {
075        return serverURI;
076    }
077
078    /**
079     * Send a JSON-RPC request to the server and return a JSON-RPC response.
080     *
081     * @param request JSON-RPC request
082     * @param responseType Response type to deserialize to
083     * @return JSON-RPC response
084     * @throws IOException when thrown by the underlying HttpURLConnection
085     * @throws JsonRpcStatusException when the HTTP response code is other than 200
086     */
087    @Override
088    protected <R> JsonRpcResponse<R> sendRequestForResponse(JsonRpcRequest request, JavaType responseType) throws IOException, JsonRpcStatusException {
089        HttpURLConnection connection = openConnection();
090
091        // TODO: Make sure HTTP keep-alive will work
092        // See: http://docs.oracle.com/javase/7/docs/technotes/guides/net/http-keepalive.html
093        // http://developer.android.com/reference/java/net/HttpURLConnection.html
094        // http://android-developers.blogspot.com/2011/09/androids-http-clients.html
095
096        if (log.isDebugEnabled()) {
097            log.debug("JsonRpcRequest: {}", mapper.writeValueAsString(request));
098        }
099        
100        try (OutputStream requestStream = connection.getOutputStream()) {
101            mapper.writeValue(requestStream, request);
102        }
103
104        int responseCode = connection.getResponseCode();
105        log.debug("HTTP Response code: {}", responseCode);
106
107        if (responseCode != 200) {
108            handleBadResponseCode(responseCode, connection);
109        }
110
111        JsonRpcResponse<R> responseJson = responseFromStream(connection.getInputStream(), responseType);
112        connection.disconnect();
113        return responseJson;
114    }
115
116    private <R> JsonRpcResponse<R> responseFromStream(InputStream inputStream, JavaType responseType) throws IOException {
117        JsonRpcResponse<R> responseJson;
118        try {
119            if (log.isDebugEnabled()) {
120                // If logging enabled, copy InputStream to string and log
121                String responseBody = convertStreamToString(inputStream);
122                log.debug("Response Body: {}", responseBody);
123                responseJson = mapper.readValue(responseBody, responseType);
124            } else {
125                // Otherwise convert directly to responseType
126                responseJson = mapper.readValue(inputStream, responseType);
127            }
128        } catch (JsonProcessingException e) {
129            log.error("JsonProcessingException: {}", e);
130            // TODO: Map to some kind of JsonRPC exception similar to JsonRPCStatusException
131            throw e;
132        }
133        return responseJson;
134    }
135
136    /**
137     * Prepare and throw JsonRPCStatusException with all relevant info
138     * @param responseCode Non-success response code
139     * @param connection the current connection
140     * @throws IOException IO Error
141     * @throws JsonRpcStatusException An exception containing the HTTP status code and a message
142     */
143    private void handleBadResponseCode(int responseCode, HttpURLConnection connection) throws IOException, JsonRpcStatusException
144    {
145        String responseMessage = connection.getResponseMessage();
146        String exceptionMessage = responseMessage;
147        int jsonRpcCode = 0;
148        JsonRpcResponse bodyJson = null;    // Body as JSON if available
149        String bodyString = null;           // Body as String if not JSON
150        InputStream errorStream = connection.getErrorStream();
151        if (errorStream != null) {
152            if (connection.getContentType().equals("application/json")) {
153                JavaType genericResponseType = mapper.getTypeFactory().
154                        constructParametricType(JsonRpcResponse.class, JsonNode.class);
155                // We got a JSON error response -- try to parse it as a JsonRpcResponse
156                bodyJson = responseFromStream(errorStream, genericResponseType);
157                JsonRpcError error = bodyJson.getError();
158                if (error != null) {
159                    // If there's a more specific message in the JSON use it instead.
160                    exceptionMessage = error.getMessage();
161                    jsonRpcCode = error.getCode();
162                    // Since this is an error at the JSON level, let's log it with `info` level and
163                    // let the higher-level software decide whether to log it as `error` or not.
164                    // i.e. The higher-level software can set error level on this module to `warn` and then
165                    // decide whether to log this "error" not based upon the JsonRpcStatusException thrown.
166                    // An example occurs in Bitcoin when a client is waiting for a server to initialize
167                    // and returns 'Still scanning.. at block 530006 of 548850'
168                    log.info("json error code: {}, message: {}", jsonRpcCode, exceptionMessage);
169                }
170            } else {
171                // No JSON, read response body as string
172                bodyString = convertStreamToString(errorStream);
173                log.error("error string: {}", bodyString);
174                errorStream.close();
175            }
176        }
177        throw new JsonRpcStatusException(exceptionMessage, responseCode, responseMessage, jsonRpcCode, bodyString, bodyJson);
178    }
179
180    private static String convertStreamToString(java.io.InputStream is) {
181        java.util.Scanner s = new java.util.Scanner(is, UTF8).useDelimiter("\\A");
182        return s.hasNext() ? s.next() : "";
183    }
184
185    private HttpURLConnection openConnection() throws IOException {
186        HttpURLConnection connection =  (HttpURLConnection) serverURI.toURL().openConnection();
187        connection.setDoOutput(true); // For writes
188        connection.setRequestMethod("POST");
189        connection.setRequestProperty("Accept-Charset", UTF8);
190        connection.setRequestProperty("Content-Type", "application/json;charset=" +  UTF8);
191        connection.setRequestProperty("Connection", "close");   // Avoid EOFException: http://stackoverflow.com/questions/19641374/android-eofexception-when-using-httpurlconnection-headers
192
193        String auth = username + ":" + password;
194        String basicAuth = "Basic " + base64Encode(auth);
195        connection.setRequestProperty ("Authorization", basicAuth);
196
197        return connection;
198    }
199
200    /**
201     * Encode username password as Base64 for basic authentication
202     *
203     * We're using an internal `Base64` utility class here (copied from Android) in order
204     * to have working, consistent behavior on JavaSE and Android. Prior to Android 8.0,
205     * Android has it's own implementation that differs from the JavaSE version.
206     * 
207     * @param authString An authorization string of the form `username:password`
208     * @return A compliant Base64 encoding of `authString`
209     */
210    protected static String base64Encode(String authString) {
211        return Base64.encodeToString(authString.getBytes(),Base64.NO_WRAP).trim();
212    }

214    // TODO: Allow for self-signed certificates without disabling all verification
215    private static void disableSslVerification() {
216        try
217        {
218            // Create a trust manager that does not validate certificate chains
219            TrustManager[] trustAllCerts = new TrustManager[] {new X509TrustManager() {
220                public java.security.cert.X509Certificate[] getAcceptedIssuers() {
221                    return null;
222                }
223                public void checkClientTrusted(X509Certificate[] certs, String authType) {
224                }
225                public void checkServerTrusted(X509Certificate[] certs, String authType) {
226                }
227            }
228            };
229
230            // Install the all-trusting trust manager
231            SSLContext sc = SSLContext.getInstance("SSL");
232            sc.init(null, trustAllCerts, new java.security.SecureRandom());
233            HttpsURLConnection.setDefaultSSLSocketFactory(sc.getSocketFactory());
234
235            // Create all-trusting host name verifier
236            HostnameVerifier allHostsValid = new HostnameVerifier() {
237                public boolean verify(String hostname, SSLSession session) {
238                    return true;
239                }
240            };
241
242            // Install the all-trusting host verifier
243            HttpsURLConnection.setDefaultHostnameVerifier(allHostsValid);
244        } catch (NoSuchAlgorithmException | KeyManagementException e ) {
245            log.error("Exception in disableSslVerification{}", e);
246            e.printStackTrace();
247        }
248    }
