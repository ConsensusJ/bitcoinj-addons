001package org.consensusj.jsonrpc.introspection;
002
003import org.consensusj.jsonrpc.JsonRpcError;
004import org.consensusj.jsonrpc.JsonRpcErrorException;
005import org.consensusj.jsonrpc.JsonRpcRequest;
006import org.consensusj.jsonrpc.JsonRpcResponse;
007import org.consensusj.jsonrpc.JsonRpcService;
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import java.lang.reflect.Method;
012import java.util.HashMap;
013import java.util.List;
014import java.util.Map;
015import java.util.concurrent.CompletableFuture;
016
017import static org.consensusj.jsonrpc.JsonRpcError.Error.METHOD_NOT_FOUND;
018import static org.consensusj.jsonrpc.JsonRpcError.Error.SERVER_EXCEPTION;
019
020/**
021 * Interface to wrap a Java class with JsonRpc support.
022 */
023public interface JsonRpcServiceWrapper extends JsonRpcService {
024    static final Logger log = LoggerFactory.getLogger(JsonRpcServiceWrapper.class);
025
026    /**
027     * Get the service object
028     *
029     * @return the service object
030     */
031    Object getServiceObject();
032
033    /**
034     * Get a Method object for a named method
035     * @param methodName the name of the method to call
036     * @return method handle
037     */
038    Method getMethod(String methodName);
039
040    /**
041     * Handle a request by calling method, getting a result, and embedding it in response.
042     * 
043     * @param req The Request POJO
044     * @return A future JSON RPC Response
045     */
046    default <R> CompletableFuture<JsonRpcResponse<R>> call(final JsonRpcRequest req) {
047        log.debug("JsonRpcServiceWrapper.call: {}", req.getMethod());
048        CompletableFuture<R> result = callMethod(req.getMethod(), req.getParams());
049        return result.handle((R r, Throwable ex) -> resultCompletionHandler(req, r, ex));
050    }
051    
052    /**
053     * Map a request plus a result or error into a response
054     *
055     * @param req The request being services
056     * @param result A result object or null
057     * @param ex exception or null
058     * @return A success or error response as appropriate
059     */
060    default <T> JsonRpcResponse<T> resultCompletionHandler(JsonRpcRequest req, T result, Throwable ex) {
061        return (result != null) ?
062                wrapResult(req, result) :
063                wrapError(req, exceptionToError(ex));
064    }
065
066    /**
067     * Map an exception (typically from callMethod) to an Error POJO
068     *
069     * @param ex Exception returned from "unwrapped" method
070     * @return An error POJO for insertion in a JsonRpcResponse
071     */
072    default JsonRpcError exceptionToError(Throwable ex) {
073        return (ex instanceof JsonRpcErrorException) ?
074                ((JsonRpcErrorException) ex).getError() :
075                JsonRpcError.of(SERVER_EXCEPTION);
076    }
077
078    /**
079     * Call an "unwrapped" method from an existing service object
080     * 
081     * @param methodName Method to call
082     * @param params List of JSON-RPC parameters
083     * @return A future result POJO
084     */
085    default <R> CompletableFuture<R> callMethod(String methodName, List<Object> params) {
086        log.debug("JsonRpcServiceWrapper.callMethod: {}", methodName);
087        CompletableFuture<R> future = new CompletableFuture<>();
088        final Method mh = getMethod(methodName);
089        if (mh != null) {
090            try {
091                R result = (R) mh.invoke(getServiceObject(), params.toArray());
092                future.complete(result);
093            } catch (Throwable throwable) {
094                // TODO: Better error handling here - at least return throwable.message to client
095                future.completeExceptionally(JsonRpcErrorException.of(SERVER_EXCEPTION));
096            }
097        } else {
098            future.completeExceptionally(JsonRpcErrorException.of(METHOD_NOT_FOUND));
099        }
100        return future;
101    }
102    
103    /**
104     * Use reflection/introspection to generate a map of methods.
105     * Generally this is called from the constructor of implementing classes.
106     * @param apiClass The service class to reflect/introspect
107     * @return a map of method names to method objects
108     */
109    static Map<String, Method> reflect(Class<?>  apiClass) {
110        java.lang.reflect.Method[] publicInheritedMethods = apiClass.getMethods();
111        Map<String, Method> methods = new HashMap<>();
112        for (Method method : publicInheritedMethods) {
113            String name = method.getName();
114            methods.put(name, method);
115        }
116        return methods;
117    }
118
119    /**
120     * Wrap a Result POJO in a JsonRpcResponse
121     * @param req The request we are responding to
122     * @param result the result to wrap
123     * @return A valid JsonRpcResponse
124     */
125    static <T> JsonRpcResponse<T> wrapResult(JsonRpcRequest req, T result) {
126        return new JsonRpcResponse<>(result, null, req.getJsonrpc(), req.getId());
127    }
128
129    static <T> JsonRpcResponse<T> wrapError(JsonRpcRequest req, JsonRpcError error) {
130        return new JsonRpcResponse<>(null, error, req.getJsonrpc(), req.getId());
131    }
132}
