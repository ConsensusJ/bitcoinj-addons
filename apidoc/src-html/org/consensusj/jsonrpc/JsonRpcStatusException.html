001package org.consensusj.jsonrpc;
002
003/**
004 * JSON-RPC returned HTTP status other than 200
005 * Additional information is usually in JSON-RPC response
006 */
007public class JsonRpcStatusException extends JsonRpcException {
008    public final String httpMessage;
009    public final int httpCode;
010    public final int jsonRpcCode;
011    public final String response;
012    public final JsonRpcResponse responseJson;
013
014    /**
015     * Default Constructor
016     *
017     * @param message Error message from Json if available, else http status message
018     * @param httpCode HTTP status code, e.g. 404
019     * @param httpMessage HTTP status message, e.g. "Not found"
020     * @param jsonRPCCode Integer error code in JSON response, if any
021     * @param responseBody responseBody body as string
022     * @param responseBodyJson responseBody body as Json Map
023     */
024    public JsonRpcStatusException(String message, int httpCode, String httpMessage, int jsonRPCCode, String responseBody, JsonRpcResponse responseBodyJson ) {
025        super(message);
026        this.httpCode = httpCode;
027        this.httpMessage = httpMessage;
028        this.jsonRpcCode = jsonRPCCode;
029        this.response = responseBody;
030        this.responseJson = responseBodyJson;
031    }
032}
