001package org.consensusj.jsonrpc;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonProperty;
005
006/**
007 * JSON-RPC Error Object POJO
008 */
009public class JsonRpcError {
010    private final int code;
011    private final String message;
012    private final Object data;
013
014
015    @JsonCreator
016    public JsonRpcError(@JsonProperty("code") int code,
017                        @JsonProperty("message") String message,
018                        @JsonProperty("data") Object data) {
019        this.code = code;
020        this.message = message;
021        this.data = data;
022    }
023
024    private JsonRpcError(Error code) {
025        this.code = code.code;
026        this.message = code.message;
027        this.data = null;
028    }
029
030    private JsonRpcError(Error code, Throwable throwable) {
031        this.code = code.code;
032        this.message = code.message + ": " + throwable.getMessage();
033        this.data = null;
034    }
035
036    public static JsonRpcError of(Error code) {
037        return new JsonRpcError(code);
038    }
039
040    public static JsonRpcError of(Error code, Throwable throwable) {
041        return new JsonRpcError(code, throwable);
042    }
043
044    public int getCode() {
045        return code;
046    }
047
048    public String getMessage() {
049        return message;
050    }
051    
052    public Object getData() {
053        return data;
054    }
055
056    public enum Error {
057        // TODO: Add other reserved error codes from the spec
058        METHOD_NOT_FOUND(-32601, "Method not found"),
059        SERVER_ERROR(-32000, "Server error"),
060        SERVER_EXCEPTION(-32001, "Server exception");
061
062        private final int code;
063        private final String message;
064
065        Error(int code, String message) {
066            this.code = code;
067            this.message = message;
068        }
069
070        public int getCode() {
071            return code;
072        }
073
074        public String getMessage() {
075            return message;
076        }
077    }
078}
