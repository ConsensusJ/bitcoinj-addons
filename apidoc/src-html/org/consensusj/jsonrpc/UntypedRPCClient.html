001package org.consensusj.jsonrpc;
002
003import java.io.IOException;
004import java.util.List;
005
006/**
007 * Basic JSON-RPC remote call interface for RPC calls
008 *
009 * The parameter list is "untyped" (declared as `List<Object>`) and implementations are responsible
010 * for converting each Java object parameter to a valid and correct JSON object.
011 *
012 * This is used to implement the `DynamicRPCFallback` trait in Groovy which is applied
013 * to various Groovy RPC client implementations that typically inherit statically-dispatched
014 * methods from Java classes, but use `methodMissing()` to add JSON-RPC methods dynamically.
015 * This may be useful in other Dynamic JVM languages, as well.
016 *
017 * @see DynamicRPCClient
018 */
019public interface UntypedRPCClient {
020    /**
021     * Call an RPC method and return default object type.
022     *
023     * Caller should cast returned object to the correct type.
024     *
025     * @param method JSON RPC method call to send
026     * @param params JSON RPC parameters using types that are convertible to JSON
027     * @param pass:[<R>] Type of result object
028     * @return the `response.result` field of the JSON-RPC response cast to type R
029     * @throws IOException network error
030     * @throws JsonRPCStatusException JSON RPC status error
031     */
032    <R> R send(String method, List<Object> params) throws IOException, JsonRPCStatusException;
033}
