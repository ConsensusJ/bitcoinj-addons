001package org.consensusj.jsonrpc;
002
003/**
004 *  Exception wrapper for JsonRpcError
005 *  Useful in server implementations, throwing an error that will be transmitted
006 *  to the client.
007 */
008public class JsonRpcErrorException extends JsonRpcException {
009    private JsonRpcError error;
010
011    public JsonRpcErrorException(JsonRpcError error) {
012        super(error.getMessage());
013        this.error = error;
014    }
015
016    public JsonRpcErrorException(JsonRpcError error, Throwable cause) {
017        super(error.getMessage(), cause);
018        this.error = error;
019    }
020
021    public JsonRpcErrorException(JsonRpcError.Error code, Throwable cause) {
022        super(code.getMessage() + ": " + cause.getMessage(), cause);
023        this.error = JsonRpcError.of(code, cause);
024    }
025
026    /**
027     * Get the JSON RPC Error POJO
028     * @return An error object
029     */
030    public JsonRpcError getError() {
031        return error;
032    }
033
034    public static JsonRpcErrorException of(JsonRpcError.Error code) {
035        return new JsonRpcErrorException(JsonRpcError.of(code));
036    }
037}
