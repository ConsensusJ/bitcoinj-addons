001package org.consensusj.exchange;
002
003import javax.money.CurrencyUnit;
004import java.util.Map;
005import java.util.concurrent.ScheduledExecutorService;
006
007/**
008 * Generic XChange ExchangeRateProvider that can wrap any XChange class
009 */
010public class DynamicXChangeRateProvider extends BaseXChangeExchangeRateProvider {
011
012    public DynamicXChangeRateProvider(String exchangeClassName,
013                                      ScheduledExecutorService scheduledExecutorService,
014                                      Map<CurrencyUnit, String> tickerSymbolConversions,
015                                      CurrencyUnitPair... pairs) {
016        super(exchangeClassName, scheduledExecutorService, tickerSymbolConversions, pairs);
017    }
018
019    public DynamicXChangeRateProvider(String exchangeClassName, ScheduledExecutorService scheduledExecutorService, CurrencyUnitPair... pairs) {
020        super(exchangeClassName, scheduledExecutorService, null, pairs);
021    }
022
023    public DynamicXChangeRateProvider(String exchangeClassName, ScheduledExecutorService scheduledExecutorService, String... pairs) {
024        super(exchangeClassName, scheduledExecutorService, pairs);
025    }
026
027    public DynamicXChangeRateProvider(String exchangeClassName, CurrencyUnitPair... pairs) {
028        this(exchangeClassName, null, pairs);
029    }
030
031
032    public DynamicXChangeRateProvider(String exchangeClassName, String... pairs) {
033        this(exchangeClassName, null, pairs);
034    }
035}
