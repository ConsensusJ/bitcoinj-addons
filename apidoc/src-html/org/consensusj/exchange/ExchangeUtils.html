001package org.consensusj.exchange;
002
003import java.util.Arrays;
004import java.util.List;
005import java.util.stream.Collectors;
006
007/**
008 *
009 */
010public interface ExchangeUtils {
011    static List<CurrencyUnitPair> pairsConvert(String[] strings) {
012        return pairsConvert(Arrays.asList(strings));
013    }
014
015    static List<CurrencyUnitPair> pairsConvert(List<String> strings) {
016        return strings.stream()
017                .map(CurrencyUnitPair::new)
018                .collect(Collectors.toList());
019    }
020}
