041public class KnowmExchangeRateProvider extends BaseExchangeRateProvider implements ExchangeRateProvider {
042    protected final String exchangeClassName;
043    protected final Map<CurrencyUnit, String> tickerSymbolConversions;
044    protected final Map<CurrencyUnitPair, MonitoredCurrency> monitoredCurrencies = new HashMap<>();
045
046    protected String name;
047    protected ProviderContext providerContext;
048    protected MarketDataService marketDataService;
049
050    private volatile boolean initialized = false;
051    protected Exchange exchange;
052
053    /**
054     * Construct using an XChange Exchange class object for a set of currencies
055     * @param exchangeClassName Classname of XChange exchange we are wrapping
056     * @param pairs pairs to monitor
057     */
058    protected KnowmExchangeRateProvider(String exchangeClassName,
059                                              Map<CurrencyUnit, String> tickerSymbolConversions,
060                                              Collection<CurrencyUnitPair> pairs) {
061        this.exchangeClassName = exchangeClassName;
062        this.tickerSymbolConversions = (tickerSymbolConversions != null) ? tickerSymbolConversions : new HashMap<>();
063        for (CurrencyUnitPair pair : pairs) {
064            KnowmExchangeRateProvider.MonitoredCurrency monitoredCurrency = new KnowmExchangeRateProvider.MonitoredCurrency(pair, xchangePair(pair));
065            monitoredCurrencies.put(pair, monitoredCurrency);
066        }
067    }
068
069
070    public synchronized void initialize()  {
071        if (!initialized) {
072            initialized = true;
073            exchange = ExchangeFactory.INSTANCE.createExchange(exchangeClassName);
074            name = exchange.getExchangeSpecification().getExchangeName();
075            providerContext = ProviderContext.of(name, RateType.DEFERRED);
076            marketDataService = exchange.getMarketDataService();
077        }
078    }
079
080    public ExchangeRateUpdate getUpdate(CurrencyUnitPair pair) {
081        MonitoredCurrency pairMonitor = monitoredCurrencies.get(pair);
082        Ticker ticker;
083        try {
084            ticker = marketDataService.getTicker(pairMonitor.exchangePair);
085        } catch (IOException e) {
086            e.printStackTrace();
087            ticker = null;
088        }
089        return tickerToUpdate(pair, ticker);
090    }
091
092    private ExchangeRateUpdate tickerToUpdate(CurrencyUnitPair pair, Ticker ticker) {
093        return (ticker != null) ?
094                new ExchangeRateUpdate(pair, DefaultNumberValue.of(ticker.getBid()), ticker.getTimestamp().getTime()) :
095                new ExchangeRateUpdate(pair, null, 0);
096    }
097
098    @Override
099    public ProviderContext getContext() {
100        return providerContext;
101    }
102
103    @Override
104    public boolean isAvailable(ConversionQuery conversionQuery) {
105        return getExchangeRate(conversionQuery) != null;
106    }
107
108    @Override
109    public ExchangeRate getExchangeRate(ConversionQuery conversionQuery) {
110        CurrencyUnitPair pair = new CurrencyUnitPair(conversionQuery.getBaseCurrency(), conversionQuery.getCurrency());
111        KnowmExchangeRateProvider.MonitoredCurrency monitoredCurrency = monitoredCurrencies.get(pair);
112        if (monitoredCurrency == null) {
113            throw new CurrencyConversionException(pair.getBase(),
114                    pair.getTarget(),
115                    null,
116                    "Pair not found.");
117        }
118        ExchangeRate rate = null;
119        try {
120            rate = buildExchangeRate(pair, monitoredCurrency.getTicker());
121        } catch (TimeoutException e) {
122            throw new MonetaryException("Timeout loading exchange rate", e);
123        } catch (InterruptedException e) {
124            Thread.currentThread().interrupt(); // Restore interruption flag
125        }
126        return rate;
127    }
128
129    @Override
130    public CurrencyConversion getCurrencyConversion(ConversionQuery conversionQuery) {
131        return new LazyBoundCurrencyConversion(conversionQuery, this, ConversionContext
132                .of(getContext().getProviderName(), getContext().getRateTypes().iterator().next()));
133    }
134
135    @SuppressWarnings("deprecation")
136    protected ExchangeRateChange buildExchangeRateChange(CurrencyUnitPair pair, Ticker ticker) {
137        Date date = ticker.getTimestamp();
138        // Not all exchanges provide a timestamp, default to 0 if it is null
139        long milliseconds = (date != null) ? date.getTime() : 0;
140
141        return new ExchangeRateChange(buildExchangeRate(pair, ticker), milliseconds);
142    }
143
144    private ExchangeRate buildExchangeRate(CurrencyUnitPair pair, Ticker ticker) {
145        return new ExchangeRateBuilder(name, RateType.DEFERRED)
146                .setBase(pair.getBase())
147                .setTerm(pair.getTarget())
148                .setFactor(DefaultNumberValue.of(ticker.getLast()))
149                .build();
150    }
151
152    /**
153     * Map from CurrencyUnitPair to XChange CurrencyPair
154     * tickerSymbolConversions is used to handle cases like ItBit that uses "XBT" instead of "BTC"
155     * @param pair  CurrencyUnitPair using JavaMoney CurrencyUnits
156     * @return  XChange CurrencyPair with exchange-specific symbols if any
157     */
158    protected CurrencyPair xchangePair(CurrencyUnitPair pair) {
159        return new CurrencyPair(convertSymbol(pair.getBase()),
160                convertSymbol(pair.getTarget()));
161    }
162
163    /**
164     * Convert a JavaMoney CurrencyUnit to an XChange currency code string
165     * (This will be exchange-specific (e.g. ItBit uses "XBT" instead of "BTC")
166     * @param currencyUnit A JavaMoney currency unit
167     * @return exchange-specific symbol for the currency
168     */
169    public String convertSymbol(CurrencyUnit currencyUnit) {
170        return tickerSymbolConversions.getOrDefault(currencyUnit, currencyUnit.getCurrencyCode());
171    }
172
173    protected static class MonitoredCurrency {
174        final public CurrencyUnitPair  pair;           // Terminating (target) JavaMoney CurrencyUnit
175        final public CurrencyPair      exchangePair;   // XChange currency pair (format used by XChange/exchange)
176        @SuppressWarnings("deprecation")
177        final public List<ExchangeRateObserver> observerList = new ArrayList<>();
178        private final UpdateableValueLatch<Ticker> tickerLatch = new UpdateableValueLatch<>();
179
180        MonitoredCurrency(CurrencyUnitPair pair, CurrencyPair exchangePair) {
181            this.pair = pair;
182            this.exchangePair = exchangePair;
183        }
184
185        public boolean isTickerAvailable() {
186            return tickerLatch.isSet();
187        }
188
189        /**
190         * Get Ticker from the UpdateableValueLatch
191         * @return ticker object
192         * @throws InterruptedException thread was interrupted by another thread
193         * @throws TimeoutException should only happen if first request times out
194         */
195        public Ticker getTicker() throws InterruptedException, TimeoutException {
196            return tickerLatch.getValue();
197        }
198
199        /**
200         * Set the ticker object, called from polling method
201         * @param ticker ticker object
202         */
203        public void setTicker(Ticker ticker) {
204            tickerLatch.setValue(ticker);
205        }
206    }
207
208    /**
209     * A value latch that will block if value not yet set, but allows
210     * the value to be updated regularly (e.g. when polling for new values)
211     * @param <T>  Type of value to latch
212     */
213    static class UpdateableValueLatch<T> {
214        private volatile T value = null;
215        private final CountDownLatch ready = new CountDownLatch(1);
216        private final long timeoutSeconds;
217
218        UpdateableValueLatch() {
219            this(120);
220        }
221
222        UpdateableValueLatch(long timeoutInSeconds) {
223            this.timeoutSeconds = timeoutInSeconds;
224        }
225
226        T getValue() throws InterruptedException, TimeoutException {
227            // were we called before first poll completed?
228            if (value == null) {
229                // wait for the CountDownLatch
230                boolean loaded = ready.await(timeoutSeconds, TimeUnit.SECONDS);
231                if (!loaded) {
232                    throw new TimeoutException("Timeout before value set");
233                }
234            }
235            return value;
236        }
237
238        /* synchronized ? */ void setValue(T newValue) {
239            value = newValue;
240            ready.countDown();
241        }
242
243        boolean isSet() {
244            return value != null;
245        }
246    }
