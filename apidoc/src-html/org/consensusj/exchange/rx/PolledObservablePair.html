001package org.consensusj.exchange.rx;
002
003import io.reactivex.rxjava3.annotations.NonNull;
004import io.reactivex.rxjava3.core.Observable;
005import io.reactivex.rxjava3.core.Observer;
006import org.consensusj.exchange.CurrencyUnitPair;
007
008import java.util.function.Supplier;
009
010/**
011 * Implements ObservablePair using an exchange rate polling function and an interval
012 */
013public class PolledObservablePair extends Observable<ExchangeRateUpdate> implements ObservablePair {
014    private final CurrencyUnitPair pair;
015    private final Observable<ExchangeRateUpdate> observablePairUpdates;
016
017    /**
018     *
019     * @param pair The currency pair to observe
020     * @param rateSupplier A polling Supplier function (for a single pair)
021     * @param interval An interval (or equivalent) to provide clock ticks to trigger polling
022     */
023    public PolledObservablePair(CurrencyUnitPair pair, Supplier<ExchangeRateUpdate> rateSupplier, Observable<Long> interval) {
024        this.pair = pair;
025        observablePairUpdates = interval
026                .map(l -> rateSupplier.get())   // Use the Supplier function to do the polling
027                .replay(1)            // Create a ConnectableObservable that can start and stop polling
028                .refCount();                    // Create a refCount-tracking observable to share the ConnectableObservable
029    }
030
031    @Override
032    public CurrencyUnitPair getPair() {
033        return pair;
034    }
035    
036    @Override
037    protected void subscribeActual(@NonNull Observer<? super ExchangeRateUpdate> observer) {
038        observablePairUpdates.subscribe(observer);
039    }
040}
