001package com.msgilligan.bitcoinj.money;
002
003import java.util.concurrent.ScheduledExecutorService;
004
005/**
006 *  Coinbase ExchangeRateProvider using XChange library
007 */
008public class CoinbaseXChangeRateProvider extends BaseXChangeExchangeRateProvider {
009    static private final String[] pairs = {"BTC/USD"};
010    static private final String xchangeClassName = "org.knowm.xchange.coinbase.CoinbaseExchange";
011
012    public CoinbaseXChangeRateProvider(ScheduledExecutorService scheduledExecutorService) {
013        super(xchangeClassName, scheduledExecutorService, pairsConvert(pairs));
014    }
015
016    public CoinbaseXChangeRateProvider() {
017        super(xchangeClassName, null, pairsConvert(pairs));
018    }
019}
