001package com.msgilligan.bitcoinj.money;
002
003import org.knowm.xchange.currency.CurrencyPair;
004
005import java.util.concurrent.ScheduledExecutorService;
006
007/**
008 *  Bitfinex ExchangeRateProvider using XChange library
009 */
010public class ItBitXChangeRateProvider extends BaseXChangeExchangeRateProvider {
011    static private final String[] pairs = {"BTC/USD", "BTC/EUR"};
012    static private final String xchangeClassName = "org.knowm.xchange.itbit.v1.ItBitExchange";
013
014    public ItBitXChangeRateProvider(ScheduledExecutorService scheduledExecutorService) {
015        super(xchangeClassName, scheduledExecutorService, pairsConvert(pairs));
016    }
017    
018    public ItBitXChangeRateProvider() {
019        super(xchangeClassName, null, pairsConvert(pairs));
020    }
021
022    @Override
023    protected CurrencyPair xchangePair(CurrencyUnitPair pair) {
024        String base = pair.getBase().getCurrencyCode();
025        if (base.equals("BTC")) {
026            base = "XBT";
027        }
028        return new CurrencyPair(base, pair.getTarget().getCurrencyCode());
029    }
030}
