001package com.msgilligan.bitcoinj.money;
002
003import org.javamoney.moneta.CurrencyUnitBuilder;
004
005import javax.money.CurrencyContext;
006import javax.money.CurrencyContextBuilder;
007import javax.money.CurrencyQuery;
008import javax.money.CurrencyUnit;
009import javax.money.spi.CurrencyProviderSpi;
010import java.util.Collections;
011import java.util.HashSet;
012import java.util.Set;
013
014/**
015 *  A BitcoinCurrencyProvider based on work in the javamoney-shelter.
016 *
017 * @author Sean Gilligan
018 * @author Werner Keil
019 */
020public class BitcoinCurrencyProvider implements CurrencyProviderSpi {
021
022    final static int bitcoinFractionDigits = 8;
023
024    // Not sure what to do here...
025    private final CurrencyContext CONTEXT = CurrencyContextBuilder.of("BitcoinCurrencyContextProvider")
026            .build();
027
028    private Set<CurrencyUnit> bitcoinSet = new HashSet<>();
029
030    public BitcoinCurrencyProvider() {
031        CurrencyUnit btcUnit = CurrencyUnitBuilder.of("BTC", CONTEXT)
032                .setDefaultFractionDigits(bitcoinFractionDigits)
033                .build();
034        bitcoinSet.add(btcUnit);
035        bitcoinSet = Collections.unmodifiableSet(bitcoinSet);
036    }
037
038    @Override
039    public String getProviderName(){
040        return "bitcoin";
041    }
042
043    /**
044     * Return a {@link CurrencyUnit} instances matching the given
045     * {@link javax.money.CurrencyContext}.
046     *
047     * @param query the {@link javax.money.CurrencyQuery} containing the parameters determining the query. not null.
048     * @return the corresponding {@link CurrencyUnit}s matching, never null.
049     */
050    @Override
051    public Set<CurrencyUnit> getCurrencies(CurrencyQuery query){
052        // only ensure BTC is the code, or it is a default query.
053        if(query.getCurrencyCodes().contains("BTC") ||  query.getCurrencyCodes().isEmpty()){
054            return bitcoinSet;
055        }
056        return Collections.emptySet();
057    }
058
059    @Override
060    public boolean isCurrencyAvailable(CurrencyQuery query) {
061        return !getCurrencies(query).isEmpty();
062    }
063
064}
