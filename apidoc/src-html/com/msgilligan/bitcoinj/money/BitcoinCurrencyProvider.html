001package com.msgilligan.bitcoinj.money;
002
003import org.javamoney.moneta.CurrencyUnitBuilder;
004
005import javax.money.CurrencyContext;
006import javax.money.CurrencyContextBuilder;
007import javax.money.CurrencyQuery;
008import javax.money.CurrencyUnit;
009import javax.money.spi.CurrencyProviderSpi;
010import java.util.Collections;
011import java.util.Set;
012
013/**
014 *  A BitcoinCurrencyProvider based on work in the javamoney-shelter.
015 *
016 * @author Sean Gilligan
017 * @author Werner Keil
018 */
019public class BitcoinCurrencyProvider implements CurrencyProviderSpi {
020    private final static int bitcoinFractionDigits = 8;
021    private final Set<CurrencyUnit> bitcoinSet;
022
023    public BitcoinCurrencyProvider() {
024        CurrencyContext CONTEXT = CurrencyContextBuilder.of("BitcoinCurrencyContextProvider")
025                .build();
026        CurrencyUnit btcUnit = CurrencyUnitBuilder.of("BTC", CONTEXT)
027                .setDefaultFractionDigits(bitcoinFractionDigits)
028                .build();
029        bitcoinSet = Collections.singleton(btcUnit);
030    }
031    
032    /**
033     * Return a {@link CurrencyUnit} instances matching the given
034     * {@link javax.money.CurrencyQuery}.
035     *
036     * @param query the {@link javax.money.CurrencyQuery} containing the parameters determining the query. not null.
037     * @return the corresponding {@link CurrencyUnit}s matching, never null.
038     */
039    @Override
040    public Set<CurrencyUnit> getCurrencies(CurrencyQuery query){
041        // Query for currencyCode BTC or default query returns bitcoinSet else emptySet.
042        return (query.getCurrencyCodes().contains("BTC") ||
043                query.getCurrencyCodes().isEmpty()) ? bitcoinSet : Collections.emptySet();
044    }
045}
