001package com.msgilligan.bitcoinj.money;
002
003import javax.money.convert.ExchangeRate;
004import javax.money.convert.ExchangeRateProvider;
005
006/**
007 * ExchangeRateProvider extension that allows an observer to be notified of exchange rate updates
008 */
009public interface ObservableExchangeRateProvider extends ExchangeRateProvider {
010    void registerExchangeRateObserver(CurrencyUnitPair pair, ExchangeRateObserver observer);
011    void start();
012    void stop();
013}
