007/**

014 */
015public class ExchangeRateChange {
016    public ExchangeRate rate;
017    public Long timestamp;
018    //Ticker ticker;
019
020    public ExchangeRateChange(ExchangeRate rate, long timestamp) {
021        this.rate = rate;
022        this.timestamp = timestamp;
023    }
