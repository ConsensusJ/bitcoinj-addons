037/**
038 *  Base ExchangeRateProvider using XChange library
039 *  Currently supports current DEFERRED rates only
040 */
041public abstract class BaseXChangeExchangeRateProvider extends BaseExchangeRateProvider implements ExchangeRateProvider, ObservableExchangeRateProvider {
042    private static final Logger log = LoggerFactory.getLogger(BaseXChangeExchangeRateProvider.class);
043    protected ProviderContext providerContext;
044    protected String name;
045    private final String exchangeClassName;
046    protected Exchange exchange;
047    protected MarketDataService marketDataService;
048    private ScheduledExecutorService stpe;
049    private volatile boolean started = false;
050    private volatile boolean stopping = false;
051    private  ScheduledFuture<?> future;
052    private final Map<CurrencyUnitPair, MonitoredCurrency> monitoredCurrencies = new HashMap<>();
053    private static final int initialDelay = 0;
054    private static final int period = 60;
055
056    /**
057     * Construct using an XChange Exchange class object for a set of currencies
058     * @param exchangeClassName Classname of XChange exchange we are wrapping
059     * @param scheduledExecutorService Executor service for running polling task
060     * @param pairs pairs to monitor
061     */
062    protected BaseXChangeExchangeRateProvider(String exchangeClassName,
063                                              ScheduledExecutorService scheduledExecutorService,
064                                              CurrencyUnitPair... pairs) {
065        this.exchangeClassName = exchangeClassName;
066        stpe = (scheduledExecutorService != null) ? scheduledExecutorService : Executors.newScheduledThreadPool(1);
067        for (CurrencyUnitPair pair : pairs) {
068            MonitoredCurrency monitoredCurrency = new MonitoredCurrency(pair, xchangePair(pair));
069            monitoredCurrencies.put(pair, monitoredCurrency);
070        }
071    }
072
073    /**
074     * Construct using an XChange Exchange class object for a set of currencies
075     * @param exchangeClass Class of XChange exchange we are wrapping
076     * @param pairs pairs to monitor
077     */
078    protected BaseXChangeExchangeRateProvider(Class<? extends Exchange> exchangeClass,
079                                              CurrencyUnitPair... pairs) {
080        this(exchangeClass,
081                null,
082                pairs);
083    }
084
085    /**
086     * Construct using an XChange Exchange class object for a set of currencies
087     * @param exchangeClass Class of XChange exchange we are wrapping
088     * @param scheduledExecutorService Executor service for running polling task

091    protected BaseXChangeExchangeRateProvider(Class<? extends Exchange> exchangeClass,
092                                              ScheduledExecutorService scheduledExecutorService,
093                                              CurrencyUnitPair... pairs) {
094        this(exchangeClass.getName(), scheduledExecutorService, pairs);
095    }

097    protected BaseXChangeExchangeRateProvider(Class<? extends Exchange> exchangeClass,
098                                              ScheduledExecutorService scheduledExecutorService,
099                                              String... pairs) {
100        this(exchangeClass, scheduledExecutorService, pairsConvert(pairs));
101    }
102
103    protected BaseXChangeExchangeRateProvider(Class<? extends Exchange> exchangeClass,
104                                              String... pairs) {
105        this(exchangeClass, pairsConvert(pairs));
106    }
107
108    protected BaseXChangeExchangeRateProvider(String exchangeClassName, ScheduledExecutorService scheduledExecutorService, String[] pairs) {
109        this(exchangeClassName, scheduledExecutorService, pairsConvert(pairs));
110    }
111
112    protected static CurrencyUnitPair[] pairsConvert(String[] strings) {
113        CurrencyUnitPair[] units = new CurrencyUnitPair[strings.length];
114        for (int i = 0 ; i < strings.length ; i++) {
115            units[i] = new CurrencyUnitPair(strings[i]);
116        }
117        return units;
118    }
119
120    /**
121     * Map from CurrencyUnitPair to XChange CurrencyPair
122     * Override to handle cases like ItBit that use "XBT" instead of "BTC"
123     * @param pair  CurrencyUnitPair using JavaMoney CurrencyUnits
124     * @return  XChange CurrencyPair
125     */
126    protected CurrencyPair xchangePair(CurrencyUnitPair pair) {
127        return new CurrencyPair(pair.getBase().getCurrencyCode(), pair.getTarget().getCurrencyCode());
128    }
129
130    /**
131     * Initialize the exchange provider and start polling thread
132     */
133    @Override
134    public synchronized void start()  {
135        if (!started) {
136            started = true;
137            exchange = ExchangeFactory.INSTANCE.createExchange(exchangeClassName);
138            name = exchange.getExchangeSpecification().getExchangeName();
139            providerContext = ProviderContext.of(name, RateType.DEFERRED);
140            marketDataService = exchange.getMarketDataService();
141
142            future = stpe.scheduleWithFixedDelay(this::poll, initialDelay, period, TimeUnit.SECONDS);
143        }
144    }
145
146    /**
147     * stop the polling thread
148     */
149    @Override
150    public synchronized void stop() {
151        if (started && !stopping) {
152            stopping = true;
153            final ScheduledFuture<?> handle = future;
154            Runnable task = () -> handle.cancel(true);
155            stpe.schedule(task, 0, TimeUnit.SECONDS);
156        }
157    }
158
159    /**
160     * Poll the exchange for updated Tickers
161     */
162    protected void poll() {
163        monitoredCurrencies.forEach((key, monitor) -> {
164            try {
165                monitor.setTicker(marketDataService.getTicker(monitor.exchangePair));
166            } catch (IOException e) {
167                // TODO: Exceptions should not be swallowed here (or at least not all exceptions)
168                // Some IOExceptions may warrant retries, but not all of them
169                // log and ignore IOException (we'll try polling again next interval)
170                // Actually I'm seeing that the CoinMarketCap ticker is returning IOException
171                // when it should return NotAvailableFromExchangeException
172                log.error("IOException in BaseXChangeExchangeRateProvider::poll: {}", e);
173            }
174            notifyExchangeRateObservers(monitor);
175        });
176    }
177
178    @Override
179    public void registerExchangeRateObserver(CurrencyUnitPair pair, ExchangeRateObserver observer) {
180        // TODO: validate rate as one this provider supports
181        MonitoredCurrency monitor = monitoredCurrencies.get(pair);
182        monitor.observerList.add(observer);
183        // If we've got data already, call observer immediately
184        if (monitor.isTickerAvailable()) {
185            notifyObserver(observer, pair, monitor);
186        }
187    }
188
189    private void notifyExchangeRateObservers(MonitoredCurrency monitor) {
190        monitor.observerList.forEach(observer -> notifyObserver(observer, monitor.pair, monitor));
191    }
192
193    private void notifyObserver(ExchangeRateObserver observer, CurrencyUnitPair pair, MonitoredCurrency monitor) {
194        try {
195            observer.onExchangeRateChange(buildExchangeRateChange(pair, monitor.getTicker()));
196        } catch (InterruptedException e) {
197            Thread.currentThread().interrupt(); // Restore interruption flag just in case
198        } catch (TimeoutException e) {
199            throw new RuntimeException(e);      // Unlikely to happen since ticker has usually been fetched
200        }
201    }
202    
203    @Override
204    public ProviderContext getContext() {
205        return providerContext;
206    }
207
208    @Override
209    public boolean isAvailable(ConversionQuery conversionQuery) {
210        return getExchangeRate(conversionQuery) != null;
211    }
212
213    @Override
214    public ExchangeRate getExchangeRate(ConversionQuery conversionQuery) {
215        CurrencyUnitPair pair = new CurrencyUnitPair(conversionQuery.getBaseCurrency(), conversionQuery.getCurrency());
216        MonitoredCurrency monitoredCurrency = monitoredCurrencies.get(pair);
217        if (monitoredCurrency == null) {
218            throw new CurrencyConversionException(pair.getBase(),
219                    pair.getTarget(),
220                    null,
221                    "Pair not found.");
222        }
223        ExchangeRate rate = null;
224        try {
225            rate = buildExchangeRate(pair, monitoredCurrency.getTicker());
226        } catch (TimeoutException e) {
227            throw new MonetaryException("Timeout loading exchange rate", e);
228        } catch (InterruptedException e) {
229            Thread.currentThread().interrupt(); // Restore interruption flag
230        }
231        return rate;
232    }
233
234    @Override
235    public CurrencyConversion getCurrencyConversion(ConversionQuery conversionQuery) {
236        return new LazyBoundCurrencyConversion(conversionQuery, this, ConversionContext
237                .of(getContext().getProviderName(), getContext().getRateTypes().iterator().next()));
238    }
239
240    private ExchangeRateChange buildExchangeRateChange(CurrencyUnitPair pair, Ticker ticker) {
241        Date date = ticker.getTimestamp();
242        // Not all exchanges provide a timestamp, default to 0 if it is null
243        long milliseconds = (date != null) ? date.getTime() : 0;
244
245        return new ExchangeRateChange(buildExchangeRate(pair, ticker), milliseconds);
246    }
247    
248    private ExchangeRate buildExchangeRate(CurrencyUnitPair pair, Ticker ticker) {
249        return new ExchangeRateBuilder(name, RateType.DEFERRED)
250                .setBase(pair.getBase())
251                .setTerm(pair.getTarget())
252                .setFactor(DefaultNumberValue.of(ticker.getLast()))
253                .build();
254    }
255
256    protected static class MonitoredCurrency {
257        final CurrencyUnitPair  pair;           // Terminating (target) JavaMoney CurrencyUnit
258        final CurrencyPair      exchangePair;   // XChange currency pair (format used by XChange/exchange)
259        final List<ExchangeRateObserver> observerList = new ArrayList<>();
260        private final UpdateableValueLatch<Ticker> tickerLatch = new UpdateableValueLatch<>();
261
262        MonitoredCurrency(CurrencyUnitPair pair, CurrencyPair exchangePair) {
263            this.pair = pair;
264            this.exchangePair = exchangePair;
265        }
266
267        boolean isTickerAvailable() {
268            return tickerLatch.isSet();
269        }
270
271        /**
272         * Get Ticker from the UpdateableValueLatch
273         * @return ticker object
274         * @throws InterruptedException thread was interrupted by another thread
275         * @throws TimeoutException should only happen if first request times out
276         */
277        Ticker getTicker() throws InterruptedException, TimeoutException {
278            return tickerLatch.getValue();
279        }
280
281        /**
282         * Set the ticker object, called from polling method
283         * @param ticker ticker object
284         */
285        void setTicker(Ticker ticker) {
286            tickerLatch.setValue(ticker);
287        }
288    }
289
290    /**
291     * A value latch that will block if value not yet set, but allows
292     * the value to be updated regularly (e.g. when polling for new values)
293     * @param <T>  Type of value to latch
294     */
295    static class UpdateableValueLatch<T> {
296        private volatile T value = null;
297        private final CountDownLatch ready = new CountDownLatch(1);
298        private final long timeoutSeconds;
299
300        UpdateableValueLatch() {
301            this(120);
302        }
303
304        UpdateableValueLatch(long timeoutInSeconds) {
305            this.timeoutSeconds = timeoutInSeconds;
306        }
307
308        T getValue() throws InterruptedException, TimeoutException {
309            // were we called before first poll completed?
310            if (value == null) {
311                // wait for the CountDownLatch
312                boolean loaded = ready.await(timeoutSeconds, TimeUnit.SECONDS);
313                if (!loaded) {
314                    throw new TimeoutException("Timeout before value set");
315                }
316            }
317            return value;
318        }
319
320        /* synchronized ? */ void setValue(T newValue) {
321            value = newValue;
322            ready.countDown();
323        }
324
325        boolean isSet() {
326            return value != null;
327        }
328    }
