001package com.msgilligan.bitcoinj.daemon.config;
002
003import com.googlecode.jsonrpc4j.spring.JsonServiceExporter;
004import com.msgilligan.bitcoinj.rpcserver.BitcoinJsonRpc;
005import com.msgilligan.bitcoinj.json.conversion.RpcServerModule;
006import com.msgilligan.bitcoinj.spring.service.PeerGroupService;
007import org.bitcoinj.core.NetworkParameters;
008import org.bitcoinj.net.discovery.DnsDiscovery;
009import org.bitcoinj.net.discovery.PeerDiscovery;
010import org.bitcoinj.params.MainNetParams;
011import org.springframework.context.annotation.Bean;
012import org.springframework.context.annotation.Configuration;
013import com.fasterxml.jackson.databind.Module;
014
015import java.io.FileNotFoundException;
016
017/**
018 * Spring configuration for bitcoinj, Bitcoin services, and JSON-RPC server
019 */
020@Configuration
021public class BitcoinConfig {
022    @Bean
023    public NetworkParameters networkParameters() {
024        return MainNetParams.get();
025    }
026
027    @Bean
028    public PeerDiscovery peerDiscovery(NetworkParameters params) throws FileNotFoundException {
029        PeerDiscovery pd;
030        pd = new DnsDiscovery(params);
031//        pd = new SeedPeers(params);
032        return pd;
033    }
034
035    @Bean
036    public Module bitcoinJMapper() {
037        return new RpcServerModule();
038    }
039
040    @Bean
041    public PeerGroupService peerGroupService(NetworkParameters params, PeerDiscovery peerDiscovery) {
042        return new PeerGroupService(params, peerDiscovery);
043    }
044
045    @Bean(name="/")
046    public JsonServiceExporter bitcoinServiceExporter(PeerGroupService peerGroupService) {
047        JsonServiceExporter exporter = new JsonServiceExporter();
048        exporter.setService(peerGroupService);
049        exporter.setServiceInterface(BitcoinJsonRpc.class);
050        exporter.setBackwardsComaptible(true);
051        return exporter;
052    }
053}
