001package com.msgilligan.bitcoinj.daemon.config;
002
003import org.springframework.web.servlet.support.AbstractAnnotationConfigDispatcherServletInitializer;
004
005import javax.servlet.ServletContext;
006import javax.servlet.ServletException;
007import javax.servlet.ServletRegistration;
008
009/**
010 * Configures the servlet environment without XML.
011 */
012public class DispatcherServletInitializer extends AbstractAnnotationConfigDispatcherServletInitializer {
013
014
015    @Override
016    protected Class<?>[] getRootConfigClasses() {
017        return new Class<?>[] {  };
018    }
019
020    @Override
021    protected Class<?>[] getServletConfigClasses() {
022        return new Class<?>[] { WebConfig.class };
023    }
024
025    @Override
026    protected String[] getServletMappings() {
027        return new String[] { "/" };
028    }
029
030    @Override
031    protected void customizeRegistration(ServletRegistration.Dynamic registration) {
032        registration.setInitParameter("dispatchOptionsRequest", "true");
033    }
034
035    @Override
036    public void onStartup(ServletContext servletContext) throws ServletException {
037        super.onStartup(servletContext);
038        servletContext.setInitParameter("org.apache.tomcat.websocket.binaryBufferSize", "60000");
039        servletContext.setInitParameter("org.apache.tomcat.websocket.textBufferSize", "60000");
040    }
041
042}
