001package com.msgilligan.bitcoinj.cli;
002
003import com.msgilligan.bitcoinj.json.pojo.BlockChainInfo;
004import com.msgilligan.bitcoinj.json.pojo.NetworkInfo;
005import org.consensusj.jsonrpc.JsonRPCException;
006
007import java.io.IOException;
008
009/**
010 * A command-line client that prints some basic information retrieved via RPC
011 */
012public class Status extends CliCommand {
013    public final static String commandName = "btcstatus";
014
015    public Status(String[] args) {
016        super(commandName, new CliOptions(), args);
017    }
018
019    public static void main(String[] args) {
020        Status command = new Status(args);
021        Integer status = command.run();
022        System.exit(status);
023    }
024
025    @Override
026    public Integer runImpl() throws IOException {
027        NetworkInfo networkInfo;
028        BlockChainInfo chainInfo;
029        try {
030            networkInfo = client.getNetworkInfo();
031            chainInfo = client.getBlockChainInfo();
032        } catch (JsonRPCException e) {
033            e.printStackTrace();
034            return 1;
035        }
036
037        Integer bitcoinVersion = networkInfo.getVersion();
038        Integer blocks = chainInfo.getBlocks();
039
040        pwout.println("Bitcoin Core Version: " + bitcoinVersion);
041        pwout.println("Block count: " + blocks);
042        return 0;
043    }
044}
