001package com.msgilligan.bitcoinj.cli.test;
002
003import com.msgilligan.bitcoinj.cli.CliCommand;
004
005import java.io.ByteArrayInputStream;
006import java.io.ByteArrayOutputStream;
007import java.io.InputStream;
008import java.io.PrintStream;
009import java.io.UnsupportedEncodingException;
010import java.nio.charset.StandardCharsets;
011
012/**
013 *  Support functions for testing command-line tools
014 */
015public interface CLITestSupport {
016    default String[] parseCommandLine(String line) {
017        String[] args = line.split(" ");     // (Overly?) simple parsing of string into args[]
018        return args;
019    }
020
021    /**
022     * Run a command and capture status and output
023     *
024     * @param command Command object instance to run
025     * @return Object containing status, stdout, stderr
026     */
027    default CLICommandResult runCommand(CliCommand command) throws UnsupportedEncodingException {
028        // Setup CommandResult to capture status and streams
029        CLICommandResult result = new CLICommandResult();
030        InputStream is = new ByteArrayInputStream(new byte[0]);
031        ByteArrayOutputStream bos = new ByteArrayOutputStream();
032        PrintStream cos = new PrintStream(bos);
033        ByteArrayOutputStream bes = new ByteArrayOutputStream();
034        PrintStream ces = new PrintStream(bes);
035
036        // Run the command
037        result.status = command.run(is, cos, ces);
038
039        // Put output and error streams in strings
040        String charset = StandardCharsets.UTF_8.toString();
041        result.output = bos.toString(charset);
042        result.error = bes.toString(charset);
043
044        return result;
045    }
046
047}
