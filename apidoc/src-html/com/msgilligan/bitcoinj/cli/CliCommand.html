001package com.msgilligan.bitcoinj.cli;
002
003import org.consensusj.jsonrpc.JsonRPCException;
004import org.apache.commons.cli.CommandLine;
005import org.apache.commons.cli.CommandLineParser;
006import org.apache.commons.cli.DefaultParser;
007import org.apache.commons.cli.HelpFormatter;
008import org.apache.commons.cli.ParseException;
009
010import java.io.IOException;
011import java.io.InputStream;
012import java.io.OutputStreamWriter;
013import java.io.PrintStream;
014import java.io.PrintWriter;
015import java.net.URI;
016import java.net.URISyntaxException;
017import java.nio.charset.StandardCharsets;
018
019import com.msgilligan.bitcoinj.rpc.BitcoinClient;
020import com.msgilligan.bitcoinj.rpc.RPCConfig;
021import org.bitcoinj.core.NetworkParameters;
022import org.bitcoinj.params.MainNetParams;
023import org.bitcoinj.params.RegTestParams;
024import org.bitcoinj.params.TestNet3Params;
025
026/**
027 * Base class for CLI commands that use Bitcoin RPC
028 */
029public abstract class CliCommand {
030    static final String defaultproto = "http";
031    static final String defaulthost = "127.0.0.1";
032    static final int defaultport = 8332;
033    static final String defaultfile = "/";
034
035    protected CommandLine line = null;
036    protected CommandLineParser parser = null;
037    protected CliOptions options;
038    protected String name;
039    protected String usage;
040    protected HelpFormatter formatter = null;
041
042    protected InputStream in;
043    protected PrintWriter pwout;
044    protected PrintWriter pwerr;
045
046    protected BitcoinClient client = null;
047
048    protected CliCommand(String name, CliOptions options, String[] args) {
049        this(name, null, options, args);
050    }
051
052    protected CliCommand(String name, String usage, CliOptions options, String[] args) {
053        this.name = name;
054        if (usage != null) {
055            this.usage = usage;
056        } else {
057            this.usage = name;
058        }
059        this.options = options;
060        parser = new DefaultParser();
061        try {
062            this.line = this.parser.parse(options, args);
063        } catch (ParseException e) {
064            e.printStackTrace();
065            throw new RuntimeException(e);
066        }
067    }
068
069    public BitcoinClient getClient() {
070        if (client == null) {
071            System.out.println("Connecting to: " + getRPCConfig().getURI());
072            RPCConfig config = getRPCConfig();
073            client = new BitcoinClient( config.getNetParams(),
074                                        config.getURI(),
075                                        config.getUsername(),
076                                        config.getPassword());
077        }
078        return client;
079    }
080
081    /**
082     * Can be used by subclasses to set a narrower type
083     * @param client An instance of BitcoinClient or subclass
084     */
085    protected void setClient(BitcoinClient client) {
086        this.client = client;
087    }
088
089    public void printHelp() {
090        if (formatter == null) {
091            formatter = new HelpFormatter();
092            formatter.setLongOptPrefix("-");
093        }
094        int leftPad = 4;
095        int descPad = 2;
096        int helpWidth = 120;
097        String header = "";
098        String footer = "";
099        formatter.printHelp(pwout, helpWidth, usage, header, options, leftPad, descPad, footer, false);
100    }
101
102    public void printError(String str) {
103        pwout.println(str);
104    }
105
106        /**
107         * Check Options and Arguments
108         *
109         * Override to customize behavior.
110         *
111         * @return status code
112         */
113    public Integer checkArgs() {
114        if (line.hasOption("?")) {
115            printHelp();
116            // Return 1 so tool can exit
117            return 1;
118        }
119        return 0;
120    }
121
122    /**
123     * Initial a client and if rpcwait option set, make sure server is accepting connections.
124     * @return status code
125     */
126    public Integer preflight() {
127        getClient();
128        if (line.hasOption("rpcwait")) {
129            boolean available = false;   // Wait up to 1 hour
130            try {
131                available = client.waitForServer(60*60);
132            } catch (JsonRPCException e) {
133                this.pwerr.println("JSON-RPC Exception: " + e.getMessage());
134                return 1;
135            }
136            if (!available) {
137                this.pwerr.println("Timeout error.");
138                return 1;
139            }
140        }
141        return 0;
142    }
143
144    public Integer run() {
145        return run(System.in, System.out, System.err);
146    }
147
148    public Integer run(InputStream in, PrintStream out, PrintStream err) {
149        this.in = in;
150        this.pwout = new PrintWriter(new OutputStreamWriter(out, StandardCharsets.UTF_8), true);
151        this.pwerr = new PrintWriter(new OutputStreamWriter(err, StandardCharsets.UTF_8), true);
152
153        Integer status = checkArgs();
154        if (status != 0) {
155            return status;
156        }
157
158        status = preflight();
159        if (status != 0) {
160            return status;
161        }
162
163        try {
164            status = runImpl();
165        } catch (JsonRPCException e) {
166            e.printStackTrace();
167            status = 1;
168        } catch (IOException e) {
169            e.printStackTrace();
170            status = 1;
171        }
172        return status;
173    }
174
175    /**
176     * Implement in subclasses
177     * @return status code
178     */
179    abstract protected Integer runImpl() throws IOException, JsonRPCException;
180
181    private URI getServerURI() {
182        String proto = defaultproto;
183        String host = defaulthost;
184        int port = defaultport;
185        String file = defaultfile;
186
187        if (line.hasOption("rpcssl")) {
188            proto = "https";
189        }
190        if (line.hasOption("rpcconnect")) {
191            host = line.getOptionValue("rpcconnect", defaulthost);
192        }
193        if (line.hasOption("regtest") || line.hasOption("testnet")) {
194            port = 18332;
195        }
196        if (line.hasOption("rpcport")) {
197            String portString = line.getOptionValue("rpcport");
198            port = Integer.parseInt(portString);
199        }
200        URI rpcServerURI = null;
201        try {
202            rpcServerURI = new URI(proto, null, host, port, file, null, null);
203        } catch (URISyntaxException e) {
204            // We should be careful that this never happens
205            e.printStackTrace();
206            // But if it does, throw an unchecked exception
207            throw new RuntimeException(e);
208        }
209        return rpcServerURI;
210    }
211
212    protected RPCConfig getRPCConfig() {
213        URI uri = getServerURI();
214        String user = line.getOptionValue("rpcuser", "");
215        String pass = line.getOptionValue("rpcpassword", "");
216        NetworkParameters netParams;
217        if (line.hasOption("regtest")) {
218            netParams = RegTestParams.get();
219        } else if (line.hasOption(("testnet"))) {
220            netParams = TestNet3Params.get();
221        } else {
222            netParams = MainNetParams.get();
223        }
224        RPCConfig cfg = new RPCConfig(netParams, uri, user, pass);
225        return cfg;
226    }
