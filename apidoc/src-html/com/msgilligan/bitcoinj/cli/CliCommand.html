027/**
028 * Base class for CLI commands that use Bitcoin RPC
029 */
030public abstract class CliCommand {
031    static final String defaultproto = "http";
032    static final String defaulthost = "127.0.0.1";
033    static final int defaultport = 8332;
034    static final String defaultfile = "/";
035
036    protected CommandLine line = null;
037    protected CommandLineParser parser = null;
038    protected CliOptions options;
039    protected String name;
040    protected String usage;
041    protected HelpFormatter formatter = null;
042
043    protected InputStream in;
044    protected PrintWriter pwout;
045    protected PrintWriter pwerr;
046
047    protected BitcoinClient client = null;
048
049    protected CliCommand(String name, CliOptions options, String[] args) {
050        this(name, null, options, args);
051    }
052
053    protected CliCommand(String name, String usage, CliOptions options, String[] args) {
054        this.name = name;
055        if (usage != null) {
056            this.usage = usage;
057        } else {
058            this.usage = name;
059        }
060        this.options = options;
061        parser = new DefaultParser();
062        try {
063            this.line = this.parser.parse(options, args);
064        } catch (ParseException e) {
065            e.printStackTrace();
066            throw new RuntimeException(e);
067        }
068    }
069
070    public BitcoinClient getClient() {
071        if (client == null) {
072            System.out.println("Connecting to: " + getRPCConfig().getURI());
073            RPCConfig config = getRPCConfig();
074            client = new BitcoinClient( config.getNetParams(),
075                                        config.getURI(),
076                                        config.getUsername(),
077                                        config.getPassword());
078        }
079        return client;
080    }
081
082    /**
083     * Can be used by subclasses to set a narrower type
084     * @param client An instance of BitcoinClient or subclass
085     */
086    protected void setClient(BitcoinClient client) {
087        this.client = client;
088    }
089
090    public void printHelp() {
091        if (formatter == null) {
092            formatter = new HelpFormatter();
093            formatter.setLongOptPrefix("-");
094        }
095        int leftPad = 4;
096        int descPad = 2;
097        int helpWidth = 120;
098        String header = "";
099        String footer = "";
100        formatter.printHelp(pwout, helpWidth, usage, header, options, leftPad, descPad, footer, false);
101    }
102
103    public void printError(String str) {
104        pwout.println(str);
105    }
106
107        /**
108         * Check Options and Arguments
109         *
110         * Override to customize behavior.
111         *
112         * @return status code
113         */
114    public Integer checkArgs() {
115        if (line.hasOption("?")) {
116            printHelp();
117            // Return 1 so tool can exit
118            return 1;
119        }
120        return 0;
121    }
122
123    /**
124     * Initial a client and if rpcwait option set, make sure server is accepting connections.
125     * @return status code
126     */
127    public Integer preflight() {
128        getClient();
129        if (line.hasOption("rpcwait")) {
130            boolean available = false;   // Wait up to 1 hour
131            try {
132                available = client.waitForServer(60*60);
133            } catch (JsonRPCException e) {
134                this.pwerr.println("JSON-RPC Exception: " + e.getMessage());
135                return 1;
136            }
137            if (!available) {
138                this.pwerr.println("Timeout error.");
139                return 1;
140            }
141        }
142        return 0;
143    }
144
145    public Integer run() {
146        return run(System.in, System.out, System.err);
147    }
148
149    public Integer run(InputStream in, PrintStream out, PrintStream err) {
150        this.in = in;
151        this.pwout = new PrintWriter(new OutputStreamWriter(out, StandardCharsets.UTF_8), true);
152        this.pwerr = new PrintWriter(new OutputStreamWriter(err, StandardCharsets.UTF_8), true);
153
154        Integer status = checkArgs();
155        if (status != 0) {
156            return status;
157        }
158
159        status = preflight();
160        if (status != 0) {
161            return status;
162        }
163
164        try {
165            status = runImpl();
166        } catch (JsonRPCException e) {
167            e.printStackTrace();
168            status = 1;
169        } catch (IOException e) {
170            e.printStackTrace();
171            status = 1;
172        }
173        return status;
174    }
175
176    /**
177     * Implement in subclasses
178     * @return status code
179     */
180    abstract protected Integer runImpl() throws IOException, JsonRPCException;
181
182    private URI getServerURI(URI confFileURI) {
183        String proto = confFileURI.getScheme();
184        String host = confFileURI.getHost();
185        int port = confFileURI.getPort();
186        String file = confFileURI.getPath();
187
188        if (line.hasOption("rpcssl")) {
189            proto = "https";
190        }
191        if (line.hasOption("rpcconnect")) {
192            host = line.getOptionValue("rpcconnect", host);
193        }
194        if (line.hasOption("regtest") || line.hasOption("testnet")) {
195            port = 18332;
196        }
197        if (line.hasOption("rpcport")) {
198            String portString = line.getOptionValue("rpcport");
199            port = Integer.parseInt(portString);
200        }
201        URI rpcServerURI = null;
202        try {
203            rpcServerURI = new URI(proto, null, host, port, file, null, null);
204        } catch (URISyntaxException e) {
205            // We should be careful that this never happens
206            e.printStackTrace();
207            // But if it does, throw an unchecked exception
208            throw new RuntimeException(e);
209        }
210        return rpcServerURI;
211    }
212
213    protected RPCConfig getRPCConfig() {
214        RPCConfig confFileConfig = BitcoinConfFile.readDefaultConfig().getRPCConfig();
215        URI uri = getServerURI(confFileConfig.getURI());
216        String user = line.getOptionValue("rpcuser", confFileConfig.getUsername());
217        String pass = line.getOptionValue("rpcpassword", confFileConfig.getPassword());
218        NetworkParameters netParams;
219        if (line.hasOption("regtest")) {
220            netParams = RegTestParams.get();
221        } else if (line.hasOption(("testnet"))) {
222            netParams = TestNet3Params.get();
223        } else {
224            netParams = MainNetParams.get();
225        }
226        RPCConfig cfg = new RPCConfig(netParams, uri, user, pass);
227        return cfg;
228    }
