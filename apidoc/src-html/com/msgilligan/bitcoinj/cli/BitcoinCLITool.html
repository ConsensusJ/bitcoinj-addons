001package com.msgilligan.bitcoinj.cli;
002
003import org.consensusj.jsonrpc.JsonRPCException;
004
005import java.io.IOException;
006import java.util.ArrayList;
007import java.util.List;
008
009/**
010 * An attempt at cloning the bitcoin-cli tool, but using Java and bitcoinj
011 *
012 */
013public class BitcoinCLITool extends CliCommand {
014    public final static String commandName = "cj-bitcoin-cli";
015
016    public BitcoinCLITool(String[] args) {
017        super(commandName, new CliOptions(), args);
018    }
019
020    /**
021     * main method for bitcoinj-cli tool.
022     *
023     * See {@link CliOptions} for options and https://bitcoin.org/en/developer-reference#bitcoin-core-apis[Bitcoin Core JSON-RPC API]
024     * for the methods and parameters. Users can use `-?` to get general help or `help <command>` to get help
025     * on a specific command.
026     *
027     * @param args options, JSON-RPC method, JSON-RPC parameters
028     */
029    public static void main(String[] args) {
030        BitcoinCLITool command = new BitcoinCLITool(args);
031        Integer status = command.run();
032        System.exit(status);
033    }
034
035    public Integer runImpl() throws IOException {
036        List<String> args = line.getArgList();
037        if (args.size() == 0) {
038            printError("rpc method required");
039            printHelp();
040            return(1);
041        }
042        String method = args.get(0);
043        args.remove(0); // remove method from list
044        List<Object> typedArgs = convertParameters(method, args);
045        Object result;
046        try {
047            result = client.send(method, typedArgs);
048        } catch (JsonRPCException e) {
049            e.printStackTrace();
050            return 1;
051        }
052        if (result != null) {
053            pwout.println(result.toString());
054        }
055        return 0;
056    }
057
058    /**
059     * Convert params from strings to Java types that will map to correct JSON types
060     *
061     * TODO: Make this better and complete
062     *
063     * @param method the JSON-RPC method
064     * @param params Params with String type
065     * @return Params with correct Java types for JSON
066     */
067    protected List<Object> convertParameters(String method, List<String> params) {
068        List<Object> typedParams = new ArrayList<>();
069        switch (method) {
070            case "setgenerate":
071                typedParams.add(0, Boolean.parseBoolean(params.get(0)));
072                break;
073
074            default:
075                // Default (for now) is to leave them all as strings
076                for (String string : params) {
077                    typedParams.add(string);
078                }
079
080        }
081        return typedParams;
082    }
083}
