001package com.msgilligan.bitcoinj.cli;
002
003import org.apache.commons.cli.Option;
004import org.apache.commons.cli.OptionGroup;
005import org.apache.commons.cli.Options;
006
007/**
008 * Command-line options for tools that communicate with Bitcoin RPC
009 */
010public class CliOptions extends Options {
011
012
013    public CliOptions() {
014        super();
015        this.addOption("?", null, false, "This help message")
016// 'conf' and 'datadir' aren't implemented yet.
017//            .addOption("c", "conf", true, "Specify configuration file (default: bitcoin.conf)")
018//            .addOption("d", "datadir", true, "Specify data directory")
019            .addOptionGroup(new OptionGroup()
020                    .addOption(new Option(null, "testnet", false, "Use the test network"))
021                    .addOption(new Option(null, "regtest", false, "Enter regression test mode")))
022            .addOption(Option.builder().longOpt("rpcconnect")
023                    .desc("Send commands to node running on <ip> (default: 127.0.0.1)")
024                    .hasArg()
025                    .argName("ip")
026                    .build())
027                .addOption(Option.builder().longOpt("rpcport")
028                        .desc("Connect to JSON-RPC on <port> (default: 8332 or testnet: 18332)")
029                        .hasArg()
030                        .argName("port")
031                        .build())
032                .addOption(null, "rpcwait", false, "Wait for RPC server to start")
033                .addOption(Option.builder().longOpt("rpcuser")
034                        .desc("Username for JSON-RPC connections")
035                        .hasArg()
036                        .argName("user")
037                        .build())
038                .addOption(Option.builder().longOpt("rpcpassword")
039                        .desc("Password for JSON-RPC connections")
040                        .hasArg()
041                        .argName("pw")
042                        .build())
043            .addOption(null, "rpcssl", false, "Use https for JSON-RPC connections");
044    }
045
046}
