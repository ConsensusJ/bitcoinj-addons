001package com.msgilligan.bitcoinj.spring.controller;
002
003/**
004 * REST and WebSocket (STOMP) Spring MVC Peer Controller
005 */
006import com.msgilligan.bitcoinj.spring.services.PeerStompService;
007import org.bitcoinj.core.Peer;
008import org.springframework.beans.factory.annotation.Autowired;
009import org.springframework.messaging.handler.annotation.MessageMapping;
010import org.springframework.web.bind.annotation.ResponseBody;
011import org.springframework.web.bind.annotation.RestController;
012import org.springframework.web.bind.annotation.RequestMapping;
013
014import java.security.Principal;
015import java.util.List;
016
017
018@RestController
019public class PeerController {
020    private final PeerStompService peerService;
021
022    @Autowired
023    public PeerController(PeerStompService peerService) {
024        this.peerService = peerService;
025    }
026
027
028    @RequestMapping("/hello")
029    public String index() {
030        return "Greetings from PeerList! We're running on: " + peerService.getNetworkParameters().getId() + ".";
031    }
032
033    @MessageMapping("/listPeers")
034    public void listPeers(Principal principal) {
035        peerService.listPeers(principal);
036    }
037
038    @RequestMapping("/listPeers")
039    @ResponseBody
040    public List<Peer> listPeersREST() {
041        return peerService.getPeers();
042    }
043}
