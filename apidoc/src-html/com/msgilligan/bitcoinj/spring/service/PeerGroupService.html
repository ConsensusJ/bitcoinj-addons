001package com.msgilligan.bitcoinj.spring.service;
002
003import com.msgilligan.bitcoinj.json.pojo.ServerInfo;
004import com.msgilligan.bitcoinj.rpcserver.BitcoinJsonRpc;
005import org.bitcoinj.core.Coin;
006import org.bitcoinj.core.NetworkParameters;
007import org.bitcoinj.core.PeerGroup;
008import org.bitcoinj.core.Transaction;
009import org.bitcoinj.net.discovery.PeerDiscovery;
010
011import javax.annotation.PostConstruct;
012import javax.inject.Inject;
013import javax.inject.Named;
014import java.math.BigDecimal;
015
016/**
017 * Implement a subset of Bitcoin JSON RPC using only a PeerGroup
018 */
019@Named
020public class PeerGroupService implements BitcoinJsonRpc {
021    private static final String userAgentName = "PeerList";
022    private static final String appVersion = "0.1";
023    private static final int version = 1;
024    private static final int protocolVersion = 1;
025    private static final int walletVersion = 0;
026
027    protected NetworkParameters netParams;
028    protected PeerGroup peerGroup;
029    private int timeOffset = 0;
030    private BigDecimal difficulty = new BigDecimal(0);
031
032    @Inject
033    public PeerGroupService(NetworkParameters params,
034                       PeerDiscovery peerDiscovery) {
035        this.netParams = params;
036        this.peerGroup = new PeerGroup(params);
037        peerGroup.setUserAgent(userAgentName, appVersion);
038        peerGroup.addPeerDiscovery(peerDiscovery);
039    }
040
041    @PostConstruct
042    public void start() {
043        peerGroup.startAsync();
044    }
045
046    public NetworkParameters getNetworkParameters() {
047        return this.netParams;
048    }
049
050    @Override
051    public Integer getblockcount() {
052        return peerGroup.getMostCommonChainHeight();
053    }
054
055    @Override
056    public Integer getconnectioncount() {
057        return peerGroup.numConnectedPeers();
058    }
059
060    @Override
061    public ServerInfo getinfo() {
062        // Dummy up a response for now.
063        // Since ServerInfo is immutable, we have to build it entirely with the constructor.
064        Coin balance = Coin.valueOf(0);
065        boolean testNet = !netParams.getId().equals(NetworkParameters.ID_MAINNET);
066        int keyPoolOldest = 0;
067        int keyPoolSize = 0;
068        return new ServerInfo(
069                version,
070                protocolVersion,
071                walletVersion,
072                balance,
073                getblockcount(),
074                timeOffset,
075                getconnectioncount(),
076                "proxy",
077                difficulty,
078                testNet,
079                keyPoolOldest,
080                keyPoolSize,
081                Transaction.REFERENCE_DEFAULT_MIN_TX_FEE,
082                Transaction.REFERENCE_DEFAULT_MIN_TX_FEE, // relayfee
083                "no errors"                               // errors
084        );
085    }
086
087}
