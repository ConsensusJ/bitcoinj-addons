016package com.msgilligan.bitcoinj.spring.services;
017
018import org.bitcoinj.core.listeners.OnTransactionBroadcastListener;
019import org.bitcoinj.core.listeners.PeerConnectedEventListener;
020import org.bitcoinj.core.listeners.PeerDisconnectedEventListener;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023import org.springframework.messaging.simp.SimpMessageSendingOperations;
024
025import org.bitcoinj.core.*;
026
027import javax.annotation.PostConstruct;
028import javax.inject.Inject;
029import javax.inject.Named;
030import java.security.Principal;
031import java.util.List;
032
033/**
034 * Use a PeerGroup to provide REST and WebSocket/STOMP services
035 * for Peers and Transactions
036 */
037@Named
038public class PeerStompService implements PeerConnectedEventListener, PeerDisconnectedEventListener, OnTransactionBroadcastListener {
039    private static final Logger log = LoggerFactory.getLogger(PeerStompService.class);
040    private final PeerGroup peerGroup;
041    private final SimpMessageSendingOperations messagingTemplate;
042    private final Context context;
043
044    @Inject
045    public PeerStompService(Context context,
046                            PeerGroup peerGroup,
047                            SimpMessageSendingOperations messagingTemplate) {
048        this.peerGroup = peerGroup;
049        this.context = context;
050        this.messagingTemplate = messagingTemplate;
051        //this.executorService = Executors.newFixedThreadPool(6, new ContextPropagatingThreadFactory("StompServiceListeners"));
052        log.info("PeerStompService: Constructed with netparams: {}", context.getParams().getId());
053    }
054
055    @PostConstruct
056    public void start() {
057        log.info("PeerStompService: start()");
058        log.debug("PeerStompService: context from thread is: {}", Context.get().getParams().getId());
059//        peerGroup.start();    // Currently this is done in BitcoinConfig
060        peerGroup.addConnectedEventListener(this);
061        peerGroup.addDisconnectedEventListener(this);
062        peerGroup.addOnTransactionBroadcastListener(this);
063    }
064    
065    public NetworkParameters getNetworkParameters() {
066        return context.getParams();
067    }
068
069    public List<Peer> getPeers() {
070        return peerGroup.getConnectedPeers();
071    }
072
073    public void listPeers(Principal principal) {
074        List<Peer> peers = peerGroup.getConnectedPeers();
075        this.messagingTemplate.convertAndSendToUser(principal.getName(), "/queue/peers", peers);
076    }
077
078    @Override
079    public void onTransaction(Peer peer, Transaction tx) {
080        log.info("PeerStompService: Got transaction: {}", tx.getTxId());
081        this.messagingTemplate.convertAndSend("/topic/tx", tx);
082    }
083
084    @Override
085    public void onPeerConnected(Peer peer, int peerCount) {
086        log.info("PeerStompService: Peer Connected: {}, count: {}", peer.getAddress(), peerCount);
087    }
088
089    @Override
090    public void onPeerDisconnected(Peer peer, int peerCount) {
091        log.info("PeerStompService: Peer Disconnected, count: {}", peerCount);
092    }
093}
