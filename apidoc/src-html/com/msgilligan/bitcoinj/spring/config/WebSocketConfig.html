001package com.msgilligan.bitcoinj.spring.config;
002
003import com.fasterxml.jackson.databind.ObjectMapper;
004import org.springframework.beans.factory.annotation.Autowired;
005import org.springframework.context.annotation.ComponentScan;
006import org.springframework.context.annotation.Configuration;
007import org.springframework.messaging.converter.DefaultContentTypeResolver;
008import org.springframework.messaging.converter.MappingJackson2MessageConverter;
009import org.springframework.messaging.converter.MessageConverter;
010import org.springframework.messaging.handler.invocation.HandlerMethodArgumentResolver;
011import org.springframework.messaging.handler.invocation.HandlerMethodReturnValueHandler;
012import org.springframework.messaging.simp.config.ChannelRegistration;
013import org.springframework.messaging.simp.config.MessageBrokerRegistry;
014import org.springframework.util.MimeTypeUtils;
015import org.springframework.web.socket.config.annotation.EnableWebSocketMessageBroker;
016import org.springframework.web.socket.config.annotation.WebSocketTransportRegistration;
017import org.springframework.web.socket.config.annotation.StompEndpointRegistry;
018import org.springframework.web.socket.config.annotation.WebSocketMessageBrokerConfigurer;
019
020import java.util.List;
021
022/**
023 */
024@Configuration
025@EnableWebSocketMessageBroker
026@ComponentScan(basePackages="com.msgilligan.bitcoinj.spring")
027public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {
028    @Autowired ObjectMapper objectMapper;   // Seems to get injected with objectMapper with our added Module
029
030    @Override
031    public void registerStompEndpoints(StompEndpointRegistry registry) {
032        registry.addEndpoint("/stomp").withSockJS();
033    }
034
035    @Override
036    public void configureWebSocketTransport(WebSocketTransportRegistration registration) {
037        registration.setSendTimeLimit(15 * 1000).setSendBufferSizeLimit(512 * 1024);
038    }
039
040    @Override
041    public void configureClientInboundChannel(ChannelRegistration registration) {
042    }
043
044    @Override
045    public void configureClientOutboundChannel(ChannelRegistration registration) {
046        registration.taskExecutor().corePoolSize(4).maxPoolSize(10);
047    }
048
049    @Override
050    public void addArgumentResolvers(List<HandlerMethodArgumentResolver> argumentResolvers) {
051        // TODO: ??
052    }
053
054    @Override
055    public void addReturnValueHandlers(List<HandlerMethodReturnValueHandler> returnValueHandlers) {
056        // TODO: ??
057    }
058
059    @Override
060    public boolean configureMessageConverters(List<MessageConverter> messageConverters) {
061        // Workaround for issue 2445: https://github.com/spring-projects/spring-boot/issues/2445
062        DefaultContentTypeResolver resolver = new DefaultContentTypeResolver();
063        resolver.setDefaultMimeType(MimeTypeUtils.APPLICATION_JSON);
064        MappingJackson2MessageConverter converter = new MappingJackson2MessageConverter();
065        converter.setObjectMapper(objectMapper);
066        converter.setContentTypeResolver(resolver);
067        messageConverters.add(converter);
068        return false;
069    }
070
071    @Override
072    public void configureMessageBroker(MessageBrokerRegistry configurer) {
073        configurer.enableSimpleBroker("/queue/", "/topic/");
074        configurer.setApplicationDestinationPrefixes("/app");
075    }
076
077}
