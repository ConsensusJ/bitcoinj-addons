001package com.msgilligan.bitcoinj.spring.config;
002
003import org.springframework.context.annotation.Bean;
004import org.springframework.context.annotation.Configuration;
005import org.springframework.security.config.annotation.web.builders.HttpSecurity;
006import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
007import org.springframework.security.config.annotation.web.configuration.WebSecurityConfigurerAdapter;
008import org.springframework.security.core.userdetails.User;
009import org.springframework.security.core.userdetails.UserDetailsService;
010import org.springframework.security.crypto.password.NoOpPasswordEncoder;
011import org.springframework.security.provisioning.InMemoryUserDetailsManager;
012
013/**
014 */
015@Configuration
016@EnableWebSecurity
017public class WebSecurityConfig extends WebSecurityConfigurerAdapter {
018
019    @Override
020    protected void configure(HttpSecurity http) throws Exception {
021        http
022                .csrf().disable() //TODO Refactor login form
023                .authorizeRequests()
024                .antMatchers("/style/**").permitAll()
025                .antMatchers("/lib/**").permitAll()
026                .antMatchers("/webjars/**").permitAll()
027                .antMatchers("/bitcoinrpc").permitAll()
028
029//                .antMatchers("/stomp/**").permitAll()
030                .anyRequest().authenticated()
031                .and()
032                .logout()
033                .logoutSuccessUrl("/login.html?logout")
034                .logoutUrl("/logout.html")
035                .permitAll()
036                .and()
037                .formLogin()
038                .defaultSuccessUrl("/index.html")
039                .loginPage("/login.html")
040                .failureUrl("/login.html?error")
041                .permitAll();
042    }
043
044
045    @Bean
046    public UserDetailsService userDetailsService() {
047        InMemoryUserDetailsManager manager = new InMemoryUserDetailsManager();
048        manager.createUser(User.withUsername("admin").password("admin").roles("ADMIN", "USER").build());
049        return manager;
050    }
051
052
053    @SuppressWarnings("deprecation")
054    @Bean
055    public static NoOpPasswordEncoder passwordEncoder() {
056        return (NoOpPasswordEncoder) NoOpPasswordEncoder.getInstance();
057    }
058}
