001package com.msgilligan.bitcoinj.spring.config;
002
003import com.googlecode.jsonrpc4j.spring.JsonServiceExporter;
004import com.msgilligan.bitcoinj.rpcserver.BitcoinJsonRpc;
005import com.msgilligan.bitcoinj.json.conversion.RpcServerModule;
006import org.bitcoinj.core.PeerGroup;
007import org.consensusj.bitcoin.services.WalletAppKitService;
008import org.bitcoinj.core.Context;
009import org.bitcoinj.core.NetworkParameters;
010import org.bitcoinj.kits.WalletAppKit;
011import org.bitcoinj.net.discovery.DnsDiscovery;
012import org.bitcoinj.net.discovery.PeerDiscovery;
013import org.bitcoinj.params.MainNetParams;
014import com.msgilligan.bitcoinj.spring.services.PeerStompService;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017import org.springframework.context.annotation.Bean;
018import org.springframework.context.annotation.Configuration;
019import com.fasterxml.jackson.databind.Module;
020import org.springframework.messaging.simp.SimpMessageSendingOperations;
021
022import java.io.File;
023import java.io.FileNotFoundException;
024
025/**
026 * Spring configuration for bitcoinj, Bitcoin services, and JSON-RPC server
027 */
028@Configuration
029public class BitcoinConfig {
030    private static final Logger log = LoggerFactory.getLogger(BitcoinConfig.class);
031
032    @Bean
033    public NetworkParameters networkParameters() {
034        return MainNetParams.get();
035    }
036
037    @Bean
038    public PeerDiscovery peerDiscovery(NetworkParameters params) throws FileNotFoundException {
039        PeerDiscovery pd;
040        pd = new DnsDiscovery(params);
041//        pd = new SeedPeers(params);
042        return pd;
043    }
044
045    @Bean
046    PeerGroup peerGroup(Context context, PeerDiscovery peerDiscovery) {
047        PeerGroup peerGroup = new PeerGroup(context);
048        peerGroup.addPeerDiscovery(peerDiscovery);
049        peerGroup.start();
050        return peerGroup;
051    }
052
053    @Bean
054    public Context getContext(NetworkParameters params) {
055        log.info("Creating context with {}", params.getId());
056        Context context = new Context(params);
057        log.info("Context is {}", context.getParams().getId());
058        return context;
059    }
060    
061    @Bean
062    public WalletAppKit getKit(NetworkParameters params) throws Exception {
063        // TODO: make File(".") and filePrefix configurable
064        File directory = new File(".");
065        String filePrefix = "ConensusJ-BTC-Daemon";
066
067        return new WalletAppKit(params, directory, filePrefix);
068    }
069
070    @Bean
071    public Module bitcoinJMapper(NetworkParameters params) {
072        return new RpcServerModule(params);
073    }
074    
075    @Bean
076    public WalletAppKitService walletAppKitService(NetworkParameters params, Context context, WalletAppKit kit) {
077        log.info("WalletAppKitService factory params are {}", params.getId());
078        log.info("WalletAppKitService factory context is {}", context.getParams().getId());
079        return new WalletAppKitService(params, context, kit);
080    }
081
082    @Bean
083    PeerStompService peerStompService(Context context,
084                                      PeerGroup peerGroup,
085                                      SimpMessageSendingOperations messagingTemplate) {
086        return new PeerStompService(context, peerGroup, messagingTemplate);
087    }
088
089    @Bean(name="/jsonrpc")
090    public JsonServiceExporter bitcoinServiceExporter(WalletAppKitService walletAppKitService) {
091        JsonServiceExporter exporter = new JsonServiceExporter();
092        exporter.setService(walletAppKitService);
093        exporter.setServiceInterface(BitcoinJsonRpc.class);
094        exporter.setBackwardsComaptible(true);
095        return exporter;
096    }
097}
