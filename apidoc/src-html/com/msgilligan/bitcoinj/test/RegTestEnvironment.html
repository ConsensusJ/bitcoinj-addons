001package com.msgilligan.bitcoinj.test;
002
003import com.msgilligan.bitcoinj.rpc.BitcoinClient;
004import org.consensusj.jsonrpc.JsonRpcException;
005import org.bitcoinj.core.Sha256Hash;
006
007import java.io.IOException;
008import java.util.List;
009
010/**
011 *  Single-threaded RegTest testing environment.
012 *  (It is conceivable someone could write tests that run in parallel in RegTest mode
013 *  just as they would on MainNet or Testnet. There would be a timer that creates blocks
014 *  every 'n' seconds, so tests could run in parallel, but would complete much quicker.
015 */
016public class RegTestEnvironment implements BlockChainEnvironment {
017    private BitcoinClient client;
018
019    public RegTestEnvironment(BitcoinClient client) {
020        this.client = client;
021    }
022
023    @Override
024    public List<Sha256Hash> waitForBlock() throws Exception {
025        return waitForBlocks(1);
026    }
027
028    @Override
029    public List<Sha256Hash> waitForBlocks(int numBlocks) throws JsonRpcException, IOException {
030        return client.generate(numBlocks);
031    }
032}
