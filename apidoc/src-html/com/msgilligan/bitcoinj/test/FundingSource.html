001package com.msgilligan.bitcoinj.test;
002
003import org.bitcoinj.core.Address;
004import org.bitcoinj.core.Coin;
005import org.bitcoinj.core.Sha256Hash;
006
007/**
008 * A source of Bitcoin funds for testing
009 *
010 * In RegTest mode, it can be a RegTestFundingSource that mines coins in RegTest mode and sends them
011 * to a requesting address. In other modes it can be a TestWallet preloaded with a certain amount of coins.
012 */
013public interface FundingSource {
014    Sha256Hash requestBitcoin(Address toAddress, Coin requestedAmount) throws Exception;
015    Address createFundedAddress(Coin amount) throws Exception;
016
017    /**
018     * An opportunity to do any necessary housekeeping. (e.g. consolidation)
019     */
020    void fundingSourceMaintenance();
021}
