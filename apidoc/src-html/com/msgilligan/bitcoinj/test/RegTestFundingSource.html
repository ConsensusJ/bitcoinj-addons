027/**
028 * FundingSource using RegTest mining
029 */
030public class RegTestFundingSource implements FundingSource {
031    final Integer defaultMaxConf = 9999999;
032    private static final Logger log = LoggerFactory.getLogger(RegTestFundingSource.class);
033    protected BitcoinExtendedClient client;
034    /**
035     * Prior to Bitcoin Core 0.19, the second parameter of sendRawTransaction
036     * was a boolean, not an integer containing maxFees
037     */
038    private boolean serverHasSendRawWithMaxFees = true;
039
040    public RegTestFundingSource(BitcoinExtendedClient client) {
041        this.client = client;
042    }
043
044    /**
045     * Check for and internally handle Bitcoin Core pre 0.19
046     * (Note that Omni Core 0.8.x is based on Bitcoin Core 0.18)
047     * @return `true` (if legacy), `false` (if modern), `null` (if error)
048     * @deprecated This method will be removed in a future release and Bitcoin Core 0.19+ will be required
049     */
050    @Deprecated
051    public Boolean checkForLegacyBitcoinCore() {
052        Boolean isLegacy = null;
053        try {
054            NetworkInfo networkInfo = client.getNetworkInfo();
055            isLegacy = networkInfo.getVersion() < 190000;
056        } catch (IOException e) {
057            log.error("Exception: ", e);
058        }
059        if (isLegacy != null && isLegacy) {
060            serverHasSendRawWithMaxFees = false;
061        }
062        return isLegacy;
063    }
064
065    /**
066     * Generate blocks and fund an address with requested amount of BTC
067     *
068     * TODO: Improve performance. Can we mine multiple blocks with a single RPC?
069     * TODO: Use client.generateToAddress() directly rather than through client.generateBlocks()
070     * If we use `toAddress` as the destination of generateToAddress(), we
071     * can skip the generation and sending of the the raw transaction below.
072     *
073     * @param toAddress Address to fund with BTC
074     * @param requestedBtc Amount of BTC to "mine" and send (minimum ending balance of toAddress?)
075     * @return The hash of transaction that provided the funds.
076     */
077    @Override
078    public Sha256Hash requestBitcoin(Address toAddress, Coin requestedBtc) throws JsonRpcException, IOException {
079        log.debug("requestBitcoin requesting {}", requestedBtc);
080        if (requestedBtc.value > NetworkParameters.MAX_MONEY.value) {
081            throw new IllegalArgumentException("request exceeds MAX_MONEY");
082        }
083        long amountGatheredSoFar = 0;
084        ArrayList<Outpoint> inputs = new ArrayList<>();
085
086        // Newly mined coins need to mature to be spendable
087        final int minCoinAge = 100;
088
089        if (client.getBlockCount() < minCoinAge) {
090            client.generateBlocks(minCoinAge - client.getBlockCount());
091        }
092
093        while (amountGatheredSoFar < requestedBtc.value) {
094            client.generateBlocks(1);
095            int blockIndex = client.getBlockCount() - minCoinAge;
096            Block block = client.getBlock(blockIndex);
097            List<Transaction> blockTxs = block.getTransactions();
098            Sha256Hash coinbaseTx = blockTxs.get(0).getTxId();
099            TxOutInfo txout = client.getTxOut(coinbaseTx, 0);
100
101            // txout is empty, if output was already spent
102            if (txout != null && txout.getValue().value > 0) {
103                log.debug("txout = {}, value = {}", txout, txout.getValue().value);
104
105                amountGatheredSoFar += txout.getValue().value;
106                inputs.add(new Outpoint(coinbaseTx, 0));
107            }
108            log.debug("amountGatheredSoFar = {}", BitcoinMath.satoshiToBtc(amountGatheredSoFar));
109        }
110
111        // Don't care about change, we mine it anyway
112        String unsignedTxHex = client.createRawTransaction(inputs, Collections.singletonMap(toAddress, requestedBtc));
113        SignedRawTransaction signingResult = client.signRawTransactionWithWallet(unsignedTxHex);
114
115        assert signingResult.isComplete();
116
117        String signedTxHex = signingResult.getHex();
118        Sha256Hash txid = sendRawTransactionUnlimitedFees(signedTxHex);
119
120        return txid;
121    }
122
123    /**
124     * Create an address and fund it with bitcoin
125     *
126     * @param amount
127     * @return Newly created address with the requested amount of bitcoin
128     */
129    public Address createFundedAddress(Coin amount) throws Exception {
130        Address address = client.getNewAddress();
131        requestBitcoin(address, amount);
132        return address;
133    }
134
135    /**
136     * Create everything needed to assemble a custom transaction
137     * @param amount Amount of BTC to be available on new address
138     * @return An address, private key, and list of unspent outputs
139     * @throws JsonRpcException
140     * @throws IOException
141     */
142    public TransactionIngredients createIngredients(Coin amount) throws JsonRpcException, IOException {
143        TransactionIngredients ingredients = new TransactionIngredients();
144        Address address = client.getNewAddress();
145        requestBitcoin(address, amount);
146        ingredients.address = address;
147        ingredients.privateKey = client.dumpPrivKey(address);
148        ingredients.outPoints = client.listUnspentOutPoints(address);
149        return ingredients;
150    }
151
152    @Override
153    public void fundingSourceMaintenance() {
154        try {
155            consolidateCoins();
156        } catch (JsonRpcException e) {
157            e.printStackTrace();
158            throw new RuntimeException(e);
159        } catch (IOException e) {
160            e.printStackTrace();
161            throw new RuntimeException(e);
162        }
163    }
164
165    /**
166     * Collects *all* unspent outputs and spends the whole amount minus `stdRelayTxFee`, which is sent
167     * to a new address, as fee, to sweep dust and to minimize the number of unspent outputs, to avoid creating too
168     * large transactions. No new block is generated afterwards.
169     *
170     * Can be used in cleanupSpec() methods of integration tests.
171     *
172     * @see <a href="https://github.com/OmniLayer/OmniJ/issues/50">Issue #50 on GitHub</a>
173     *
174     * @return True, if enough outputs with a value of at least {@code stdRelayTxFee} were spent
175     */
176     void consolidateCoins() throws JsonRpcException, IOException {
177        long amountIn = 0;
178        List<Outpoint> inputs = new ArrayList<Outpoint>();
179        List<UnspentOutput> unspentOutputs = client.listUnspent(1,defaultMaxConf);
180
181        // Gather inputs
182        for (UnspentOutput unspentOutput : unspentOutputs) {
183            amountIn += unspentOutput.getAmount().value;
184            inputs.add(new Outpoint(unspentOutput.getTxid(), unspentOutput.getVout()));
185
186        }
187
188        // Check if there is a sufficient high amount to sweep at all
189        if (amountIn < client.stdRelayTxFee.value) {
190            return; //false;
191        }
192
193        // No receiver, just spend most of it as fee (!)
194        Map<Address,Coin> outputs = new HashMap<>();
195        outputs.put(client.getNewAddress(), client.stdRelayTxFee);
196
197        String unsignedTxHex = client.createRawTransaction(inputs, outputs);
198        SignedRawTransaction signingResult = client.signRawTransactionWithWallet(unsignedTxHex);
199
200        boolean complete = signingResult.isComplete();
201        assert complete;
202
203        String signedTxHex = signingResult.getHex();
204        Sha256Hash txid = sendRawTransactionUnlimitedFees(signedTxHex);
205
206        return; //true;
207    }
208
209    private Sha256Hash sendRawTransactionUnlimitedFees(String hexTx) throws IOException {
210        Sha256Hash txid;
211        if (serverHasSendRawWithMaxFees) {
212            txid = client.sendRawTransaction(hexTx, Coin.ZERO);
213        } else {
214            txid = client.sendRawTransaction(hexTx, true);
215        }
216        return txid;
217    }
