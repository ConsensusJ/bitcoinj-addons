001package com.msgilligan.bitcoinj.rpc.bitcoind;
002
003import org.slf4j.Logger;
004import org.slf4j.LoggerFactory;
005
006import java.io.File;
007import java.io.IOException;
008import java.nio.charset.Charset;
009import java.nio.file.Files;
010import java.nio.file.NoSuchFileException;
011import java.util.List;
012
013/**
014 * Currently focused on just getting connection info
015 */
016public class BitcoinConfFile {
017    private static final Logger log = LoggerFactory.getLogger(BitcoinConfFile.class);
018    private static String BITCOINAPPNAME = "Bitcoin";
019    private File file;
020
021    public BitcoinConfFile() {
022        this(new File(AppDataDirectory.forAppName(BITCOINAPPNAME), "bitcoin.conf"));
023    }
024
025    public BitcoinConfFile(File confFile) {
026        file = confFile;
027    }
028
029    public static BitcoinConf readDefaultConfig() {
030        return new BitcoinConfFile().readWithFallback();
031    }
032
033    //
034
035    /**
036     * Read a `bitcoin.conf` file
037     *
038     * @return The configuration object
039     */
040    public BitcoinConf read() throws IOException {
041        List<String> lines = Files.readAllLines(file.toPath(), Charset.defaultCharset() );
042        return parseLines(lines);
043    }
044
045    /**
046     * Try to read `.conf` file, fallback to defaults on error
047     *
048     * TODO: If any exception occurs we return default values
049     * maybe we should only do this on fileNotFound or accessDenied, etc.
050     * 
051     * @return Configuration read or defaults if read error
052     */
053    public BitcoinConf readWithFallback() {
054        BitcoinConf conf = new BitcoinConf();
055        setDefaults(conf);
056
057        try {
058            conf = read();
059        } catch (IOException e) {
060            if (e instanceof NoSuchFileException) {
061                log.warn("NoSuchFileException: " + file.getAbsolutePath()   );
062            } else {
063                log.error("Error reading " + file.getAbsolutePath() + ": " + e.getMessage());
064                e.printStackTrace();
065            }
066        }
067        return conf;
068    }
069
070    /**
071     * Not-very-well tested parsing of Bitcoin.conf (or similar) file
072     * 
073     * @param lines An array of one-line-Strings from the file
074     * @return a BitcoinConf object
075     */
076    private BitcoinConf parseLines(List<String> lines) {
077        BitcoinConf conf = new BitcoinConf();
078        for (String line : lines) {
079            if (!line.startsWith("#")) {
080                String trimmed = line.split("#")[0].trim();
081                String[] kv = trimmed.split("=");
082                if (kv.length == 2) {
083                    conf.put(kv[0], kv[1]);
084                }
085
086            }
087        }
088        return conf;
089    }
090
091    private void setDefaults(BitcoinConf conf) {
092        conf.put("rpcconnect", "127.0.0.1");
093        conf.put("rpcport", "8332");
094        conf.put("rpcuser", "");
095        conf.put("rpcpassword", "");
096    }
097}
