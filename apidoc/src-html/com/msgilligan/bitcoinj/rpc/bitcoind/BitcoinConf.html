001package com.msgilligan.bitcoinj.rpc.bitcoind;
002
003import com.msgilligan.bitcoinj.rpc.RPCConfig;
004import org.bitcoinj.params.MainNetParams;
005
006import java.net.URI;
007import java.net.URISyntaxException;
008import java.util.HashMap;
009
010/**
011 * Bitcoin configuration from bitcoin.conf
012 * First cut is just a map, but future versions may be strongly typed
013 */
014public class BitcoinConf extends HashMap<String, String> {
015    public final String rpcconnect = "rpconnect";
016    public final String rpcport = "rpcport";
017    public final String RPCCONNECT_DEFAULT = "localhost";
018    public final String RPCPORT_DEFAULT = "8332";
019
020    /**
021     *  Create a BitcoinConf with a default RPCConfig configuration.
022     */
023    public BitcoinConf() {
024        super();
025        this.put(rpcconnect, RPCCONNECT_DEFAULT);
026        this.put(rpcport, RPCPORT_DEFAULT);
027    }
028
029    public RPCConfig getRPCConfig() {
030        URI uri = null;
031        try {
032            uri = new URI("http://" + get(rpcconnect) + ":" + get(rpcport));
033        } catch (URISyntaxException e) {
034            try {
035                uri = new URI("http://127.0.0.1:8332");
036            } catch (URISyntaxException e1) {
037                throw new RuntimeException("Error creating RPC URI", e1);
038            }
039        }
040        // TODO: Determine MainNet, TestNet, or RegTest from contents of .conf file
041        RPCConfig config = new RPCConfig(MainNetParams.get(), uri,
042                get("rpcuser"), get("rpcpassword"));
043        return config;
044    }
045}
