001package com.msgilligan.bitcoinj.rpc.bitcoind;
002
003import java.io.File;
004
005/**
006 * Utility class to find Application data directory.
007 *
008 */
009public class AppDataDirectory {
010    private static final String osName = System.getProperty("os.name").toLowerCase();
011
012    public static File forAppName(String appName) {
013        final String applicationDataDirectoryName;
014
015        // Locations are OS-dependent
016        if (isWindows()) {
017            // Windows
018            applicationDataDirectoryName = System.getenv("APPDATA") + File.separator + appName;
019        } else if (isMac()) {
020            // TODO: Isn't there a way to do this to account for non-english systems?
021            applicationDataDirectoryName = System.getProperty("user.home") + "/Library/Application Support/" + appName;
022        } else {
023            // Other (probably a Unix variant)
024            // Keep a clean home directory by prefixing with "."
025            applicationDataDirectoryName = System.getProperty("user.home") + "/." + appName.toLowerCase();
026        }
027
028        File applicationDataDirectory = new File(applicationDataDirectoryName);
029
030        return applicationDataDirectory;
031
032    }
033
034    static private boolean isWindows() {
035        return osName.contains("windows");
036    }
037
038    static private boolean isMac() {
039        return osName.startsWith("mac") || osName.startsWith("darwin");
040    }
041
042
043}
