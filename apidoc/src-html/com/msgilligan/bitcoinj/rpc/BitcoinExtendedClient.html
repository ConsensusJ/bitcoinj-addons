134    /**
135     * Creates a raw transaction, spending from a single address, whereby no new change address is created, and
136     * remaining amounts are returned to {@code fromAddress}.
137     *
138     * Note: the transaction inputs are not signed, and the transaction is not stored in the wallet or transmitted to
139     * the network.
140     *
141     * @param fromAddress The source to spend from
142     * @param outputs The destinations and amounts to transfer
143     * @return The hex-encoded raw transaction
144     */
145    public String createRawTransaction(Address fromAddress, Map<Address, Coin> outputs) throws JsonRpcStatusException, IOException {
146        // Get unspent outputs via RPC
147        List<UnspentOutput> unspentOutputs = listUnspent(0, defaultMaxConf, Collections.singletonList(fromAddress));
148
149        // Gather inputs
150        List<Outpoint> inputs = new ArrayList<>();
151        for (UnspentOutput input : unspentOutputs) {
152            inputs.add(new Outpoint(input.getTxid(), input.getVout()));
153        }
154
155        // Calculate change
156        long amountIn = 0;
157        long amountOut = 0;
158        for (UnspentOutput it : unspentOutputs) {
159            amountIn += it.getAmount().value;
160        }
161        for (Coin it : outputs.values()) {
162            amountOut += it.value;
163        }
164        Coin amountChange = Coin.valueOf(amountIn - amountOut - stdTxFee.value);
165        if (amountIn < (amountOut + stdTxFee.value)) {
166            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFee}"
167        }
168        // Copy the Map (which may be immutable) and add change output if needed.
169        Map<Address,Coin> outputsWithChange = new HashMap<>(outputs);
170        if (amountChange.value > 0) {
171            outputsWithChange.put(fromAddress, amountChange);
172        }
173
174        return createRawTransaction(inputs, outputsWithChange);
175    }

202    public Coin getBitcoinBalance(Address address) throws JsonRpcStatusException, IOException {
203        // NOTE: because null is currently removed from the argument lists passed via RPC, using it here for default
204        // values would result in the RPC call "listunspent" with arguments [["address"]], which is invalid, similar
205        // to a call with arguments [null, null, ["address"]], as expected arguments are either [], [int], [int, int]
206        // or [int, int, array]
207        return getBitcoinBalance(address, 1, defaultMaxConf);
208    }

221    /**
222     * Returns the Bitcoin balance of an address where spendable outputs have at least {@code minConf} and not more
223     * than {@code maxConf} confirmations.
224     *
225     * @param address The address (must be in wallet)
226     * @param minConf Minimum amount of confirmations
227     * @param maxConf Maximum amount of confirmations
228     * @return The balance
229     */
230    public Coin getBitcoinBalance(Address address, Integer minConf, Integer maxConf) throws JsonRpcStatusException, IOException {
231        long btcBalance = 0;
232        List<UnspentOutput> unspentOutputs = listUnspent(minConf, maxConf, Collections.singletonList(address));
233
234        for (UnspentOutput unspentOutput : unspentOutputs) {
235            btcBalance += unspentOutput.getAmount().value;
236        }
237
238        return Coin.valueOf(btcBalance);
239    }

255    /**
256     * Sends BTC from an address to the destinations, whereby no new change address is created, and any leftover is
257     * returned to the sending address.
258     *
259     * @param fromAddress The source to spent from
260     * @param outputs     The destinations and amounts to transfer
261     * @return The transaction hash
262     */
263    public Sha256Hash sendBitcoin(Address fromAddress, Map<Address, Coin> outputs) throws JsonRpcStatusException, IOException {
264        String unsignedTxHex = createRawTransaction(fromAddress, outputs);
265        SignedRawTransaction signingResult = signRawTransactionWithWallet(unsignedTxHex);
266
267        Boolean complete = signingResult.isComplete();
268        assert complete;
269
270        String signedTxHex = signingResult.getHex();
271        Sha256Hash txid = sendRawTransaction(signedTxHex);
272
273        return txid;
274    }

276    public Transaction createSignedTransaction(ECKey fromKey, List<TransactionOutput> outputs) throws JsonRpcStatusException, IOException {
277        Address fromAddress = Address.fromKey(getNetParams(), fromKey, Script.ScriptType.P2PKH);
278        Transaction tx = new Transaction(getNetParams());
279
280        List<TransactionOutput> unspentOutputs = listUnspentJ(fromAddress);
281
282        // Add outputs to the transaction
283        for (TransactionOutput it : outputs) {
284            tx.addOutput(it);
285        }
286
287        // Calculate change (units are satoshis)
288//        long amountIn     = (long) unspentOutputs.sum { TransactionOutput it -> it.value.longValue() }
289        long amountIn = 0;
290        for (TransactionOutput it : unspentOutputs) {
291            amountIn += it.getValue().value;
292        }
293//        long amountOut    = (long) outputs.sum { TransactionOutput it -> it.value.longValue() }
294        long amountOut = 0;
295        for (TransactionOutput it : outputs) {
296            amountOut += it.getValue().value;
297        }
298        long amountChange = amountIn - amountOut - stdTxFeeSatoshis;
299        if (amountChange < 0) {
300            // TODO: Throw Exception
301            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFeeSatoshis}"
302        }
303        if (amountChange > 0) {
304            // Add a change output
305            tx.addOutput(Coin.valueOf(amountChange), fromAddress);
306        }
307
308        // Add all UTXOs for fromAddress as inputs
309        for (TransactionOutput it : unspentOutputs) {
310            tx.addSignedInput(it, fromKey);
311        }
312        return tx;
313    }

315    public Transaction createSignedTransaction(ECKey fromKey, Address toAddress, Coin amount) throws JsonRpcStatusException, IOException {
316        List<TransactionOutput> outputs = Collections.singletonList(
317                new TransactionOutput(getNetParams(), null, amount, toAddress));
318        return createSignedTransaction(fromKey, outputs);

321    /**
322     * Build a list of bitcoinj <code>TransactionOutput</code>s using <code>listUnspent</code>
323     * and <code>getRawTransaction</code> RPCs
324     *
325     * @param fromAddress Address to get UTXOs for
326     * @return All unspent TransactionOutputs for fromAddress
327     */
328    public List<TransactionOutput> listUnspentJ(Address fromAddress) throws JsonRpcStatusException, IOException {
329        List<Address> addresses = Collections.singletonList(fromAddress);
330        List<UnspentOutput> unspentOutputsRPC = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects
331        List<TransactionOutput> unspentOutputsJ = new ArrayList<TransactionOutput>();
332        for (UnspentOutput it : unspentOutputsRPC) {
333            unspentOutputsJ.add(getRawTransaction(it.getTxid()).getOutput(it.getVout()));
334        }
335        return unspentOutputsJ;
336    }

338    public List<TransactionOutPoint> listUnspentOutPoints(Address fromAddress) throws JsonRpcStatusException, IOException {
339        List<Address> addresses = Collections.singletonList(fromAddress);
340        List<UnspentOutput> unspentOutputsRPC = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects
341        List<TransactionOutPoint> unspentOutPoints = new ArrayList<TransactionOutPoint>();
342        for (UnspentOutput it : unspentOutputsRPC) {
343            unspentOutPoints.add(new TransactionOutPoint(getNetParams(), it.getVout(), it.getTxid()));
344        }
345        return unspentOutPoints;
346    }
