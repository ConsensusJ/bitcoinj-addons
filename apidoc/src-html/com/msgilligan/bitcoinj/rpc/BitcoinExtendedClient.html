063    /**
064     * Creates a raw transaction, spending from a single address, whereby no new change address is created, and
065     * remaining amounts are returned to {@code fromAddress}.
066     *
067     * Note: the transaction inputs are not signed, and the transaction is not stored in the wallet or transmitted to
068     * the network.
069     *
070     * @param fromAddress The source to spend from
071     * @param outputs The destinations and amounts to transfer
072     * @return The hex-encoded raw transaction
073     */
074    public String createRawTransaction(Address fromAddress, Map<Address, Coin> outputs) throws JsonRpcStatusException, IOException {
075        // Get unspent outputs via RPC
076        List<UnspentOutput> unspentOutputs = listUnspent(0, defaultMaxConf, Collections.singletonList(fromAddress));
077
078        // Gather inputs
079        List<Outpoint> inputs = new ArrayList<>();
080        for (UnspentOutput input : unspentOutputs) {
081            inputs.add(new Outpoint(input.getTxid(), input.getVout()));
082        }
083
084        // Calculate change
085        long amountIn = 0;
086        long amountOut = 0;
087        for (UnspentOutput it : unspentOutputs) {
088            amountIn += it.getAmount().value;
089        }
090        for (Coin it : outputs.values()) {
091            amountOut += it.value;
092        }
093        Coin amountChange = Coin.valueOf(amountIn - amountOut - stdTxFee.value);
094        if (amountIn < (amountOut + stdTxFee.value)) {
095            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFee}"
096        }
097        // Copy the Map (which may be immutable) and add change output if needed.
098        Map<Address,Coin> outputsWithChange = new HashMap<>(outputs);
099        if (amountChange.value > 0) {
100            outputsWithChange.put(fromAddress, amountChange);
101        }
102
103        return createRawTransaction(inputs, outputsWithChange);
104    }

106    /**
107     * Creates a raw transaction, sending {@code amount} from a single address to a destination, whereby no new change
108     * address is created, and remaining amounts are returned to {@code fromAddress}.
109     *
110     * Note: the transaction inputs are not signed, and the transaction is not stored in the wallet or transmitted to
111     * the network.
112     *
113     * @param fromAddress The source to spent from
114     * @param toAddress The destination
115     * @param amount The amount
116     * @return The hex-encoded raw transaction
117     */
118    public String createRawTransaction(Address fromAddress, Address toAddress, Coin amount) throws JsonRpcStatusException, IOException {
119        Map<Address, Coin> outputs = Collections.singletonMap(toAddress, amount);
120        return createRawTransaction(fromAddress, outputs);
121    }

126     * Returns the Bitcoin balance of an address.
127     *
128     * @param address The address
129     * @return The balance
130     */
131    public Coin getBitcoinBalance(Address address) throws JsonRpcStatusException, IOException {
132        // NOTE: because null is currently removed from the argument lists passed via RPC, using it here for default
133        // values would result in the RPC call "listunspent" with arguments [["address"]], which is invalid, similar
134        // to a call with arguments [null, null, ["address"]], as expected arguments are either [], [int], [int, int]
135        // or [int, int, array]
136        return getBitcoinBalance(address, 1, defaultMaxConf);
137    }

144     * @return The balance

147        return getBitcoinBalance(address, minConf, defaultMaxConf);

150    /**
151     * Returns the Bitcoin balance of an address where spendable outputs have at least {@code minConf} and not more
152     * than {@code maxConf} confirmations.
153     *
154     * @param address The address (must be in wallet)
155     * @param minConf Minimum amount of confirmations
156     * @param maxConf Maximum amount of confirmations
157     * @return The balance
158     */
159    public Coin getBitcoinBalance(Address address, Integer minConf, Integer maxConf) throws JsonRpcStatusException, IOException {
160        long btcBalance = 0;
161        List<UnspentOutput> unspentOutputs = listUnspent(minConf, maxConf, Collections.singletonList(address));
162
163        for (UnspentOutput unspentOutput : unspentOutputs) {
164            btcBalance += unspentOutput.getAmount().value;
165        }
166
167        return Coin.valueOf(btcBalance);
168    }

170    /**
171     * Sends BTC from an address to a destination, whereby no new change address is created, and any leftover is
172     * returned to the sending address.
173     *
174     * @param fromAddress The source to spent from
175     * @param toAddress   The destination address
176     * @param amount      The amount to transfer
177     * @return The transaction hash
178     */
179    public Sha256Hash sendBitcoin(Address fromAddress, Address toAddress, Coin amount) throws JsonRpcStatusException, IOException {
180        Map<Address, Coin> outputs = Collections.singletonMap(toAddress, amount);
181        return sendBitcoin(fromAddress, outputs);
182    }

184    /**
185     * Sends BTC from an address to the destinations, whereby no new change address is created, and any leftover is
186     * returned to the sending address.
187     *
188     * @param fromAddress The source to spent from
189     * @param outputs     The destinations and amounts to transfer
190     * @return The transaction hash
191     */
192    public Sha256Hash sendBitcoin(Address fromAddress, Map<Address, Coin> outputs) throws JsonRpcStatusException, IOException {
193        String unsignedTxHex = createRawTransaction(fromAddress, outputs);
194        SignedRawTransaction signingResult = signRawTransaction(unsignedTxHex);
195
196        Boolean complete = signingResult.isComplete();
197        assert complete;
198
199        String signedTxHex = signingResult.getHex();
200        Sha256Hash txid = sendRawTransaction(signedTxHex);
201
202        return txid;
203    }

205    public Transaction createSignedTransaction(ECKey fromKey, List<TransactionOutput> outputs) throws JsonRpcStatusException, IOException {
206        Address fromAddress = fromKey.toAddress(getNetParams());
207        Transaction tx = new Transaction(getNetParams());
208
209        List<TransactionOutput> unspentOutputs = listUnspentJ(fromAddress);
210
211        // Add outputs to the transaction
212        for (TransactionOutput it : outputs) {
213            tx.addOutput(it);
214        }
215
216        // Calculate change (units are satoshis)
217//        long amountIn     = (long) unspentOutputs.sum { TransactionOutput it -> it.value.longValue() }
218        long amountIn = 0;
219        for (TransactionOutput it : unspentOutputs) {
220            amountIn += it.getValue().value;
221        }
222//        long amountOut    = (long) outputs.sum { TransactionOutput it -> it.value.longValue() }
223        long amountOut = 0;
224        for (TransactionOutput it : outputs) {
225            amountOut += it.getValue().value;
226        }
227        long amountChange = amountIn - amountOut - stdTxFeeSatoshis;
228        if (amountChange < 0) {
229            // TODO: Throw Exception
230            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFeeSatoshis}"
231        }
232        if (amountChange > 0) {
233            // Add a change output
234            tx.addOutput(Coin.valueOf(amountChange), fromAddress);
235        }
236
237        // Add all UTXOs for fromAddress as inputs
238        for (TransactionOutput it : unspentOutputs) {
239            tx.addSignedInput(it, fromKey);
240        }
241        return tx;
242    }

244    public Transaction createSignedTransaction(ECKey fromKey, Address toAddress, Coin amount) throws JsonRpcStatusException, IOException {
245        List<TransactionOutput> outputs = Collections.singletonList(
246                new TransactionOutput(getNetParams(), null, amount, toAddress));
247        return createSignedTransaction(fromKey, outputs);
248    }

250    /**
251     * Build a list of bitcoinj <code>TransactionOutput</code>s using <code>listUnspent</code>
252     * and <code>getRawTransaction</code> RPCs
253     *
254     * @param fromAddress Address to get UTXOs for
255     * @return All unspent TransactionOutputs for fromAddress
256     */
257    public List<TransactionOutput> listUnspentJ(Address fromAddress) throws JsonRpcStatusException, IOException {
258        List<Address> addresses = Collections.singletonList(fromAddress);
259        List<UnspentOutput> unspentOutputsRPC = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects
260        List<TransactionOutput> unspentOutputsJ = new ArrayList<TransactionOutput>();
261        for (UnspentOutput it : unspentOutputsRPC) {
262            unspentOutputsJ.add(getRawTransaction(it.getTxid()).getOutput(it.getVout()));
263        }
264        return unspentOutputsJ;
265    }

267    public List<TransactionOutPoint> listUnspentOutPoints(Address fromAddress) throws JsonRpcStatusException, IOException {
268        List<Address> addresses = Collections.singletonList(fromAddress);
269        List<UnspentOutput> unspentOutputsRPC = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects
270        List<TransactionOutPoint> unspentOutPoints = new ArrayList<TransactionOutPoint>();
271        for (UnspentOutput it : unspentOutputsRPC) {
272            unspentOutPoints.add(new TransactionOutPoint(getNetParams(), it.getVout(), it.getTxid()));
273        }
274        return unspentOutPoints;
275    }
