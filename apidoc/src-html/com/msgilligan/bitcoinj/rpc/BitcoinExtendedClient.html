119    /**
120     * Creates a raw transaction, spending from a single address, whereby no new change address is created, and
121     * remaining amounts are returned to {@code fromAddress}.
122     * <p>
123     * Note: the transaction inputs are not signed, and the transaction is not stored in the wallet or transmitted to
124     * the network.
125     *
126     * @param fromAddress The source to spend from
127     * @param outputs The destinations and amounts to transfer
128     * @return The hex-encoded raw transaction
129     */
130    public String createRawTransaction(Address fromAddress, Map<Address, Coin> outputs) throws JsonRpcStatusException, IOException {
131        // Copy the Map (which may be immutable) and add prepare change output if needed.
132        Map<Address, Coin> outputsWithChange = new HashMap<>(outputs);
133        // Get unspent outputs via RPC
134        List<UnspentOutput> unspentOutputs = listUnspent(0, defaultMaxConf, Collections.singletonList(fromAddress));
135
136        // Gather inputs as OutPoints
137        List<Outpoint> inputs = unspentOutputs.stream()
138                .map(this::unspentToOutpoint)
139                .collect(Collectors.toList());
140
141        // Calculate change
142        final long amountIn = unspentOutputs
143                .stream()
144                .map(UnspentOutput::getAmount)
145                .mapToLong(Coin::getValue)
146                .sum();
147        final long amountOut = outputs.values()
148                .stream()
149                .mapToLong(Coin::getValue)
150                .sum();
151
152        // Change is the difference less the standard transaction fee
153        final long amountChange = amountIn - amountOut - stdTxFeeSatoshis;
154        if (amountChange < 0) {
155            // TODO: Throw Exception
156            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFee}"
157        }
158        if (amountChange > 0) {
159            // Add a change output that returns change to sending address
160            outputsWithChange.put(fromAddress, Coin.valueOf(amountChange));
161        }
162
163        // Call the server to create the transaction
164        return createRawTransaction(inputs, outputsWithChange);
165    }

184     * Returns the Bitcoin balance of an address (in the server-side wallet.)
185     *
186     * @param address The address
187     * @return The balance
188     */
189    public Coin getBitcoinBalance(Address address) throws JsonRpcStatusException, IOException {
190        // NOTE: because null is currently removed from the argument lists passed via RPC, using it here for default
191        // values would result in the RPC call "listunspent" with arguments [["address"]], which is invalid, similar
192        // to a call with arguments [null, null, ["address"]], as expected arguments are either [], [int], [int, int]
193        // or [int, int, array]
194        return getBitcoinBalance(address, 1, defaultMaxConf);
195    }

209    /**
210     * Returns the Bitcoin balance of an address (in the server-side wallet) where spendable outputs have at least
211     * {@code minConf} and not more than {@code maxConf} confirmations.
212     *
213     * @param address The address (must be in wallet)
214     * @param minConf Minimum amount of confirmations
215     * @param maxConf Maximum amount of confirmations
216     * @return The balance
217     */
218    public Coin getBitcoinBalance(Address address, Integer minConf, Integer maxConf) throws JsonRpcStatusException, IOException {
219        long btcBalance = 0;
220        List<UnspentOutput> unspentOutputs = listUnspent(minConf, maxConf, Collections.singletonList(address));
221
222        for (UnspentOutput unspentOutput : unspentOutputs) {
223            btcBalance += unspentOutput.getAmount().value;
224        }
225
226        return Coin.valueOf(btcBalance);
227    }

243    /**
244     * Sends BTC from an address to the destinations, whereby no new change address is created, and any leftover is
245     * returned to the sending address.
246     *
247     * @param fromAddress The source to spent from
248     * @param outputs     The destinations and amounts to transfer
249     * @return The transaction hash
250     */
251    public Sha256Hash sendBitcoin(Address fromAddress, Map<Address, Coin> outputs) throws JsonRpcStatusException, IOException {
252        String unsignedTxHex = createRawTransaction(fromAddress, outputs);
253        SignedRawTransaction signingResult = signRawTransactionWithWallet(unsignedTxHex);
254
255        Boolean complete = signingResult.isComplete();
256        assert complete;
257
258        String signedTxHex = signingResult.getHex();
259        Sha256Hash txid = sendRawTransaction(signedTxHex);
260
261        return txid;
262    }

264    /**
265     * Create a signed transaction locally (i.e. with a client-side key.) Finds UTXOs this
266     * key can spend (assuming they are Script.ScriptType.P2PKH UTXOs)
267     *
268     * @param fromKey Signing key
269     * @param outputs Outputs to sign
270     * @return A bitcoinj Transaction objects that is properly signed
271     * @throws JsonRpcStatusException A JSON-RPC error was returned
272     * @throws IOException            An I/O error occured
273     */
274    public Transaction createSignedTransaction(ECKey fromKey, List<TransactionOutput> outputs) throws JsonRpcStatusException, IOException {
275        Address fromAddress = Address.fromKey(getNetParams(), fromKey, Script.ScriptType.P2PKH);
276
277        Transaction tx = new Transaction(getNetParams());   // Create a new transaction
278        outputs.forEach(tx::addOutput);                     // Add all requested outputs to it
279
280        // Fetch all UTXOs for the sending Address
281        List<TransactionOutput> unspentOutputs = listUnspentJ(fromAddress);
282
283        // Calculate change (units are satoshis)
284        // First sum all available UTXOs
285        final long amountIn = unspentOutputs.stream()
286                .map(TransactionOutput::getValue)
287                .mapToLong(Coin::getValue)
288                .sum();
289        // Then sum the requested outputs for this transaction
290        final long amountOut = outputs.stream()
291                .map(TransactionOutput::getValue)
292                .mapToLong(Coin::getValue)
293                .sum();
294        // Change is the difference less the standard transaction fee
295        final long amountChange = amountIn - amountOut - stdTxFeeSatoshis;
296        if (amountChange < 0) {
297            // TODO: Throw Exception
298            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFeeSatoshis}"
299        }
300        if (amountChange > 0) {
301            // Add a change output that returns change to sending address
302            tx.addOutput(Coin.valueOf(amountChange), fromAddress);
303        }
304
305        // Add *all* UTXOs for fromAddress as inputs (this perhaps unnecessarily consolidates coins, with
306        // a higher tx fee and some loss of privacy) and sign them
307        unspentOutputs.forEach(unspent -> tx.addSignedInput(unspent, fromKey));
308        return tx;
309    }

330     * and {@link BitcoinClient#getRawTransaction} RPCs.
331     *
332     * @param fromAddress Address to get UTXOs for
333     * @return All unspent TransactionOutputs for fromAddress

335    public List<TransactionOutput> listUnspentJ(Address fromAddress) throws JsonRpcStatusException, IOException {
336        List<Address> addresses = Collections.singletonList(fromAddress);
337        List<UnspentOutput> unspentOutputs = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects

357    /**
358     * Convert an {@link UnspentOutput} JSONRPC-POJO to a *bitcoinj* {@link TransactionOutput} (that's out-PUT).
359     * Calls {@link BitcoinClient#getRawTransaction(Sha256Hash)}
360     * for every item. Throws {@link RuntimeException} if any of those RPC calls fails.
361     *
362     * @param unspentOutput The input POJO
363     * @return The *bitcoinj* object  (that's out-PUT)
364     */
365    private TransactionOutput unspentToTransactionOutput(UnspentOutput unspentOutput) {
366        try {
367            return getRawTransaction(unspentOutput.getTxid())
368                    .getOutput(unspentOutput.getVout());
369        } catch (IOException e) {
370            throw new RuntimeException(e);
371        }
372    }
373
374    /**
375     * Convert an {@link UnspentOutput} JSONRPC-POJO to a *bitcoinj* {@link TransactionOutPoint} (that's out-POINT).
376     *
377     * @param unspentOutput The input POJO
378     * @return The *bitcoinj* object  (that's out-POINT)
379     */
380    private TransactionOutPoint unspentToTransactionOutpoint(UnspentOutput unspentOutput) {
381        return new TransactionOutPoint(getNetParams(), unspentOutput.getVout(), unspentOutput.getTxid());
382    }
383
384    /**
385     * Convert an {@link UnspentOutput} JSONRPC-POJO to a JSONRPC-POJO {@link Outpoint} .
386     *
387     * @param unspentOutput The input UnspentOutput POJO
388     * @return the Outpoint POJO
389     */
390    private Outpoint unspentToOutpoint(UnspentOutput unspentOutput) {
391        return new Outpoint(unspentOutput.getTxid(), unspentOutput.getVout());
