051    /**
052     * Creates a raw transaction, spending from a single address, whereby no new change address is created, and
053     * remaining amounts are returned to {@code fromAddress}.
054     *
055     * Note: the transaction inputs are not signed, and the transaction is not stored in the wallet or transmitted to
056     * the network.
057     *
058     * @param fromAddress The source to spend from
059     * @param outputs The destinations and amounts to transfer
060     * @return The hex-encoded raw transaction
061     */
062    public String createRawTransaction(Address fromAddress, Map<Address, Coin> outputs) throws JsonRPCStatusException, IOException {
063        // Get unspent outputs via RPC
064        List<UnspentOutput> unspentOutputs = listUnspent(0, defaultMaxConf, Collections.singletonList(fromAddress));
065
066        // Gather inputs
067        List<Outpoint> inputs = new ArrayList<>();
068        for (UnspentOutput input : unspentOutputs) {
069            inputs.add(new Outpoint(input.getTxid(), input.getVout()));
070        }
071
072        // Calculate change
073        long amountIn = 0;
074        long amountOut = 0;
075        for (UnspentOutput it : unspentOutputs) {
076            amountIn += it.getAmount().value;
077        }
078        for (Coin it : outputs.values()) {
079            amountOut += it.value;
080        }
081        Coin amountChange = Coin.valueOf(amountIn - amountOut - stdTxFee.value);
082        if (amountIn < (amountOut + stdTxFee.value)) {
083            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFee}"
084        }
085        // Copy the Map (which may be immutable) and add change output if needed.
086        Map<Address,Coin> outputsWithChange = new HashMap<>(outputs);
087        if (amountChange.value > 0) {
088            outputsWithChange.put(fromAddress, amountChange);
089        }
090
091        return createRawTransaction(inputs, outputsWithChange);
092    }

094    /**
095     * Creates a raw transaction, sending {@code amount} from a single address to a destination, whereby no new change
096     * address is created, and remaining amounts are returned to {@code fromAddress}.
097     *
098     * Note: the transaction inputs are not signed, and the transaction is not stored in the wallet or transmitted to
099     * the network.
100     *
101     * @param fromAddress The source to spent from
102     * @param toAddress The destination
103     * @param amount The amount
104     * @return The hex-encoded raw transaction
105     */
106    public String createRawTransaction(Address fromAddress, Address toAddress, Coin amount) throws JsonRPCStatusException, IOException {
107        Map<Address, Coin> outputs = Collections.singletonMap(toAddress, amount);
108        return createRawTransaction(fromAddress, outputs);
109    }

113    /**
114     * Returns the Bitcoin balance of an address.
115     *
116     * @param address The address
117     * @return The balance
118     */
119    public Coin getBitcoinBalance(Address address) throws JsonRPCStatusException, IOException {
120        // NOTE: because null is currently removed from the argument lists passed via RPC, using it here for default
121        // values would result in the RPC call "listunspent" with arguments [["address"]], which is invalid, similar
122        // to a call with arguments [null, null, ["address"]], as expected arguments are either [], [int], [int, int]
123        // or [int, int, array]
124        return getBitcoinBalance(address, 1, defaultMaxConf);
125    }

127    /**
128     * Returns the Bitcoin balance of an address where spendable outputs have at least {@code minConf} confirmations.
129     *
130     * @param address The address
131     * @param minConf Minimum amount of confirmations
132     * @return The balance
133     */
134    public Coin getBitcoinBalance(Address address, Integer minConf) throws JsonRPCStatusException, IOException {
135        return getBitcoinBalance(address, minConf, defaultMaxConf);
136    }

138    /**
139     * Returns the Bitcoin balance of an address where spendable outputs have at least {@code minConf} and not more
140     * than {@code maxConf} confirmations.
141     *
142     * @param address The address (must be in wallet)
143     * @param minConf Minimum amount of confirmations
144     * @param maxConf Maximum amount of confirmations
145     * @return The balance
146     */
147    public Coin getBitcoinBalance(Address address, Integer minConf, Integer maxConf) throws JsonRPCStatusException, IOException {
148        long btcBalance = 0;
149        List<UnspentOutput> unspentOutputs = listUnspent(minConf, maxConf, Collections.singletonList(address));
150
151        for (UnspentOutput unspentOutput : unspentOutputs) {
152            btcBalance += unspentOutput.getAmount().value;
153        }
154
155        return Coin.valueOf(btcBalance);
156    }

158    /**
159     * Sends BTC from an address to a destination, whereby no new change address is created, and any leftover is
160     * returned to the sending address.
161     *
162     * @param fromAddress The source to spent from
163     * @param toAddress   The destination address
164     * @param amount      The amount to transfer
165     * @return The transaction hash
166     */
167    public Sha256Hash sendBitcoin(Address fromAddress, Address toAddress, Coin amount) throws JsonRPCStatusException, IOException {
168        Map<Address, Coin> outputs = Collections.singletonMap(toAddress, amount);
169        return sendBitcoin(fromAddress, outputs);
170    }

172    /**
173     * Sends BTC from an address to the destinations, whereby no new change address is created, and any leftover is
174     * returned to the sending address.
175     *
176     * @param fromAddress The source to spent from
177     * @param outputs     The destinations and amounts to transfer
178     * @return The transaction hash
179     */
180    public Sha256Hash sendBitcoin(Address fromAddress, Map<Address, Coin> outputs) throws JsonRPCStatusException, IOException {
181        String unsignedTxHex = createRawTransaction(fromAddress, outputs);
182        SignedRawTransaction signingResult = signRawTransaction(unsignedTxHex);
183
184        Boolean complete = signingResult.isComplete();
185        assert complete;
186
187        String signedTxHex = signingResult.getHex();
188        Sha256Hash txid = sendRawTransaction(signedTxHex);
189
190        return txid;
191    }

193    public Transaction createSignedTransaction(ECKey fromKey, List<TransactionOutput> outputs) throws JsonRPCStatusException, IOException {
194        Address fromAddress = fromKey.toAddress(getNetParams());
195        Transaction tx = new Transaction(getNetParams());
196
197        List<TransactionOutput> unspentOutputs = listUnspentJ(fromAddress);
198
199        // Add outputs to the transaction
200        for (TransactionOutput it : outputs) {
201            tx.addOutput(it);
202        }
203
204        // Calculate change (units are satoshis)
205//        long amountIn     = (long) unspentOutputs.sum { TransactionOutput it -> it.value.longValue() }
206        long amountIn = 0;
207        for (TransactionOutput it : unspentOutputs) {
208            amountIn += it.getValue().value;
209        }
210//        long amountOut    = (long) outputs.sum { TransactionOutput it -> it.value.longValue() }
211        long amountOut = 0;
212        for (TransactionOutput it : outputs) {
213            amountOut += it.getValue().value;
214        }
215        long amountChange = amountIn - amountOut - stdTxFeeSatoshis;
216        if (amountChange < 0) {
217            // TODO: Throw Exception
218            System.out.println("Insufficient funds"); // + ": ${amountIn} < ${amountOut + stdTxFeeSatoshis}"
219        }
220        if (amountChange > 0) {
221            // Add a change output
222            tx.addOutput(Coin.valueOf(amountChange), fromAddress);
223        }
224
225        // Add all UTXOs for fromAddress as inputs
226        for (TransactionOutput it : unspentOutputs) {
227            tx.addSignedInput(it, fromKey);
228        }
229        return tx;
230    }

232    public Transaction createSignedTransaction(ECKey fromKey, Address toAddress, Coin amount) throws JsonRPCStatusException, IOException {
233        List<TransactionOutput> outputs = Collections.singletonList(
234                new TransactionOutput(getNetParams(), null, amount, toAddress));
235        return createSignedTransaction(fromKey, outputs);
236    }

238    /**
239     * Build a list of bitcoinj <code>TransactionOutput</code>s using <code>listUnspent</code>
240     * and <code>getRawTransaction</code> RPCs
241     *
242     * @param fromAddress Address to get UTXOs for
243     * @return All unspent TransactionOutputs for fromAddress
244     */
245    public List<TransactionOutput> listUnspentJ(Address fromAddress) throws JsonRPCStatusException, IOException {
246        List<Address> addresses = Collections.singletonList(fromAddress);
247        List<UnspentOutput> unspentOutputsRPC = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects
248        List<TransactionOutput> unspentOutputsJ = new ArrayList<TransactionOutput>();
249        for (UnspentOutput it : unspentOutputsRPC) {
250            unspentOutputsJ.add(getRawTransaction(it.getTxid()).getOutput(it.getVout()));
251        }
252        return unspentOutputsJ;
253    }
254
255    public List<TransactionOutPoint> listUnspentOutPoints(Address fromAddress) throws JsonRPCStatusException, IOException {
256        List<Address> addresses = Collections.singletonList(fromAddress);
257        List<UnspentOutput> unspentOutputsRPC = listUnspent(0, defaultMaxConf, addresses); // RPC UnspentOutput objects
258        List<TransactionOutPoint> unspentOutPoints = new ArrayList<TransactionOutPoint>();
259        for (UnspentOutput it : unspentOutputsRPC) {
260            unspentOutPoints.add(new TransactionOutPoint(getNetParams(), it.getVout(), it.getTxid()));
261        }
262        return unspentOutPoints;
263    }
