001package com.msgilligan.bitcoinj.rpc;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnore;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.NetworkParameters;
007import org.bitcoinj.params.RegTestParams;
008
009import java.net.URI;
010import java.net.URISyntaxException;
011
012/**
013 * Configuration class for JSON-RPC client
014 *
015 * Contains complete URL, username, and password.
016 *
017 */
018public class RPCConfig {
019
020    private final NetworkParameters netParams;
021    private final URI uri;
022    private final String   username;
023    private final String   password;
024
025    public RPCConfig(NetworkParameters netParams, URI uri, String username, String password) {
026        this.netParams = netParams;
027        this.uri = uri;
028        this.username = username;
029        this.password = password;
030    }
031
032    @JsonCreator
033    public RPCConfig(@JsonProperty("netid")     String netIdString,
034                     @JsonProperty("uri")       String uri,
035                     @JsonProperty("username")  String username,
036                     @JsonProperty("password")  String password) throws URISyntaxException {
037        this(NetworkParameters.fromID(netIdString),
038                new URI(uri),
039                username,
040                password);
041    }
042
043    @Deprecated
044    public RPCConfig(URI uri, String username, String password) {
045        this(RegTestParams.get(), uri, username, password);
046    }
047
048    @JsonIgnore
049    public NetworkParameters getNetParams() {
050        return netParams;
051    }
052
053    @JsonProperty("netid")
054    public String getNetIdString() {
055        return netParams.getId();
056    }
057
058    public URI getURI() {
059        return uri;
060    }
061
062    public String getUsername() {
063        return username;
064    }
065
066    public String getPassword() {
067        return password;
068    }
069}
