043/**
044 * = JSON-RPC Client for *Bitcoin Core*
045 *
046 * A strongly-typed wrapper for a {@index Bitcoin} RPC client using the
047 * https://bitcoin.org/en/developer-reference#bitcoin-core-apis[Bitcoin Core JSON-RPC API].
048 * https://bitcoinj.github.io[bitcoinj] types are used where appropriate.
049 * For example, requesting a block hash will return a {@link org.bitcoinj.core.Sha256Hash}:
050 *
051 * [source,java]
052 * --
053 * Sha256Hash hash = client.getBlockHash(342650);
054 * --
055 *
056 * Requesting a Bitcoin balance will return the amount as a {@link org.bitcoinj.core.Coin}:
057 * [source,java]
058 * --
059 * Coin balance = client.getBalance();
060 * --
061 *
062 * This version is written to be compatible with Bitcoin Core 0.10.4 and later. If used with
063 * Omni Core (an enhanced Bitcoin Core with Omni Protocol support) Omni Core 0.0.11.1
064 * or later is required.
065 *
066 * NOTE: This is still a work-in-progress and the API will change.
067 *
068 */

136    /**
137     * Wait until the server is available.
138     *
139     * Keep trying, ignoring (and logging) a known list of exception conditions that may occur while waiting for
140     * a `bitcoind` server to start up. This is similar to the behavior enabled by the `-rpcwait`
141     * option to the `bitcoin-cli` command-line tool.
142     *
143     * @param timeout Timeout in seconds
144     * @return true if ready, false if timeout or interrupted
145     */
146    public Boolean waitForServer(int timeout) throws JsonRpcException {
147
148        log.debug("Waiting for server RPC ready...");
149
150        String status;          // Status message for logging
151        String statusLast = null;
152        int seconds = 0;
153        while (seconds < timeout) {
154            try {
155                Integer block = this.getBlockCount();
156                if (block != null) {
157                    log.debug("RPC Ready.");
158                    return true;
159                }
160                status = "getBlock returned null";
161            } catch (SocketException se) {
162                // These are expected exceptions while waiting for a server
163                if (se.getMessage().equals("Unexpected end of file from server") ||
164                        se.getMessage().equals("Connection reset") ||
165                        se.getMessage().contains("Connection refused") ||
166                        se.getMessage().equals("recvfrom failed: ECONNRESET (Connection reset by peer)")) {
167                    status = se.getMessage();
168                } else {
169                    throw new JsonRpcException("Unexpected exception in waitForServer", se) ;
170                }
171
172            } catch (EOFException ignored) {
173                /* Android exception, ignore */
174                // Expected exceptions on Android, RoboVM
175                status = ignored.getMessage();
176            } catch (JsonRpcStatusException e) {
177                // If server is in "warm-up" mode, e.g. validating/parsing the blockchain...
178                if (e.jsonRpcCode == -28) {
179                    // ...then grab text message for status logging
180                    status = e.getMessage();
181                } else {
182                    throw e;
183                }
184            } catch (IOException e) {
185                // Ignore all IOExceptions
186                status = e.getMessage();
187            }
188            try {
189                // Log status messages only once, if new or updated
190                if (!status.equals(statusLast)) {
191                    log.info("RPC Status: " + status);
192                    statusLast = status;
193                }
194                Thread.sleep(RETRY_SECONDS * SECOND_IN_MSEC);
195                seconds += RETRY_SECONDS;
196            } catch (InterruptedException e) {
197                log.error(e.toString());
198                Thread.currentThread().interrupt();
199                return false;
200            }
201        }
202
203        log.error("waitForServer() timed out after {} seconds.", timeout);
204        return false;
205    }

207    /**
208     * Wait for RPC server to reach specified block height.
209     *
210     * @param blockHeight Block height to wait for
211     * @param timeout     Timeout in seconds
212     * @throws JsonRpcStatusException JSON RPC status exception
213     * @throws IOException network error
214     * @return True if blockHeight reached, false if timeout or interrupted
215     */
216    public Boolean waitForBlock(int blockHeight, int timeout) throws JsonRpcStatusException, IOException {
217
218        log.info("Waiting for server to reach block " + blockHeight);
219
220        int seconds = 0;
221        while (seconds < timeout) {
222            Integer block = this.getBlockCount();
223            if (block >= blockHeight) {
224                log.info("Server is at block " + block + " returning 'true'.");
225                return true;
226            } else {
227                try {
228                    if (seconds % MESSAGE_SECONDS == 0) {
229                        log.debug("Server at block " + block);
230                    }
231                    Thread.sleep(RETRY_SECONDS * SECOND_IN_MSEC);
232                    seconds += RETRY_SECONDS;
233                } catch (InterruptedException e) {
234                    log.error(e.toString());
235                    Thread.currentThread().interrupt();
236                    return false;
237                }
238            }
239        }
240
241        log.error("Timeout waiting for block " + blockHeight);
242        return false;
243    }

837        for (List<List<Object>> rawGrouping : raw) {

874    /**
875     * Returns a list of available commands.
876     *
877     * Commands which are unavailable will not be listed, such as wallet RPCs, if wallet support is disabled.
878     *
879     * @return The list of commands
880     * @throws JsonRpcStatusException JSON RPC status exception
881     * @throws IOException network error
882     */
883    public List<String> getCommands() throws JsonRpcStatusException, IOException {
884        List<String> commands = new ArrayList<String>();
885        for (String entry : help().split("\n")) {
886            if (!entry.isEmpty() && !entry.matches("== (.+) ==")) {
887                String command = entry.split(" ")[0];
888                commands.add(command);
889            }
890        }
891        return commands;
892    }
