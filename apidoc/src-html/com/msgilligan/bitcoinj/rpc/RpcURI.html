001package com.msgilligan.bitcoinj.rpc;
002
003import java.net.URI;
004
005/**
006 * Utility class with default connection information for Bitcoin JSON-RPC
007 */
008public class RpcURI {
009    public static final String rpcproto = "http";
010    public static final String rpcssl = "https";
011    public static final String rpchost = "127.0.0.1";
012    public static final String rpcfile = "/";
013
014    public static final int RPCPORT_MAINNET = 8332;
015    public static final int RPCPORT_TESTNET = 18332;
016    public static final int RPCPORT_REGTEST = 18332;
017
018    public static URI getDefaultMainNetURI() {
019        return createURI(rpcproto, rpchost, RPCPORT_MAINNET, rpcfile);
020    }
021
022    public static URI getDefaultTestNetURI() {
023        return createURI(rpcproto, rpchost, RPCPORT_TESTNET, rpcfile);
024    }
025
026    public static URI getDefaultRegTestURI() {
027        return createURI(rpcproto, rpchost, RPCPORT_REGTEST, rpcfile);
028    }
029
030    private static URI createURI(String proto, String host, int port, String file) {
031        return URI.create(rpcproto + "://" + rpchost + ":" + port + rpcfile);
032    }
033
034}
