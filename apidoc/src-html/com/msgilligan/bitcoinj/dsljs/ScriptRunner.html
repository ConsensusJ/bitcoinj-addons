001package com.msgilligan.bitcoinj.dsljs;
002
003import com.msgilligan.bitcoinj.rpc.BitcoinExtendedClient;
004import com.msgilligan.bitcoinj.rpc.RPCURI;
005import com.msgilligan.bitcoinj.rpc.test.TestServers;
006import com.msgilligan.bitcoinj.test.RegTestEnvironment;
007import com.msgilligan.bitcoinj.test.RegTestFundingSource;
008import jdk.nashorn.api.scripting.NashornScriptEngineFactory;
009import org.bitcoinj.core.Coin;
010import org.bitcoinj.params.RegTestParams;
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014import javax.script.ScriptEngine;
015import javax.script.ScriptException;
016import java.io.InputStreamReader;
017import java.nio.charset.StandardCharsets;
018import java.util.function.BiFunction;
019import java.util.function.Function;
020import java.util.function.Supplier;
021
022
023/**
024 *
025 */
026public class ScriptRunner {
027    private static final Logger log = LoggerFactory.getLogger(ScriptRunner.class);
028    private ScriptEngine engine;
029
030    public ScriptRunner() throws ScriptException {
031        NashornScriptEngineFactory factory = new NashornScriptEngineFactory();
032        engine = factory.getScriptEngine("-scripting");
033        BitcoinExtendedClient client = new BitcoinExtendedClient(RegTestParams.get(),
034                RPCURI.getDefaultRegTestURI(),
035                TestServers.getInstance().getRpcTestUser(),
036                TestServers.getInstance().getRpcTestPassword());
037        RegTestEnvironment env = new RegTestEnvironment(client);
038        RegTestFundingSource funder = new RegTestFundingSource(client);
039        engine.put("client", client);
040        engine.put("env", env);
041        engine.put("funder", funder);
042        engine.put("satoshi", (Function<Number, Coin>) (satoshis) -> Coin.valueOf(satoshis.longValue()));
043        engine.put("btc", (Function<Number, Coin>) (btc) -> Coin.valueOf(btc.longValue() * Coin.COIN.longValue()));
044        engine.put("coin", (BiFunction<Number, Number, Coin>) (btc, cents) -> Coin.valueOf(btc.intValue(), cents.intValue()));
045        engine.put("getBlockCount", (Supplier<Integer>)() -> {
046            try { return client.getBlockCount(); }
047            catch (Exception e) { log.error("Exception: {}", e); throw new RuntimeException(e); }
048        });
049    }
050
051    public Object evalResource(String resourcePath) throws ScriptException {
052        log.info("Running resource: {}", resourcePath);
053        InputStreamReader reader =
054                new InputStreamReader(getClass().getResourceAsStream(resourcePath), StandardCharsets.UTF_8);
055        return engine.eval(reader);
056    }
057}
