001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonParser;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.JsonToken;
006import com.fasterxml.jackson.databind.DeserializationContext;
007import com.fasterxml.jackson.databind.JsonDeserializer;
008import org.bitcoinj.core.Coin;
009
010import java.io.IOException;
011import java.math.BigDecimal;
012
013/**
014 *
015 */
016public class CoinDeserializer extends JsonDeserializer<Coin> {
017    @Override
018    public Coin deserialize(JsonParser p, DeserializationContext ctxt) throws IOException, JsonProcessingException {
019        JsonToken token = p.getCurrentToken();
020        switch (token) {
021
022            case VALUE_NUMBER_FLOAT:
023                BigDecimal bd = p.getDecimalValue();
024                return BitcoinMath.btcToCoin(bd);
025
026            case VALUE_NUMBER_INT:
027                long val = p.getNumberValue().longValue(); // should be optimal, whatever it is
028                return Coin.valueOf(val);
029
030            default:
031                return (Coin) ctxt.handleUnexpectedToken(Coin.class, p);
032        }
033    }
034}
