001package com.msgilligan.bitcoinj.json.conversion;
002
003import org.bitcoinj.core.Coin;
004
005import java.math.BigDecimal;
006import java.math.MathContext;
007import java.math.RoundingMode;
008
009/**
010 * Utilities for Bitcoin Math
011 */
012public class BitcoinMath {
013    public static final MathContext DEFAULT_CONTEXT = new MathContext(0, RoundingMode.UNNECESSARY);
014    public static final int DEFAULT_SCALE = Coin.SMALLEST_UNIT_EXPONENT;
015    public static final BigDecimal satoshiPerCoinDecimal = new BigDecimal(Coin.COIN.value, DEFAULT_CONTEXT);
016
017    /**
018     * Convert from BTC `BigDecimal` value to satoshi `long`.
019     *
020     * @param btc Bitcoin amount in BTC units
021     * @return number of satoshi (long)
022     */
023    public static long btcToSatoshi(final BigDecimal btc) {
024        BigDecimal satoshisDecimal = btc.multiply(satoshiPerCoinDecimal);
025        return satoshisDecimal.longValueExact();
026    }
027
028    /**
029     * Convert from satoshi `long` to BTC `BigDecimal`.
030     *
031     * @param satoshi number of satoshi (long)
032     * @return Bitcoin amount in BTC units
033     */
034    public static BigDecimal satoshiToBtc(final long satoshi) {
035        BigDecimal bdSatoshi = new BigDecimal(satoshi, BitcoinMath.DEFAULT_CONTEXT);
036        return bdSatoshi.divide(satoshiPerCoinDecimal, DEFAULT_SCALE, RoundingMode.UNNECESSARY);
037    }
038
039    /**
040     * Convert from BTC `BigDecimal` value to `Coin` type.
041     *
042     * @param btc Bitcoin amount in BTC units
043     * @return bitcoinj `Coin` type (uses satoshi unit internally)
044     */
045    public static Coin btcToCoin(final BigDecimal btc) {
046        return Coin.valueOf(btcToSatoshi(btc));
047    }
048
049    /**
050     * Convert from `Coin` type to BTC `BigDecimal` value .
051     *
052     * @param coin Coin value to convert to BTC
053     * @return  Bitcoin amount in BTC units
054     */
055    public static BigDecimal coinToBTC(final Coin coin) {
056         return satoshiToBtc(coin.value);
057    }
058}
