001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.JsonSerializer;
006import com.fasterxml.jackson.databind.SerializerProvider;
007
008import java.io.IOException;
009import java.util.regex.Pattern;
010
011/**
012 * If string is an integer (valid long) serialize as Number, else serialize as String
013 */
014public class NumberStringSerializer extends JsonSerializer<String> {
015    private static Pattern numberRegEx = Pattern.compile("^-?\\d+$");
016
017    @Override
018    public void serialize(String numberOrString, JsonGenerator jsonGenerator,
019                          SerializerProvider serializerProvider)
020            throws IOException, JsonProcessingException {
021
022        if (numberRegEx.matcher(numberOrString).matches()) {
023            try {
024                jsonGenerator.writeNumber(Long.valueOf(numberOrString));
025                return;
026            } catch (NumberFormatException ignored) {}
027        }
028        jsonGenerator.writeString(numberOrString);
029    }
030}
