001package com.msgilligan.bitcoinj.json.conversion;
002
003import java.util.Formatter;
004
005/**
006 * Hex string to hex conversion utility class.
007 */
008public class HexUtil {
009    /**
010     * Converts a hex-encoded string into a byte array.
011     *
012     * @param s A string to convert
013     * @return The byte array
014     */
015    public static byte[] hexStringToByteArray(String s) {
016        int len = s.length();
017        byte[] data = new byte[len / 2];
018        for (int i = 0; i < len; i += 2) {
019            data[i / 2] = (byte) ((Character.digit(s.charAt(i), 16) << 4)
020                    + Character.digit(s.charAt(i + 1), 16));
021        }
022        return data;
023    }
024
025    // From: http://bitcoin.stackexchange.com/questions/8475/how-to-get-hex-string-from-transaction-in-bitcoinj
026    public static String bytesToHexString(byte[] bytes) {
027        final StringBuilder sb = new StringBuilder();
028        Formatter formatter = new Formatter(sb);
029        for (byte b : bytes) {
030            formatter.format("%02x", b);
031        }
032        formatter.close();
033        return sb.toString();
034    }
035}
