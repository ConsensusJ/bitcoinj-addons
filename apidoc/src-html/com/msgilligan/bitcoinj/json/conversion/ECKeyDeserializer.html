001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonParser;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.JsonToken;
006import com.fasterxml.jackson.databind.DeserializationContext;
007import com.fasterxml.jackson.databind.JsonDeserializer;
008import com.fasterxml.jackson.databind.exc.InvalidFormatException;
009import org.bitcoinj.core.Address;
010import org.bitcoinj.core.AddressFormatException;
011import org.bitcoinj.core.DumpedPrivateKey;
012import org.bitcoinj.core.ECKey;
013
014import java.io.IOException;
015
016/**
017 *
018 */
019public class ECKeyDeserializer extends JsonDeserializer<ECKey> {
020    @Override
021    public ECKey deserialize(JsonParser p, DeserializationContext ctxt) throws IOException, JsonProcessingException {
022        JsonToken token = p.getCurrentToken();
023        switch (token) {
024            case VALUE_STRING:
025                try {
026                    return DumpedPrivateKey.fromBase58(null, p.getValueAsString()).getKey();
027                } catch (AddressFormatException e) {
028                    throw new InvalidFormatException(p, "Invalid Key", p.getValueAsString(), ECKey.class);
029                }
030            default:
031                return (ECKey) ctxt.handleUnexpectedToken(ECKey.class, p);
032        }
033    }
034}
