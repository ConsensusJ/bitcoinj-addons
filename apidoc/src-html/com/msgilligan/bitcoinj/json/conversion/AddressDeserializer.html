001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonParser;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.JsonToken;
006import com.fasterxml.jackson.databind.DeserializationContext;
007import com.fasterxml.jackson.databind.JsonDeserializer;
008import com.fasterxml.jackson.databind.exc.InvalidFormatException;
009import org.bitcoinj.core.Address;
010import org.bitcoinj.core.AddressFormatException;
011import org.bitcoinj.core.NetworkParameters;
012import org.bitcoinj.core.Sha256Hash;
013
014import java.io.IOException;
015
016/**
017 *
018 */
019public class AddressDeserializer extends JsonDeserializer<Address> {
020    private NetworkParameters netParams;
021
022    public AddressDeserializer(NetworkParameters netParams) {
023        this.netParams = netParams;
024    }
025
026    @Override
027    public Address deserialize(JsonParser p, DeserializationContext ctxt) throws IOException, JsonProcessingException {
028        JsonToken token = p.getCurrentToken();
029        switch (token) {
030            case VALUE_STRING:
031                try {
032                    return Address.fromBase58(netParams, p.getValueAsString());
033                } catch (AddressFormatException e) {
034                    throw new InvalidFormatException(p, "Invalid Address", p.getValueAsString(), Address.class);
035                }
036            default:
037                return (Address) ctxt.handleUnexpectedToken(Address.class, p);
038        }
039    }
040}
