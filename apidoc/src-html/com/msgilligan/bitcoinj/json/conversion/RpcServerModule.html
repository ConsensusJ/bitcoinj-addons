001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.Version;
004import com.fasterxml.jackson.databind.module.SimpleModule;
005import org.bitcoinj.core.Address;
006import org.bitcoinj.core.Coin;
007import org.bitcoinj.core.ECKey;
008import org.bitcoinj.core.NetworkParameters;
009import org.bitcoinj.core.Peer;
010import org.bitcoinj.core.Sha256Hash;
011import org.bitcoinj.core.Transaction;
012
013/**
014 *
015 */
016public class RpcServerModule extends SimpleModule {
017    public RpcServerModule(NetworkParameters netParams) {
018        super("BitcoinJMappingServer", new Version(1, 0, 0, null, null, null));
019
020        this.addDeserializer(Address.class, new AddressDeserializer(netParams))  // Null means use default list of netParams
021                .addDeserializer(Coin.class, new CoinDeserializer())
022                .addDeserializer(ECKey.class, new ECKeyDeserializer())
023                .addDeserializer(Sha256Hash.class, new Sha256HashDeserializer())
024                .addSerializer(Address.class, new AddressSerializer())
025                .addSerializer(Coin.class, new CoinSerializer())
026                .addSerializer(ECKey.class, new ECKeySerializer())
027                .addSerializer(Peer.class, new PeerSerializer())
028                .addSerializer(Sha256Hash.class, new Sha256HashSerializer())
029                .addSerializer(Transaction.class, new TransactionSerializer());
030    }
031}
