001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.Version;
004import com.fasterxml.jackson.databind.module.SimpleModule;
005import org.bitcoinj.core.Address;
006import org.bitcoinj.core.Coin;
007import org.bitcoinj.core.ECKey;
008import org.bitcoinj.core.Peer;
009import org.bitcoinj.core.Sha256Hash;
010import org.bitcoinj.core.Transaction;
011
012/**
013 *
014 */
015public class RpcServerModule extends SimpleModule {
016    public RpcServerModule() {
017        super("BitcoinJMappingServer", new Version(1, 0, 0, null, null, null));
018
019        this.addDeserializer(Address.class, new AddressDeserializer(null))  // Null means use default list of netParams
020                .addDeserializer(Coin.class, new CoinDeserializer())
021                .addDeserializer(ECKey.class, new ECKeyDeserializer())
022                .addDeserializer(Sha256Hash.class, new Sha256HashDeserializer())
023                .addSerializer(Address.class, new AddressSerializer())
024                .addSerializer(Coin.class, new CoinSerializer())
025                .addSerializer(ECKey.class, new ECKeySerializer())
026                .addSerializer(Peer.class, new PeerSerializer())
027                .addSerializer(Sha256Hash.class, new Sha256HashSerializer())
028                .addSerializer(Transaction.class, new TransactionSerializer());
029    }
030}
