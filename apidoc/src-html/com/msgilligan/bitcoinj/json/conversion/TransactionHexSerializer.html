001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonGenerator;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.databind.JsonSerializer;
006import com.fasterxml.jackson.databind.SerializerProvider;
007import org.bitcoinj.core.Transaction;
008
009import java.io.IOException;
010import java.util.Formatter;
011
012/**
013 *
014 */
015public class TransactionHexSerializer extends JsonSerializer<Transaction> {
016    @Override
017    public void serialize(Transaction value, JsonGenerator gen, SerializerProvider serializers) throws IOException, JsonProcessingException {
018        gen.writeString(bytesToHexString(value.bitcoinSerialize()));
019    }
020
021    // From: http://bitcoin.stackexchange.com/questions/8475/how-to-get-hex-string-from-transaction-in-bitcoinj
022    public static String bytesToHexString(byte[] bytes) {
023        final StringBuilder sb = new StringBuilder();
024        Formatter formatter = new Formatter(sb);
025        for (byte b : bytes) {
026            formatter.format("%02x", b);
027        }
028        formatter.close();
029        return sb.toString();
030    }
031}
