001package com.msgilligan.bitcoinj.json.conversion;
002
003import com.fasterxml.jackson.core.JsonParser;
004import com.fasterxml.jackson.core.JsonProcessingException;
005import com.fasterxml.jackson.core.JsonToken;
006import com.fasterxml.jackson.databind.DeserializationContext;
007import com.fasterxml.jackson.databind.JsonDeserializer;
008import org.bitcoinj.core.Coin;
009import org.bitcoinj.core.Sha256Hash;
010
011import java.io.IOException;
012import java.math.BigDecimal;
013
014/**
015 *
016 */
017public class Sha256HashDeserializer  extends JsonDeserializer<Sha256Hash> {
018    @Override
019    public Sha256Hash deserialize(JsonParser p, DeserializationContext ctxt) throws IOException, JsonProcessingException {
020        JsonToken token = p.getCurrentToken();
021        switch (token) {
022            case VALUE_STRING:
023                return Sha256Hash.wrap(p.getValueAsString());
024            default:
025                return (Sha256Hash) ctxt.handleUnexpectedToken(Sha256Hash.class, p);
026        }
027    }
028}
