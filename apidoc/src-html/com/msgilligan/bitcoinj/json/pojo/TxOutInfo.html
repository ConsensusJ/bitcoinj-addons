001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Coin;
007import org.bitcoinj.core.Sha256Hash;
008
009import java.util.Map;
010
011/**
012 * Result of `gettxout`
013 */
014@JsonIgnoreProperties(ignoreUnknown = true)
015public class TxOutInfo {
016    private final Sha256Hash bestblock;
017    private final int        confirmations;
018    private final Coin       value;
019    private final Map        scriptPubKey;
020    private final int        version;
021    private final boolean    coinbase;
022
023    @JsonCreator
024    public TxOutInfo(@JsonProperty("bestblock")     Sha256Hash  bestblock,
025                     @JsonProperty("confirmations") int         confirmations,
026                     @JsonProperty("value")         Coin        value,
027                     @JsonProperty("scriptPubKey")  Map         scriptPubKey,
028                     @JsonProperty("version")       int         version,
029                     @JsonProperty("coinbase")      boolean     coinbase) {
030        this.bestblock = bestblock;
031        this.confirmations = confirmations;
032        this.value = value;
033        this.scriptPubKey = scriptPubKey;
034        this.version = version;
035        this.coinbase = coinbase;
036    }
037
038    public Sha256Hash getBestblock() {
039        return bestblock;
040    }
041
042    public int getConfirmations() {
043        return confirmations;
044    }
045
046    public Coin getValue() {
047        return value;
048    }
049
050    public Map getScriptPubKey() {
051        return scriptPubKey;
052    }
053
054    public int getVersion() {
055        return version;
056    }
057
058    public boolean isCoinbase() {
059        return coinbase;
060    }
061}
