001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
004import org.bitcoinj.core.Address;
005import org.bitcoinj.core.Coin;
006import org.bitcoinj.core.NetworkParameters;
007
008import java.util.List;
009
010/**
011 * For listaddressgroupings response
012 * Note: In the JSON response this is actually an array
013 */
014@JsonIgnoreProperties(ignoreUnknown = true)
015public class AddressGroupingItem {
016    private final Address address;
017    private final Coin balance;
018    private final String account;
019
020    public AddressGroupingItem(Address address, Coin balance, String account) {
021        this.address = address;
022        this.balance = balance;
023        this.account = account;
024    }
025
026    public AddressGroupingItem(List<Object> addressItem, NetworkParameters netParams) {
027        String addressStr = (String) addressItem.get(0);
028        //TODO: Try to avoid using Double
029        Double balanceDouble = (Double) addressItem.get(1);
030        account = (addressItem.size() > 2) ? (String) addressItem.get(2) : null;
031        address = Address.fromString(netParams, addressStr);
032        balance = Coin.valueOf(((Double)(balanceDouble * 100000000.0)).longValue());
033
034    }
035
036    public Address getAddress() {
037        return address;
038    }
039
040    public Coin getBalance() {
041        return balance;
042    }
043
044    public String getAccount() {
045        return account;
046    }
047}
