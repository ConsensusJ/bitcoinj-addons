001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
004import com.fasterxml.jackson.annotation.JsonProperty;
005import org.bitcoinj.core.Sha256Hash;
006
007import java.math.BigDecimal;
008import java.util.ArrayList;
009import java.util.List;
010
011/**
012 * BlockInfo POJO returned by GetBlockInfo
013 */
014@JsonIgnoreProperties(ignoreUnknown=true)
015// "strippedsize" property added (present in Bitcoin 0.13)
016public class BlockInfo {
017    public final Sha256Hash hash;
018    public final int confirmations;
019    public final int size;
020    public final int height;
021    public final int version;
022    public final Sha256Hash merkleroot;
023    public final Sha256HashList tx;
024    public final int time;
025    public final long nonce;
026    public final String bits;
027    public final BigDecimal difficulty;
028    public final String chainwork;
029    public final Sha256Hash previousblockhash;
030    public final Sha256Hash nextblockhash;
031
032    public BlockInfo(@JsonProperty("hash")              Sha256Hash hash,
033                     @JsonProperty("confirmations")     int confirmations,
034                     @JsonProperty("size")              int size,
035                     @JsonProperty("height")            int height,
036                     @JsonProperty("version")           int version,
037                     @JsonProperty("merkleroot")        Sha256Hash merkleroot,
038                     @JsonProperty("tx")                Sha256HashList tx,
039                     @JsonProperty("time")              int time,
040                     @JsonProperty("nonce")             long nonce,
041                     @JsonProperty("bits")              String bits,
042                     @JsonProperty("difficulty")        BigDecimal difficulty,
043                     @JsonProperty("chainwork")         String chainwork,
044                     @JsonProperty("previousblockhash") Sha256Hash previousblockhash,
045                     @JsonProperty("nextblockhash")     Sha256Hash nextblockhash) {
046        this.hash = hash;
047        this.confirmations = confirmations;
048        this.size = size;
049        this.height = height;
050        this.version = version;
051        this.merkleroot = merkleroot;
052        this.tx = tx;
053        this.time = time;
054        this.nonce = nonce;
055        this.bits = bits;
056        this.difficulty = difficulty;
057        this.chainwork = chainwork;
058        this.previousblockhash = previousblockhash;
059        this.nextblockhash = nextblockhash;
060    }
061
062    public static class Sha256HashList extends ArrayList<Sha256Hash> {
063    }
064
065}
