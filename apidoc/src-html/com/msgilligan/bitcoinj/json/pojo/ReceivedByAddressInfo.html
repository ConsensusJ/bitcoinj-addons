001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Address;
007import org.bitcoinj.core.Coin;
008import org.bitcoinj.core.Sha256Hash;
009
010import java.util.List;
011
012/**
013 *
014 */
015@JsonIgnoreProperties(ignoreUnknown = true)
016public class ReceivedByAddressInfo {
017    public final Address address;
018    public final String account;
019    public final Coin amount;
020    public final int confirmations;
021    public final List<Sha256Hash> txids;
022    public final String label;
023
024    @JsonCreator
025    public ReceivedByAddressInfo(@JsonProperty("address") Address address,
026                                 @JsonProperty("account") String account,
027                                 @JsonProperty("amount") Coin amount,
028                                 @JsonProperty("confirmations") int confirmations,
029                                 @JsonProperty("txids") List<Sha256Hash> txids,
030                                 @JsonProperty("label") String label) {
031        this.address = address;
032        this.account = account;
033        this.amount = amount;
034        this.confirmations = confirmations;
035        this.txids = txids;
036        this.label = label;
037    }
038
039    public Address getAddress() {
040        return address;
041    }
042
043    public String getAccount() {
044        return account;
045    }
046
047    public Coin getAmount() {
048        return amount;
049    }
050
051    public int getConfirmations() {
052        return confirmations;
053    }
054
055    public List<Sha256Hash> getTxids() {
056        return txids;
057    }
058}
