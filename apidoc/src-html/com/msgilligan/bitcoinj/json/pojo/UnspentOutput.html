001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Address;
007import org.bitcoinj.core.Coin;
008import org.bitcoinj.core.Sha256Hash;
009
010/**
011 * Data class for UnspentOutput as returned by listUnspent RPC
012 * Because the class is immutable we have to give Jackson some hints via annotations.
013 */
014@JsonIgnoreProperties(ignoreUnknown = true)
015public class UnspentOutput {
016    private final Sha256Hash  txid;
017    private final int         vout;
018    private final Address     address;
019    private final String      account;
020    private final String      scriptPubKey;
021    private final Coin        amount;
022    private final int         confirmations;
023    private final boolean     spendable;
024    private final Boolean     solvable;
025
026    @JsonCreator
027    public UnspentOutput(@JsonProperty("txid")          Sha256Hash  txid,
028                         @JsonProperty("vout")          int         vout,
029                         @JsonProperty("address")       Address     address,
030                         @JsonProperty("account")       String      account,
031                         @JsonProperty("scriptPubKey")  String      scriptPubKey,
032                         @JsonProperty("amount")        Coin        amount,
033                         @JsonProperty("confirmations") int         confirmations,
034                         @JsonProperty("spendable")     boolean     spendable,
035                         @JsonProperty("solvable")      Boolean     solvable) {
036        this.txid = txid;
037        this.vout = vout;
038        this.address = address;
039        this.account = account;
040        this.scriptPubKey = scriptPubKey;
041        this.amount = amount;
042        this.confirmations = confirmations;
043        this.spendable = spendable;
044        this.solvable = solvable;
045    }
046
047    public Sha256Hash getTxid() {
048        return txid;
049    }
050
051    public int getVout() {
052        return vout;
053    }
054
055    public Address getAddress() {
056        return address;
057    }
058
059    public String getAccount() {
060        return account;
061    }
062
063    public String getScriptPubKey() {
064        return scriptPubKey;
065    }
066
067    public Coin getAmount() {
068        return amount;
069    }
070
071    public int getConfirmations() {
072        return confirmations;
073    }
074
075    public boolean getSpendable() { return spendable; }
076}
