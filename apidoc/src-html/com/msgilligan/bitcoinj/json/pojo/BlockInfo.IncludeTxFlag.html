001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Sha256Hash;
007
008import java.math.BigDecimal;
009import java.util.ArrayList;
010import java.util.List;
011
012/**
013 * BlockInfo POJO returned by GetBlockInfo
014 */
015@JsonIgnoreProperties(ignoreUnknown=true)
016// "strippedsize" property added (present in Bitcoin 0.13)
017public class BlockInfo {
018    public enum IncludeTxFlag {
019        NO,         // Don't include transactions
020        IDONLY,     // Only include the Transaction ID/hash
021        YES         // Include transactions (in the format of getrawtransaction RPC)
022    }
023    public final Sha256Hash hash;
024    public final int confirmations;
025    public final int size;
026    public final int height;
027    public final int version;
028    public final Sha256Hash merkleroot;
029    private final int nTx;
030    public final Sha256HashList tx;
031    public final int time;
032    public final long nonce;
033    public final String bits;
034    public final BigDecimal difficulty;
035    public final String chainwork;
036    public final Sha256Hash previousblockhash;
037    public final Sha256Hash nextblockhash;
038
039    public BlockInfo(@JsonProperty("hash")              Sha256Hash hash,
040                     @JsonProperty("confirmations")     int confirmations,
041                     @JsonProperty("size")              int size,
042                     @JsonProperty("height")            int height,
043                     @JsonProperty("version")           int version,
044                     @JsonProperty("merkleroot")        Sha256Hash merkleroot,
045                     @JsonProperty("nTx")               int nTx,
046                     @JsonProperty("tx")                Sha256HashList tx,
047                     @JsonProperty("time")              int time,
048                     @JsonProperty("nonce")             long nonce,
049                     @JsonProperty("bits")              String bits,
050                     @JsonProperty("difficulty")        BigDecimal difficulty,
051                     @JsonProperty("chainwork")         String chainwork,
052                     @JsonProperty("previousblockhash") Sha256Hash previousblockhash,
053                     @JsonProperty("nextblockhash")     Sha256Hash nextblockhash) {
054        this.hash = hash;
055        this.confirmations = confirmations;
056        this.size = size;
057        this.height = height;
058        this.version = version;
059        this.merkleroot = merkleroot;
060        this.nTx = nTx;
061        this.tx = tx;
062        this.time = time;
063        this.nonce = nonce;
064        this.bits = bits;
065        this.difficulty = difficulty;
066        this.chainwork = chainwork;
067        this.previousblockhash = previousblockhash;
068        this.nextblockhash = nextblockhash;
069    }
070
071    public static class Sha256HashList extends ArrayList<Sha256Hash> {
072        @JsonCreator
073        public Sha256HashList(List<Sha256Hash> collection) {
074            super(collection);
075        }
076    }
077
078    public Sha256Hash getHash() {
079        return hash;
080    }
081
082    public int getConfirmations() {
083        return confirmations;
084    }
085
086    public int getSize() {
087        return size;
088    }
089
090    public int getHeight() {
091        return height;
092    }
093
094    public int getVersion() {
095        return version;
096    }
097
098    public Sha256Hash getMerkleroot() {
099        return merkleroot;
100    }
101
102    public int getNTx() {
103        return nTx;
104    }
105
106    public Sha256HashList getTx() {
107        return tx;
108    }
109
110    public int getTime() {
111        return time;
112    }
113
114    public long getNonce() {
115        return nonce;
116    }
117
118    public String getBits() {
119        return bits;
120    }
121
122    public BigDecimal getDifficulty() {
123        return difficulty;
124    }
125
126    public String getChainwork() {
127        return chainwork;
128    }
129
130    public Sha256Hash getPreviousblockhash() {
131        return previousblockhash;
132    }
133
134    public Sha256Hash getNextblockhash() {
135        return nextblockhash;
136    }
137}
