001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
004import org.bitcoinj.core.Sha256Hash;
005
006/**
007 * Data class for Outpoint as used by RPC methods
008 */
009@JsonIgnoreProperties(ignoreUnknown = true)
010public class Outpoint {
011    private final Sha256Hash txid;
012    private final int        vout;
013
014    public Outpoint(Sha256Hash txid, int vout) {
015        this.txid = txid;
016        this.vout = vout;
017    }
018
019    public Sha256Hash getTxid() {
020        return txid;
021    }
022
023    public int getVout() {
024        return vout;
025    }
026
027    public String toString() {
028        return "{ " + txid + ", " + vout + " }";
029    }
030}
