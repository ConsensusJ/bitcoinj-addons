001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Address;
007import org.bitcoinj.core.Coin;
008import org.bitcoinj.core.Sha256Hash;
009
010import java.util.ArrayList;
011import java.util.List;
012
013/**
014 * Detailed information about an in-wallet transaction (from gettransaction RPC)
015 */
016@JsonIgnoreProperties(ignoreUnknown = true)
017public class WalletTransactionInfo {
018    private final Coin amount;
019    private final Coin fee;
020    private final int confirmations;
021    private final Sha256Hash blockhash;
022    private final int blockindex;
023    private final int blocktime;
024    private final Sha256Hash txid;
025    private final WalletConflictList walletconflicts;
026    private final int time;
027    private final int timereceived;
028    private final String bip125Replaceable;
029    private final DetailList details;
030    private final String hex;
031
032    @JsonCreator
033    public WalletTransactionInfo(@JsonProperty("amount")            Coin amount,
034                                 @JsonProperty("fee")               Coin fee,
035                                 @JsonProperty("confirmations")     int confirmations,
036                                 @JsonProperty("blockhash")         Sha256Hash blockhash,
037                                 @JsonProperty("blockindex")        int blockindex,
038                                 @JsonProperty("blocktime")         int blocktime,
039                                 @JsonProperty("txid")              Sha256Hash txid,
040                                 @JsonProperty("walletconflicts")   WalletConflictList walletconflicts,
041                                 @JsonProperty("time")              int time,
042                                 @JsonProperty("timereceived")      int timereceived,
043                                 @JsonProperty("bip125-replaceable") String bip125Replaceable,
044                                 @JsonProperty("details")           DetailList details,
045                                 @JsonProperty("hex")               String hex) {
046        this.amount = amount;
047        this.fee = fee;
048        this.confirmations = confirmations;
049        this.blockhash = blockhash;
050        this.blockindex = blockindex;
051        this.blocktime = blocktime;
052        this.txid = txid;
053        this.walletconflicts = walletconflicts;
054        this.time = time;
055        this.timereceived = timereceived;
056        this.bip125Replaceable = bip125Replaceable;
057        this.details = details;
058        this.hex = hex;
059    }
060
061    public Coin getAmount() {
062        return amount;
063    }
064
065    public Coin getFee() {
066        return fee;
067    }
068
069    public int getConfirmations() {
070        return confirmations;
071    }
072
073    public Sha256Hash getBlockhash() {
074        return blockhash;
075    }
076
077    public int getBlockindex() {
078        return blockindex;
079    }
080
081    public int getBlocktime() {
082        return blocktime;
083    }
084
085    public Sha256Hash getTxid() {
086        return txid;
087    }
088
089    public List<Sha256Hash> getWalletconflicts() {
090        return walletconflicts;
091    }
092
093    public int getTime() {
094        return time;
095    }
096
097    public int getTimereceived() {
098        return timereceived;
099    }
100
101    public List<Detail> getDetails() {
102        return details;
103    }
104
105    public String getHex() {
106        return hex;
107    }
108
109    public static class WalletConflictList extends ArrayList<Sha256Hash> {
110    }
111
112    public static class DetailList extends ArrayList<Detail> {
113    }
114
115    @JsonIgnoreProperties(ignoreUnknown = true)
116    public static class Detail {
117        private final String account;
118        private final Address address;
119        private final String category;
120        private final Coin amount;
121        private final String label;
122        private final int vout;
123        private final Coin fee;
124
125        public Detail(@JsonProperty("account")  String account,
126                      @JsonProperty("address")  Address address,
127                      @JsonProperty("category") String category,
128                      @JsonProperty("amount")   Coin amount,
129                      @JsonProperty("label")    String label,
130                      @JsonProperty("vout")     int vout,
131                      @JsonProperty("fee")      Coin fee) {
132            this.account = account;
133            this.address = address;
134            this.category = category;
135            this.amount = amount;
136            this.label = label;
137            this.vout = vout;
138            this.fee = fee;
139        }
140
141        public String getAccount() {
142            return account;
143        }
144
145        public Address getAddress() {
146            return address;
147        }
148
149        public String getCategory() {
150            return category;
151        }
152
153        public Coin getAmount() {
154            return amount;
155        }
156
157        public int getVout() {
158            return vout;
159        }
160
161        public Coin getFee() {
162            return fee;
163        }
164    }
165}
