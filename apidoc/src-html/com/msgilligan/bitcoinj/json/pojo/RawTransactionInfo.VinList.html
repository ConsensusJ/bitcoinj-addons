001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import com.msgilligan.bitcoinj.json.conversion.HexUtil;
007import org.bitcoinj.core.Coin;
008import org.bitcoinj.core.Sha256Hash;
009import org.bitcoinj.core.Transaction;
010import org.bitcoinj.core.TransactionInput;
011import org.bitcoinj.core.TransactionOutput;
012
013import java.util.ArrayList;
014import java.util.List;
015
016/**
017 * RawTransaction POJO
018 */
019@JsonIgnoreProperties(ignoreUnknown=true)
020// "hash" property added (present in Bitcoin 0.13)
021public class RawTransactionInfo {
022    public final String hex;
023    public final Sha256Hash txid;
024    public final long version;
025    public final long locktime;
026    public final VinList vin;
027    public final VoutList vout;
028    public final Sha256Hash blockhash;
029    public final int confirmations;
030    public final long time;
031    public final long blocktime;
032
033    @JsonCreator
034    public RawTransactionInfo(@JsonProperty("hex")              String hex,
035                              @JsonProperty("txid")             Sha256Hash txid,
036                              @JsonProperty("version")          long version,
037                              @JsonProperty("locktime")         long locktime,
038                              @JsonProperty("vin")              VinList vin,
039                              @JsonProperty("vout")             VoutList vout,
040                              @JsonProperty("blockhash")        Sha256Hash blockhash,
041                              @JsonProperty("confirmations")    int confirmations,
042                              @JsonProperty("time")             long time,
043                              @JsonProperty("blocktime")        long blocktime) {
044        this.hex = hex;
045        this.txid = txid;
046        this.version = version;
047        this.locktime = locktime;
048        this.vin = vin;
049        this.vout = vout;
050        this.blockhash = blockhash;
051        this.confirmations = confirmations;
052        this.time = time;
053        this.blocktime = blocktime;
054    }
055
056    /**
057     * Construct from a bitcoinj transaction
058     * @param transaction A bitcoinj confirmed or unconfirmed transaction
059     */
060    public RawTransactionInfo(Transaction transaction) {
061        this.hex = HexUtil.bytesToHexString(transaction.bitcoinSerialize());
062        this.txid = transaction.getTxId();
063        this.version = transaction.getVersion();
064        this.locktime = transaction.getLockTime();
065        this.blockhash = null;  // For now
066        this.confirmations = transaction.getConfidence().getDepthInBlocks();
067        this.time = 0; // TODO: block header time of block including transaction
068        this.blocktime = this.time; // same as time (see API doc)
069        vin = new VinList();
070        for (TransactionInput input : transaction.getInputs()) {
071            vin.add(new Vin(txid,
072                            input.getOutpoint().getIndex(),
073                            input.getScriptSig().toString(),
074                            input.getSequenceNumber()));
075        }
076        vout = new VoutList();
077        for (TransactionOutput output : transaction.getOutputs()) {
078            vout.add(new Vout(output.getValue(),
079                                output.getIndex(),
080                                output.getScriptPubKey().toString()));
081        }
082    }
083
084    public String getHex() {
085        return hex;
086    }
087
088    public Sha256Hash getTxid() {
089        return txid;
090    }
091
092    public long getVersion() {
093        return version;
094    }
095
096    public long getLocktime() {
097        return locktime;
098    }
099
100    public List<Vin> getVin() {
101        return vin;
102    }
103
104    public List<Vout> getVout() {
105        return vout;
106    }
107
108    public Sha256Hash getBlockhash() {
109        return blockhash;
110    }
111
112    public int getConfirmations() {
113        return confirmations;
114    }
115
116    public long getTime() {
117        return time;
118    }
119
120    public long getBlocktime() {
121        return blocktime;
122    }
123
124    public static class VinList extends ArrayList<Vin> {
125    }
126
127    public static class VoutList extends ArrayList<Vout> {
128    }
129
130    public static class Vin {
131        public final Sha256Hash txid;
132        public final  long vout;
133        public final  Object scriptSig;
134        public final  long sequence;
135
136        public Vin(@JsonProperty("txid") Sha256Hash txid,
137                   @JsonProperty("vout") long vout,
138                   @JsonProperty("scriptSig") Object scriptSig,
139                   @JsonProperty("sequence") long sequence) {
140            this.txid = txid;
141            this.vout = vout;
142            this.scriptSig = scriptSig;
143            this.sequence = sequence;
144        }
145
146        public Sha256Hash getTxid() {
147            return txid;
148        }
149
150        public long getVout() {
151            return vout;
152        }
153
154        public Object getScriptSig() {
155            return scriptSig;
156        }
157
158        public long getSequence() {
159            return sequence;
160        }
161    }
162
163    public static class Vout {
164        public final  Coin value;
165        public final  int n;
166        public final  Object scriptPubKey;
167
168        public Vout(@JsonProperty("value")          Coin value,
169                    @JsonProperty("n")              int n,
170                    @JsonProperty("scriptPubKey")   Object scriptPubKey) {
171            this.value = value;
172            this.n = n;
173            this.scriptPubKey = scriptPubKey;
174        }
175
176        public Coin getValue() {
177            return value;
178        }
179
180        public int getN() {
181            return n;
182        }
183
184        public Object getScriptPubKey() {
185            return scriptPubKey;
186        }
187    }
188}
