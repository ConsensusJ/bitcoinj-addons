001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006import org.bitcoinj.core.Sha256Hash;
007
008import java.math.BigDecimal;
009
010/**
011 * POJO for `getblockchaininfo` RPC response.
012 */
013@JsonIgnoreProperties(ignoreUnknown = true)
014public class BlockChainInfo {
015    private final String      chain;
016    private final int         blocks;
017    private final int         headers;
018    private final Sha256Hash  bestBlockHash;
019    private final BigDecimal  difficulty;
020    private final BigDecimal  verificationProgress;
021    private final byte[]      chainWork;
022
023    @JsonCreator
024    public BlockChainInfo(@JsonProperty("chain")            String chain,
025                          @JsonProperty("blocks")           int blocks,
026                          @JsonProperty("headers")          int headers,
027                          @JsonProperty("bestblockhash")        Sha256Hash bestBlockHash,
028                          @JsonProperty("difficulty")           BigDecimal difficulty,
029                          @JsonProperty("verificationProgress") BigDecimal verificationProgress,
030                          @JsonProperty("chainWork")            byte[] chainWork) {
031        this.chain = chain;
032        this.blocks = blocks;
033        this.headers = headers;
034        this.bestBlockHash = bestBlockHash;
035        this.difficulty = difficulty;
036        this.verificationProgress = verificationProgress;
037        this.chainWork = chainWork;
038    }
039
040    public String getChain() {
041        return chain;
042    }
043
044    public int getBlocks() {
045        return blocks;
046    }
047
048    public int getHeaders() {
049        return headers;
050    }
051
052    public Sha256Hash getBestBlockHash() {
053        return bestBlockHash;
054    }
055
056    public BigDecimal getDifficulty() {
057        return difficulty;
058    }
059
060    public BigDecimal getVerificationProgress() {
061        return verificationProgress;
062    }
063
064    public byte[] getChainWork() {
065        return chainWork;
066    }
067}
