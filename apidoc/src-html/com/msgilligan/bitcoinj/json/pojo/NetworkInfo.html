001package com.msgilligan.bitcoinj.json.pojo;
002
003import com.fasterxml.jackson.annotation.JsonCreator;
004import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
005import com.fasterxml.jackson.annotation.JsonProperty;
006
007/**
008 * POJO for `getnetworkinfo` RPC response.
009 * Warning: `network` and `address` will be upgraded to POJOs in the future
010 */
011@JsonIgnoreProperties(ignoreUnknown = true)
012public class NetworkInfo {
013    private final int version;
014    private final String subVersion;
015    private final int protocolVersion;
016    private final int timeOffset;
017    private final int connections;
018    private final String proxy;
019    private final int relayFee;
020    private final byte[] localServices;
021    private final Object[]  network;
022    private final Object[]  address;
023
024    @JsonCreator
025    public NetworkInfo(@JsonProperty("version")         int version,
026                       @JsonProperty("subversion")      String subVersion,
027                       @JsonProperty("protocolversion") int protocolVersion,
028                       @JsonProperty("timeOffset")      int timeOffset,
029                       @JsonProperty("connections")     int connections,
030                       @JsonProperty("proxy")           String proxy,
031                       @JsonProperty("relayFee")        int relayFee,
032                       @JsonProperty("localServices")   byte[] localServices,
033                       @JsonProperty("network")         Object[] network,
034                       @JsonProperty("address")         Object[] address) {
035        this.version = version;
036        this.subVersion = subVersion;
037        this.protocolVersion = protocolVersion;
038        this.timeOffset = timeOffset;
039        this.connections = connections;
040        this.proxy = proxy;
041        this.relayFee = relayFee;
042        this.localServices = localServices;
043        this.network = network;
044        this.address = address;
045    }
046
047    public int getVersion() {
048        return version;
049    }
050
051    public String getSubVersion() {
052        return subVersion;
053    }
054
055    public int getProtocolVersion() {
056        return protocolVersion;
057    }
058
059    public int getTimeOffset() {
060        return timeOffset;
061    }
062
063    public int getConnections() {
064        return connections;
065    }
066
067    public String getProxy() {
068        return proxy;
069    }
070
071    public int getRelayFee() {
072        return relayFee;
073    }
074
075    public byte[] getLocalServices() {
076        return localServices;
077    }
078
079    public Object[] getNetwork() {
080        return network;
081    }
082
083    public Object[] getAddress() {
084        return address;
085    }
086}
