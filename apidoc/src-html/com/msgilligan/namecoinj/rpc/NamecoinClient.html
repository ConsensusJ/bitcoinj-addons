001package com.msgilligan.namecoinj.rpc;
002
003import com.msgilligan.bitcoinj.rpc.BitcoinClient;
004import org.consensusj.jsonrpc.JsonRPCStatusException;
005import com.msgilligan.bitcoinj.rpc.RPCConfig;
006import com.msgilligan.bitcoinj.rpc.bitcoind.AppDataDirectory;
007import com.msgilligan.bitcoinj.rpc.bitcoind.BitcoinConfFile;
008import com.msgilligan.namecoinj.core.NMCMainNetParams;
009import com.msgilligan.namecoinj.json.pojo.NameData;
010import org.bitcoinj.core.NetworkParameters;
011
012import java.io.File;
013import java.io.IOException;
014import java.net.URI;
015
016/**
017 * Namecoin RPC client
018 * See: https://wiki.namecoin.org/index.php?title=Client_API
019 */
020public class NamecoinClient extends BitcoinClient {
021    public NamecoinClient(NetworkParameters netParams, URI server, String rpcuser, String rpcpassword) {
022        super(netParams, server, rpcuser, rpcpassword);
023    }
024
025    public NamecoinClient(RPCConfig config) {
026        this(config.getNetParams(), config.getURI(), config.getUsername(), config.getPassword());
027    }
028
029    /**
030     * name_show
031     *
032     * @param identifier namespace/name, e.g. 'd/beelin'
033     * @return Data object
034     * @throws IOException
035     * @throws JsonRPCStatusException
036     */
037    public NameData nameShow(String identifier) throws IOException, JsonRPCStatusException {
038        return send("name_show", NameData.class, identifier);
039    }
040
041    /**
042     * Read Namecoin conf file. Namecoin conf file uses same format as bitcoin.conf
043     * @return object containing RPC endpoint information
044     */
045    public static RPCConfig readConfig() {
046        File file = new File(AppDataDirectory.forAppName("Namecoin"), "namecoin.conf");
047        BitcoinConfFile conf = new BitcoinConfFile(file);
048        RPCConfig config = conf.readWithFallback().getRPCConfig();
049        // Since config is immutable we have to make a new one with NameCoin parameters
050        config = new RPCConfig(NMCMainNetParams.get(),
051                config.getURI(),
052                config.getUsername(),
053                config.getPassword());
054        return config;
055    }
056
057}
