001package com.msgilligan.namecoinj.cli;
002
003import org.consensusj.jsonrpc.JsonRPCStatusException;
004import com.msgilligan.namecoinj.json.pojo.NameData;
005import com.msgilligan.namecoinj.rpc.NamecoinClient;
006import org.bitcoinj.core.Address;
007import org.bitcoinj.core.AddressFormatException;
008
009import java.io.IOException;
010import java.net.URISyntaxException;
011
012/**
013 * Simple command-line tool to do Namecoin name lookup
014 */
015public class NmcLookup {
016
017    public static void main(String[] args) throws URISyntaxException, IOException, JsonRPCStatusException, AddressFormatException {
018        NamecoinClient client = new NamecoinClient(NamecoinClient.readConfig());
019        NameData result = client.nameShow("d/beelin");
020        System.out.println(result.getValue());
021        Address owningAddress = result.getAddress();
022        System.out.println("Owning Address: " + owningAddress);
023    }
024}
