001package com.msgilligan.ethereum.rpc;
002
003import org.consensusj.jsonrpc.JsonRPCStatusException;
004import org.consensusj.jsonrpc.JsonRpcRequest;
005import org.consensusj.jsonrpc.RPCClient;
006
007import java.io.IOException;
008import java.math.BigInteger;
009import java.net.URI;
010import java.net.URISyntaxException;
011import java.util.List;
012
013/**
014 * A partial implementation of an Ethereum RPC Client
015 *
016 * How to mine just a little for "reg test mode" in Eth:
017 * https://github.com/ethereum/go-ethereum/wiki/bitchin-tricks
018 * See also:
019 * https://github.com/ethereum/go-ethereum/wiki/Management-APIs
020 */
021public class EthereumClient extends RPCClient {
022    public static URI DEFAULT_LOCALHOST;
023
024    static {
025        try {
026            DEFAULT_LOCALHOST = new URI("http://localhost:8545");
027        } catch (URISyntaxException e) {
028            DEFAULT_LOCALHOST = null;
029        }
030    }
031
032    /**
033     * Construct a JSON-RPC client from URI, username, and password
034     *
035     * @param server      server URI should not contain username/password
036     * @param rpcuser     username for the RPC HTTP connection
037     * @param rpcpassword password for the RPC HTTP connection
038     */
039    public EthereumClient(URI server, String rpcuser, String rpcpassword) {
040        super(server, rpcuser, rpcpassword);
041    }
042
043    public EthereumClient() {
044        super(DEFAULT_LOCALHOST, null, null);
045    }
046
047    /**
048     * Override to send JSON RPC version "2.0"
049     *
050     */
051    @Override
052    protected JsonRpcRequest buildJsonRequest(String method, List<Object> params) {
053        return new JsonRpcRequest(method, params, JsonRpcRequest.JSON_RPC_VERSION_2);
054    }
055
056    public String ethProtocolVersion() throws IOException, JsonRPCStatusException {
057        return this.send("eth_protocolVersion");
058    }
059
060    public long ethBlockNumber() throws IOException, JsonRPCStatusException {
061        String blockNumString = this.send("eth_blockNumber");
062        long blockNum = Long.decode(blockNumString);
063        return blockNum;
064    }
065
066    public BigInteger ethGetBalance(String address, String block) throws IOException, JsonRPCStatusException {
067        String weiAsHexString = this.send("eth_getBalance", address, block);
068        return quantityToInt(weiAsHexString);
069    }
070
071    public String ethCall(EthTxCallObject callObject, String block) throws IOException, JsonRPCStatusException {
072        String data = this.send("eth_call", callObject, block);
073        return data;
074    }
075
076    public String web3ClientVersion() throws IOException, JsonRPCStatusException {
077        return this.send("web3_clientVersion");
078    }
079
080    /**
081     * Returns Keccak-256 (not the standardized SHA3-256) of the given data
082     * @param dataToHash
083     * @return Keccak-256 hash of the data
084     * @throws IOException
085     * @throws JsonRPCStatusException
086     */
087    public String web3Sha3(String dataToHash) throws IOException, JsonRPCStatusException {
088        return this.send("web3_sha3", dataToHash);
089    }
090
091    public boolean minerStart(int numberOfThreads) throws IOException, JsonRPCStatusException {
092        return this.send("miner_start", "0x" + Integer.toHexString(numberOfThreads));
093    }
094
095    public boolean minerStop() throws IOException, JsonRPCStatusException {
096        return this.send("miner_stop");
097    }
098
099    private BigInteger quantityToInt(String weiAsHexString) {
100        return new BigInteger(weiAsHexString.substring(2), 16);
101    }
102}
